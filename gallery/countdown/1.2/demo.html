6 notify：cd.notify(xx, fn); 当倒计时还剩xx毫秒时调用fn

剩余时间：${h}时${m}分${s}秒
    HTML:
        <div class="cd cd-large cd-slide" data-total="4000">剩余时间：<span class="clock">${h}时${m}分${s}秒</span></div>

    JavaScript:
        KISSY.use("gallery/countdown/1.2/index, gallery/countdown/1.2/assets/countdown.css",function(S, Countdown){
            S.all('#demo6 .cd').each(function(node) {
                Countdown(node, {effect: 'slide'}).notify(0, function() {
                    this.left = 3000;
                });
            });
        });

7 大于24小时配置

    JavaScript:
        KISSY.use("gallery/countdown/1.2/index, gallery/countdown/1.2/assets/countdown.css",function(S, Countdown){
            S.all('#demo7 .cd').each(function(node) {
                Countdown(node, {
                    effect: 'slide',
                    _clock: ['d', 100, 2, 'h', 100, 2, 'm', 60, 2, 's', 60, 2, 'u', 10, 1]
                });
            });
        });
