1  //监听加减按钮click事件，以及出价框keyup事件
  2             function registerEvent() {
  3                 /*
  4                  * 监听click事件，实现数值加减功能
  5                  * */
  6                 E.on(ELE_SIGN, 'click', function(e) {
  7                     if (D.hasClass(ELE_SIGN, 'sleepLink')) {
  8                         return;
  9                     } //通过sleepLink属性来判断是否启用加减按钮
 10                     e.halt(true);
 11                     var trigger = S.one(e.target),
 12                         inputEl = D.children(D.parent(trigger), 'input')[0],
 13                         inputValue = formatPrice(inputEl.value),
 14                         range = formatPrice(D.attr(inputEl, 'data-range')),
 15                         min = formatPrice(D.attr(inputEl, 'data-min')),
 16                         //如果是荷兰拍，需要取得最大值
 17                         max = parseInt(D.attr(inputEl, 'data-max'), 10);
 18                     if (trigger.hasClass('plus') || trigger.hasClass('plus-sign')) {
 19                         inputValue += +range;
 20                     } else if (trigger.hasClass('minus') || trigger.hasClass('minus-sign')) {
 21                         inputValue -= +range;
 22                     }
 23 
 24                     inputValue = inputValue >= min ? inputValue : min;
 25                     inputValue = Math.min(9999999.99, inputValue);
 26                     //荷兰拍限定购买数量
 27                     if (max) {
 28                         inputValue = inputValue >= max ? max : inputValue;
 29                         D.val(inputEl, inputValue.toFixed(0));
 30                         D.hide(CON_ERROR);//click会自动调整输入值，所以可以不用显示提示。
 31                         D.removeAttr(ELE_BUTTON, 'disabled');//同上，click时，按钮可用
 32                         return;
 33                     }
 34                     D.val(inputEl, inputValue.toFixed(2));
 35                 }),
 36 
 37                     /*
 38                      * 监听数量框keyup事件，实现输入框内容的格式化
 39                      * */
 40                     E.on(ELE_QUANTITY_INPUT, 'keyup', function(e) {
 41                         var self = D.get(this),
 42                             min = formatPrice(D.attr(self, 'data-min')),
 43                             max = formatQuantity(D.attr(self, 'data-max')),
 44                             inputValue = formatPrice(D.val(self).replace(/[^\d\.]/g, ''));
 45                         //荷兰拍
 46                         if (max) {
 47                             if (!quantityValidation(D.val(self),max)) {
 48                                 D.show(CON_ERROR);
 49                                 D.get(ELE_BUTTON).disabled = true;
 50                                 return;
 51                             }
 52                             else{
 53                                 D.hide(CON_ERROR);
 54                                 D.removeAttr(ELE_BUTTON, 'disabled');
 55                                 D.val(self, inputValue.toFixed(0));
 56                                 return;
 57                             }
 58                         }
 59 
 60                     });
 61 
 62                     /*
 63                     * 监听价格框blur事件，实现输入框内容的格式化
 64                     * */
 65                     E.on(ELE_PRICE_INPUT, 'blur', function() {
 66                         var self = D.get(this),
 67                             min = formatPrice(D.attr(self, 'data-min')),
 68                             inputValue = formatPrice(D.val(self).replace(/[^\d\.]/g, ''));
 69                         inputValue = isNaN(inputValue) ? min : Math.max(min, inputValue),
 70                             inputValue = Math.min(9999999.99, inputValue);
 71 
 72                         D.val(self, inputValue.toFixed(2));
 73                     });
 74             }
 75 
 76             return{
 77                 init: registerEvent
 78             }
 79         }();
