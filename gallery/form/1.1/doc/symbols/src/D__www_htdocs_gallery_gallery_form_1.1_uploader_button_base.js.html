1 /**
  2  * @fileoverview 文件上传按钮base
  3  * @author: 紫英(橘子)<daxingplay@gmail.com>, 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/button/base',function(S, Node, Base) {
  6     var EMPTY = '',
  7         LOG_PREFIX = '[Uploader-Button] ',
  8         $ = Node.all;
  9     /**
 10      * @name Button
 11      * @class 文件上传按钮，ajax和iframe上传方式使用
 12      * @constructor
 13      * @extends Base
 14      * @param {String} target *，目标元素
 15      * @param {Object} config 配置对象
 16      * @param {String} config.name  *，隐藏的表单上传域的name值
 17      * @param {Boolean} config.disabled 是否禁用按钮
 18      * @param {Boolean} config.multiple 是否开启多选支持
 19      */
 20     function Button(target, config) {
 21         var self = this;
 22         config = S.merge({target:$(target)}, config);
 23         //超类初始化
 24         Button.superclass.constructor.call(self, config);
 25     }
 26 
 27     S.mix(Button, {
 28         //支持的事件
 29         event : {
 30             'beforeShow': 'beforeShow',
 31             'afterShow': 'afterShow',
 32             'beforeHide': 'beforeHide',
 33             'afterHide': 'afterHide',
 34             'beforeRender' : 'beforeRender',
 35             'afterRender' : 'afterRender',
 36             'CHANGE' : 'change'
 37         },
 38         /**
 39          * 获取文件名称（从表单域的值中提取）
 40          * @param {String} path 文件路径
 41          * @return {String}
 42          */
 43         getFileName : function(path) {
 44             return path.replace(/.*(\/|\\)/, "");
 45         }
 46     });
 47 
 48     S.extend(Button, Base, /** @lends Button.prototype*/{
 49         /**
 50          * 运行
 51          * @return {Button} Button的实例
 52          */
 53         render : function() {
 54             var self = this,
 55                 target = self.get('target'),
 56                 render = self.fire(Button.event.beforeRender);
 57             if (render === false) {
 58                 S.log(LOG_PREFIX + 'button render was prevented.');
 59                 return false;
 60             } else {
 61                 if (target == null) {
 62                     S.log(LOG_PREFIX + 'Cannot find target!');
 63                     return false;
 64                 }
 65                 self._createInput();
 66                 self._setDisabled(self.get('disabled'));
 67                 self._setMultiple(self.get('multiple'));
 68                 self.fire(Button.event.afterRender);
 69                 return self;
 70             }
 71         },
 72         /**
 73          * 显示按钮
 74          * @return {Button} Button的实例
 75          */
 76         show : function() {
 77             var self = this, target = self.get('target');
 78             target.show();
 79             self.fire(Button.event.afterShow);
 80             return Button;
 81         },
 82         /**
 83          * 隐藏按钮
 84          * @return {Button} Button的实例
 85          */
 86         hide : function() {
 87             var self = this, target = self.get('target');
 88             target.hide();
 89             self.fire(Button.event.afterHide);
 90             return Button;
 91         },
 92         /**
 93          * 重置按钮
 94          * @return {Button} Button的实例
 95          */
 96         reset : function() {
 97             var self = this,
 98                 inputContainer = self.get('inputContainer');
 99             //移除表单上传域容器
100             $(inputContainer).remove();
101             self.set('inputContainer', EMPTY);
102             self.set('fileInput', EMPTY);
103             //重新创建表单上传域
104             self._createInput();
105             return self;
106         },
107         /**
108          * 创建隐藏的表单上传域
109          * @return {HTMLElement} 文件上传域容器
110          */
111         _createInput : function() {
112             var self = this,
113                 target = self.get('target'),
114                 name = self.get('name'),
115                 tpl = self.get('tpl'),
116                 html,
117                 inputContainer,
118                 fileInput;
119             if (!S.isString(name) || !S.isString(tpl)) {
120                 S.log(LOG_PREFIX + 'No name or tpl specified.');
121                 return false;
122             }
123             html = S.substitute(tpl, {
124                 'name' : name
125             });
126             // TODO: inputContainer = DOM.create(html);
127             inputContainer = $(html);
128             //向body添加表单文件上传域
129             $(inputContainer).appendTo(target);
130             fileInput = $(inputContainer).children('input');
131             //上传框的值改变后触发
132             $(fileInput).on('change', self._changeHandler, self);
133             //DOM.hide(fileInput);
134             self.set('fileInput', fileInput);
135             self.set('inputContainer', inputContainer);
136             // self.resetContainerCss();
137             return inputContainer;
138         },
139         /**
140          * 文件上传域的值改变时触发
141          * @param {Object} ev 事件对象
142          */
143         _changeHandler : function(ev) {
144             var self = this,
145                 fileInput = self.get('fileInput'),
146                 value = $(fileInput).val(),
147                 //IE取不到files
148                 oFiles = ev.target.files,files = [];
149             if (value == EMPTY) {
150                 S.log(LOG_PREFIX + 'No file selected.');
151                 return false;
152             }
153             if(oFiles){
154                 S.each(oFiles,function(v){
155                     if(S.isObject(v)){
156                         files.push({'name' : v.name,'type' : v.type,'size' : v.size,data:v});
157                     }
158                 });
159             }else{
160                 files.push({'name' : Button.getFileName(value)});
161             }
162             self.fire(Button.event.CHANGE, {
163                 files: files,
164                 input: fileInput.getDOMNode()
165             });
166             self.reset();
167         },
168         /**
169          * 设置上传组件的禁用
170          * @param {Boolean} disabled 是否禁用
171          * @return {Boolean}
172          */
173         _setDisabled : function(disabled){
174             var self = this,
175                 cls = self.get('cls'),disabledCls = cls.disabled,
176                 $target = self.get('target'),
177                 input = self.get('fileInput');
178             if(!$target.length || !S.isBoolean(disabled)) return false;
179             if(!disabled){
180                 $target.removeClass(disabledCls);
181                 $(input).show();
182             }else{
183                 $target.addClass(disabledCls);
184                 $(input).hide();
185             }
186             return disabled;
187         },
188         /**
189          * 设置上传组件的禁用
190          * @param {Boolean} multiple 是否禁用
191          * @return {Boolean}
192          */
193         _setMultiple : function(multiple){
194             var self = this,fileInput = self.get('fileInput');
195             if(!fileInput.length) return false;
196             multiple && fileInput.attr('multiple','multiple') || fileInput.removeAttr('multiple');
197             return multiple;
198         }
199     }, {
200         ATTRS : /** @lends Button.prototype */{
201             /**
202              * 按钮目标元素
203              * @type KISSY.Node
204              * @default null
205              */
206             target: {
207                 value: null
208             },
209             /**
210              * 对应的表单上传域
211              * @type KISSY.Node
212              * @default ""
213              */
214             fileInput: {
215                 value: EMPTY
216             },
217             /**
218              * 文件上传域容器
219              * @type KISSY.Node
220              * @default ""
221              */
222             inputContainer: {
223                 value: EMPTY
224             },
225             /**
226              * 隐藏的表单上传域的模板
227              * @type String
228              */
229             tpl : {
230                 value : '<div class="file-input-wrapper"><input type="file" name="{name}" hidefoucs="true" class="file-input" /></div>'
231             },
232             /**
233              * 隐藏的表单上传域的name值
234              * @type String
235              * @default "fileInput"
236              */
237             name : {
238                 value : 'fileInput',
239                 setter : function(v) {
240                     if (this.get('fileInput')) {
241                         $(this.get('fileInput')).attr('name', v);
242                     }
243                     return v;
244                 }
245             },
246             /**
247              * 是否可用,false为可用
248              * @type Boolean
249              * @default false
250              */
251             disabled : {
252                 value : false,
253                 setter : function(v) {
254                     this._setDisabled(v);
255                     return v;
256                 }
257             },
258             /**
259              * 是否开启多选支持，多选目前有兼容性问题，建议禁用
260              * @type Boolean
261              * @default false
262              */
263             multiple : {
264                 value : false,
265                 setter : function(v){
266                     this._setMultiple(v);
267                     return v;
268                 }
269             },
270             /**
271              * 样式
272              * @type Object
273              * @default  { disabled : 'uploader-button-disabled' }
274              */
275             cls : {
276                 value : {
277                     disabled : 'uploader-button-disabled'
278                 }
279             }
280         }
281     });
282 
283     return Button;
284 
285 }, {
286     requires:[
287         'node',
288         'base'
289     ]
290 });
