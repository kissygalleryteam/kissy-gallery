1 /**
  2  * @fileoverview 运行文件上传组件
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/index',function (S, Base, Node, Uploader, Button,SwfButton,Auth) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploaderRender]:',
  7         dataName = {
  8             CONFIG:'data-config',
  9             BUTTON_CONFIG : 'data-button-config',
 10             THEME_CONFIG : 'data-theme-config',
 11             AUTH : 'data-auth'
 12         },
 13         THEME_PREFIX='gallery/form/1.1/uploader/themes/';
 14     /**
 15      * 解析组件在页面中data-config成为组件的配置
 16      * @param {String} hook 组件钩子
 17      * @param {String} dataConfigName 配置名
 18      * @return {Object}
 19      */
 20     S.parseConfig = function(hook, dataConfigName) {
 21         var config = {}, sConfig, DATA_CONFIG = dataConfigName || dataName.CONFIG;
 22         sConfig = $(hook).attr(DATA_CONFIG);
 23         if (!S.isString(sConfig)) return {};
 24         try {
 25             config = S.JSON.parse(sConfig);
 26         } catch (err) {
 27             S.log(LOG_PREFIX + '请检查' + hook + '上' + DATA_CONFIG + '属性内的json格式是否符合规范！');
 28         }
 29         return config;
 30     };
 31 
 32     /**
 33      * @name RenderUploader
 34      * @class 异步文件上传入口文件，会从按钮的data-config='{}' 伪属性中抓取组件配置
 35      * @constructor
 36      * @param {String | HTMLElement} buttonTarget *，上传按钮目标元素
 37      * @param {String | HTMLElement} queueTarget *，文件队列目标元素
 38      * @param {Object} config 配置，该配置好覆盖data-config伪属性中的数据
 39      * @requires Uploader,Button,SwfButton,Auth
 40      * @example
 41      * <a id="J_UploaderBtn" class="uploader-button" data-config=
 42      '{"type" : "auto",
 43      "serverConfig":{"action":"upload.php"},
 44      "name":"Filedata",
 45      "urlsInputName":"fileUrls"}'
 46      href="#">
 47      选择要上传的文件
 48      </a>
 49      <ul id="J_UploaderQueue">
 50 
 51      </ul>
 52      * @example
 53      *
 54 KISSY.use('gallery/form/1.1/uploader/index', function (S, RenderUploader) {
 55      var ru = new RenderUploader('#J_UploaderBtn', '#J_UploaderQueue');
 56      ru.on("init", function (ev) {
 57         var uploader = ev.uploader;
 58      })
 59 })
 60      */
 61     function RenderUploader(buttonTarget, queueTarget, config) {
 62         var self = this;
 63         //合并配置
 64         config = S.mix(S.parseConfig(buttonTarget), config);
 65         //超类初始化
 66         RenderUploader.superclass.constructor.call(self, config);
 67         self.set('buttonTarget', buttonTarget);
 68         self.set('queueTarget', queueTarget);
 69         self.set('uploaderConfig', config);
 70         self._init();
 71     }
 72     /**
 73      * @name RenderUploader#init
 74      * @desc 上传组件完全初始化成功后触发，对uploader的操作务必先监听init事件
 75      * @event
 76      * @param {Uploader} ev.uploader   Uploader的实例
 77      */
 78 
 79     S.extend(RenderUploader, Base, /** @lends RenderUploader.prototype*/{
 80         /**
 81          * 初始化组件
 82          */
 83         _init:function () {
 84             var self = this, uploaderConfig = self.get('uploaderConfig'),
 85                 name = self.get('name'),
 86                 button = self._initButton(),
 87                 queue;
 88             self.set('button', button);
 89             self._initThemes(function (theme) {
 90                 queue = theme.get('queue');
 91                 S.mix(uploaderConfig.serverConfig,{'fileDataName':name});
 92                 //配置增加按钮实例和队列实例
 93                 S.mix(uploaderConfig, {button:button, queue:queue});
 94                 var uploader = new Uploader(uploaderConfig);
 95                 uploader.render();
 96                 self.set('uploader', uploader);
 97                 theme.set('uploader',uploader);
 98                 theme.set('button',button);
 99                 var auth = self._auth();
100                 theme.set('auth',auth);
101                 if(theme.afterUploaderRender) theme.afterUploaderRender(uploader);
102                 self.fire('init', {uploader:uploader});
103             });
104         },
105         /**
106          * 初始化模拟的上传按钮
107          * @return {Button}
108          */
109         _initButton:function () {
110             var self = this,
111                 target = self.get('buttonTarget'),
112                 //从html标签的伪属性中抓取配置
113                 config = S.parseConfig(target,dataName.BUTTON_CONFIG),
114                 name = self.get('name'),
115                 type = self.get('type');
116             //合并配置
117             config = S.merge({name:name},config);
118             //实例化上传按钮
119             return type != 'flash' && new Button(target, config) || new SwfButton(target);
120         },
121         /**
122          * 初始化主题
123          * @param {Function} callback 主题加载完成后的执行的回调函数
124          */
125         _initThemes:function (callback) {
126             var self = this, theme = self.get('theme'),
127                 target = self.get('buttonTarget'),
128                 //从html标签的伪属性中抓取配置
129                 config = S.parseConfig(target,dataName.THEME_CONFIG),
130                 reg=/\//;
131             //如果只是传递主题名，组件自行拼接
132             if(!reg.test(theme)){
133                 theme = THEME_PREFIX + theme;
134             }
135             theme = theme + '/index';
136             self.set('theme',theme);
137             S.use(theme, function (S, Theme) {
138                 var queueTarget = self.get('queueTarget'),
139                     theme;
140                 S.mix(config,{queueTarget:queueTarget});
141                 theme = new Theme(config);
142                 callback && callback.call(self, theme);
143             })
144         },
145         /**
146          * 文件上传验证
147          */
148         _auth:function () {
149             var self = this,buttonTarget = self.get('buttonTarget'),
150                 uploader = self.get('uploader'),
151                 rules, auth = EMPTY;
152             //存在验证配置
153             if($(buttonTarget).attr(dataName.AUTH)){
154                 rules = S.parseConfig(buttonTarget,dataName.AUTH);
155                 auth = new Auth(uploader,{rules : rules});
156                 uploader.set('auth',auth);
157             }
158             return auth;
159         }
160     }, {
161         ATTRS:/** @lends RenderUploader.prototype*/{
162             /**
163              * 主题引用路径
164              * @type String
165              * @default  “gallery/form/1.1/uploader/themes/default”
166              */
167             theme:{value:'gallery/form/1.1/uploader/themes/default' },
168             /**
169              * 按钮目标元素
170              * @type String|HTMLElement|KISSY.NodeList
171              * @default ""
172              */
173             buttonTarget:{value:EMPTY},
174             /**
175              * 队列目标元素
176              * @default ""
177              * @type String|HTMLElement|KISSY.Node
178              */
179             queueTarget:{value:EMPTY},
180             /**
181              * 上传组件配置
182              * @type Object
183              * @default {}
184              */
185             uploaderConfig:{},
186             /**
187              * Button（上传按钮）的实例
188              * @type Button
189              * @default ""
190              */
191             button:{value:EMPTY},
192             /**
193              * Queue（上传队列）的实例
194              * @type Queue
195              * @default ""
196              */
197             queue:{value:EMPTY},
198             /**
199              * 上传组件实例
200              * @type Uploader
201              * @default ""
202              */
203             uploader:{value:EMPTY},
204             /**
205              * 上传验证实例
206              * @type Auth
207              * @default ""
208              */
209             auth : {value:EMPTY}
210         }
211     });
212     return RenderUploader;
213 }, {requires:['base', 'node', './base', './button/base','./button/swfButton','./auth/base']});
