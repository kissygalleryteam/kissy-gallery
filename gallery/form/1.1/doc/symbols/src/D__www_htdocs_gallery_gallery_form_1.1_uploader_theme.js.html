  2  * @fileoverview 上传组件主题基类

  6 KISSY.add('gallery/form/1.1/uploader/theme', function (S, Node, Base, Queue) {
  7     var EMPTY = '', $ = Node.all;
  8 
  9     /**
 10      * @name Theme
 11      * @class 上传组件主题基类
 12      * @constructor
 13      * @extends Base
 14      * @requires Node
 15      */
 16     function Theme(config) {
 17         var self = this;
 18         //调用父类构造函数
 19         Theme.superclass.constructor.call(self, config);
 20         self._LoaderCss();
 21         self._init();
 22     }
 23 
 24     S.extend(Theme, Base, /** @lends Theme.prototype*/{
 25         /**
 26          * 初始化
 27          */
 28         _init:function () {
 29             var self = this,
 30                 queue = self._initQueue(),
 31                 name=self.get('name');
 32             if(name != EMPTY){
 33                 var $queueTarget = queue.get('target');
 34                 if($queueTarget.length){
 35                     $queueTarget.addClass(name+'-queue');
 36                 }
 37             }
 38         },
 39         /**
 40          * 初始化队列
 41          */
 42         _initQueue:function(){
 43             var self = this, queueTarget = self.get('queueTarget'), queue,
 44                 tpl = self.get('fileTpl'),
 45                 config = {tpl:tpl};
 46             //合并从伪属性中抓取的配置
 47             S.mix(config,S.parseConfig(queueTarget, 'data-queue-config'));
 48             queue = new Queue(queueTarget, config);
 49             queue.set('theme',self);
 50             self.set('queue', queue);
 51             queue.on('addData',function(ev){
 52                 queue.updateFile(ev.index,{
 53                     statusWrapper:self._getStatusWrapper(ev.file.target)
 54                 });
 55             });
 56             queue.on('add',function(ev){
 57                 self._addFileHandler(ev);
 58             });
 59             queue.on('statusChange',function(ev){
 60                 self._setStatusVisibility(ev);
 61             });
 62             return queue;
 63         },
 64         /**
 65          * 获取状态容器
 66          * @param {KISSY.NodeList} target 文件的对应的dom（一般是li元素）
 67          * @return {KISSY.NodeList}
 68          */
 69         _getStatusWrapper:function(target){
 70             return target.children('.J_FileStatus');
 71         },
 72         /**
 73          * 加载css文件
 74          */
 75         _LoaderCss:function () {
 76             var self = this,
 77                 isUseCss = self.get('isUseCss'),
 78                 cssUrl = self.get('cssUrl');
 79             //加载css文件
 80             if (isUseCss) {
 81                 S.use(cssUrl, function () {
 82                     S.log(cssUrl + '加载成功！');
 83                 }); }
 84         },
 85         /**
 86          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 87          * @param {Uploader} uploader
 88          */
 89         afterUploaderRender:function (uploader) {
 90 
 91         },
 92         /**
 93          * 添加完文件后触发的监听器
 94          */
 95         _addFileHandler:function(ev){
 96             var self = this,
 97                 queue = self.get('queue'),
 98                 uploader = ev.uploader,
 99                 index = ev.index,
100                 //文件id
101                 fileId = ev.file.id,
102                 //上传链接
103                 $upload = $('.J_Upload_' + fileId),
104                 //取消链接
105                 $cancel = $('.J_Cancel_' + fileId),
106                 //删除链接
107                 $del = $(".J_Del_"+fileId);
108             //点击上传
109             $upload.on('click',function(ev){
110                 ev.preventDefault();
111                 if (!S.isObject(uploader)) return false;
112                 uploader.upload(index);
113             });
114             //点击取消
115             $cancel.on('click', function(ev) {
116                 ev.preventDefault();
117                 uploader.cancel(index);
118             });
119             //点击删除
120             $del.on('click',function(){
121                 ev.preventDefault();
122                 //删除队列中的文件
123                 queue.remove(fileId);
124             }) ;
125         },
126         //设置状态层的可见性
127         _setStatusVisibility:function(ev){
128             var $statusWrapper = ev.file.statusWrapper,$status,
129                 file = ev.file,status = file.status;
130             if(!$statusWrapper.length){
131                 S.log('状态容器层不存在！');
132             }
133             $status = $statusWrapper.children('.status');
134             $status.hide();
135             $statusWrapper.children('.' + status + '-status').show();
136         },
137         /**
138          * 文件处于等待上传状态时触发
139          */
140         _waitingHandler:function(ev){
141 
142         },
143         /**
144          * 文件处于开始上传状态时触发
145          */
146         _startHandler : function(ev){
147 
148         },
149         /**
150          * 文件处于正在上传状态时触发
151          */
152         _progressHandler:function(ev){
153 
154         },
155         /**
156          * 文件处于上传成功状态时触发
157          */
158         _successHandler:function(ev){
159 
160         },
161         /**
162          * 文件处于上传错误状态时触发
163          */
164         _errorHandler:function(ev){
165 
166         },
167         /**
168          * 从路径隐藏域抓取文件，往队列添加文件后触发
169          * @param ev
170          */
171         _restoreHandler:function(ev){
172 
173         }
174     }, {ATTRS:/** @lends Theme.prototype*/{
175         /**
176          *  主题名
177          * @type String
178          * @default ""
179          */
180         name:{value:EMPTY},
181         /**
182          * 是否引用css文件
183          * @type Boolean
184          * @default true
185          */
186         isUseCss:{value:true},
187         /**
188          * css模块路径
189          * @type String
190          * @default ""
191          */
192         cssUrl:{value:EMPTY},
193         /**
194          * 队列使用的模板
195          * @type String
196          * @default ""
197          */
198         fileTpl:{value:EMPTY },
199         /**
200          * 队列目标元素（一般是ul），队列的实例化过程在Theme中
201          * @type String
202          * @default ""
203          */
204         queueTarget:{value:EMPTY},
205         /**
206          * Queue（上传队列）实例
207          * @type Queue
208          * @default ""
209          */
210         queue:{value:EMPTY},
211         /**
212          * Auth（上传验证）实例
213          * @type Auth
214          * @default ""
215          */
216         auth:{value:EMPTY}
217     }});
218     return Theme;
219 }, {requires:['node', 'base', './queue']});
