1 /**
  2  * @fileoverview 多选框美化
  3  * @author 伯方<bofang.zxj@taobao.com>
  4  *
  5  **/
  6 KISSY.add('gallery/form/1.1/checkbox/index', function(S, Base, Node) {
  7 	var $ = Node.all;
  8 	/**
  9 	 * @name Checkbox
 10 	 * @class 多选框美化
 11 	 * @constructor
 12 	 * @extends Base
 13 	 * @param {String} target 目标
 14 	 * @param {Object} config * ,组件配置
 15 	 * @param {Object} config.cls *，组件的样式
 16 	 * @example
 17 	 * var ck = new Checkbox('#J_Content input')
 18 	 */
 19 
 20 	function Checkbox(target, config) {
 21 		//调用父类构造器
 22 		var self = this;
 23 		Checkbox.superclass.constructor.call(self, config);
 24 		self.set('target', target);
 25 	}
 26 	//方法
 27 	S.extend(Checkbox, Base,/** @lends Checkbox.prototype*/ {
 28 		/**
 29 		 * 开始执行
 30 		 */
 31 		render: function() {
 32 			var self = this;
 33 			//加载css
 34 			self._loadCss();
 35 			//开始替换
 36 			self._replaceCheckbox();
 37 			//事件绑定
 38 			self._bindEvent();
 39 		},
 40 		/**
 41 		 * 还原checkbox为原生的checkbox
 42 		 * @return {Object} return self
 43 		 */
 44 		recoverCheckbox: function() {
 45 			var self = this,
 46 				targets = self.get('target'),
 47 				checkboxs = self.get('checkboxs');
 48 			$(checkboxs).each(function(value, key) {
 49 				value.hide();
 50 				$(targets[key]).show();
 51 			})
 52 			//self.set('checkboxs',[]);// = null;
 53 			return self;
 54 		},
 55 		/**
 56 		 * 用span替换checkbox，关键步骤
 57 		 */
 58 		_replaceCheckbox: function() {
 59 			var self = this,
 60 				target = self.get('target'),
 61 				html = self._getHtml(0),
 62 				disabledHTML = self._getHtml(2),
 63 				checkedHTML = self._getHtml(1),
 64 				checkbox, checkboxArr = [];
 65 			if (target.length === 0) {
 66 				return false;
 67 			}
 68 			target.each(function(value, key) {
 69 				value.hide();
 70 				if (self._isDisabled(value)) {
 71 					checkbox = $(disabledHTML).insertBefore(value).attr('ks-checkbox-disabled', 'disabled');
 72 				} else {
 73 					//如果本身是选中的状态
 74 					if (self._isChecked(value)) {
 75 						checkbox = $(checkedHTML).insertBefore(value);
 76 					} else {
 77 						checkbox = $(html).insertBefore(value);
 78 					}
 79 				}
 80 				checkboxArr.push(checkbox);
 81 			})
 82 			self.set('checkboxs', checkboxArr);
 83 		},
 84 		/**
 85 		 * 加载css
 86 		 */
 87 		_loadCss: function() {
 88 			var self = this,
 89 				isUseCss = self.get('isUseCss'),
 90 				cssUrl = self.get('cssUrl');
 91 			//加载css文件
 92 			if (isUseCss) {
 93 				S.use(cssUrl, function(S) {});
 94 			}
 95 		},
 96 		/**
 97 		 * 根据样式返回html字符串
 98 		 * @param  {Number} key 0→DEFAULT;1→CHECKED;2→DISABLED
 99 		 * @return {String} 返回html
100 		 */
101 		_getHtml: function(key) {
102 			var self = this,
103 				getClass = self.get('cls'),
104 				defaultClass = getClass.init,
105 				checkedClass = getClass.checked,
106 				disabledClass = getClass.disabled,
107 				htmlStr = '<span class="{defalutName} {secondName}"></span>',
108 				obj = {
109 					defalutName: defaultClass
110 				};
111 			switch (key) {
112 			case 0:
113 				obj.secondName = '';
114 				break;
115 			case 1:
116 				obj.secondName = checkedClass;
117 				break;
118 			case 2:
119 				obj.secondName = disabledClass;
120 				break;
121 			default:
122 				break;
123 			}
124 			return S.substitute(htmlStr, obj);
125 		},
126 		/**
127 		 * 绑定事件，包括mouseenter mouseleave click
128 		 */
129 		_bindEvent: function() {
130 			var self = this,
131 				checkboxs = $(self.get('checkboxs')),
132 				hoverClass = this.get('cls').hover;
133 			checkboxs.each(function(value, key) {
134 				value.on('mouseenter mouseleave', function(ev) {
135 					//如果本身是选中状态或者是禁用状态，则不做处理
136 					if (self._isChecked(value) || self._isDisabled(value)) {
137 						return;
138 					}
139 					//value.toggleClass('ks-checkbox-hover') 在初始化的时候就已经选中的无效
140 					switch (ev.type) {
141 					case 'mouseenter':
142 						value.addClass(hoverClass);
143 						break;
144 					case 'mouseleave':
145 						value.removeClass(hoverClass);
146 						break;
147 					default:
148 						break;
149 					}
150 				}).on('click', function() {
151 					if (self._isDisabled(value)) return;
152 					self._clickHandler.call(self, key);
153 					//return false;				
154 				})
155 			})
156 		},
157 		/**
158 		 * 单击事件
159 		 * @param  {Number} targetIndex 数组checkboxs的索引
160 		 */
161 		_clickHandler: function(targetIndex) {
162 			var that = this,
163 				targets = that.get('target'),
164 				checkbox = $(that.get('checkboxs')[targetIndex]),
165 				checkedClass = this.get('cls').checked;
166 			//触发原生dom节点的点击事件
167 			$(targets[targetIndex]).fire('click');
168 			checkbox.toggleClass(checkedClass);
169 		},
170 		/**
171 		 * 判断是否处于禁用状态
172 		 * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
173 		 * @return {Boolean}
174 		 */
175 		_isDisabled: function(target) {
176 			var protoDisabled = $(target).attr('disabled'),
177 				modifyDisabled = $(target).attr('ks-checkbox-disabled');
178 			return protoDisabled === 'disabled' || modifyDisabled === 'disabled';
179 		},
180 		/**
181 		 * 判断是否处于禁用状态
182 		 * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
183 		 * @return {Boolean}
184 		 */
185 		_isChecked: function(target) {
186 			var protoChecked = $(target).prop('checked'),
187 				hasCheckedClass = $(target).hasClass(this.get('cls').checked);
188 			return protoChecked || hasCheckedClass;
189 		},
190 		/**
191 		 * 设置某个checkbox为disabled状态
192 		 * @param {Number} targetElement 数组checkboxs的索引
193 		 */
194 		setDisabled: function(targetElement) {
195 			var self = this,
196 				checkboxs = self.get('checkboxs'),
197 				targets = self.get('target'),
198 				checkbox, target, getClass = this.get('cls'),
199 				checkedClass = getClass.checked,
200 				disabledClass = getClass.disabled,
201 				hoverClass = getClass.hover;
202 			//如果传递的是数字索引
203 			if (typeof targetElement === 'number') {
204 				checkbox = $(checkboxs[targetElement]);
205 				target = $(targets[targetElement]);
206 				checkbox.attr('ks-checkbox-disabled', 'disabled').removeClass(checkedClass + ' ' + hoverClass).addClass(disabledClass);
207 				target.attr('disabled', 'disabled');
208 			}
209 			return self;
210 		},
211 		/**
212 		 * 模拟的checkbox全选
213 		 * @return {Object} return self
214 		 */
215 		selectAll: function() {
216 			var self = this,
217 				checkboxs = self.get('checkboxs');
218 			$(checkboxs).each(function(value, key) {
219 				if (self._isChecked(value)) return;
220 				value.fire('click');
221 			})
222 			return self;
223 		},
224 		/**
225 		 * 清空模拟的checkbox
226 		 * @return {Object} return self
227 		 */
228 		resetAll: function() {
229 			var self = this,
230 				checkboxs = self.get('checkboxs'),
231 				hoverClass = self.get('cls').hover;
232 			$(checkboxs).each(function(value, key) {
233 				if (!self._isChecked(value)) return;
234 				value.fire('click').removeClass(hoverClass);
235 			})
236 			return self;
237 		},
238 		/**
239 		 * 获取所有选中的checkboxs索引
240 		 * @return {Array} 选中的checkboxs索引数组集合
241 		 */
242 		getAllChecked: function() {
243 			var self = this,
244 				target = self.get('target'),
245 				checkedArr = [],
246 				value;
247 			for (i = 0, len = target.length; i < len; i++) {
248 				value = $(target[i]);
249 				if (self._isDisabled(value)) {
250 					continue;
251 				}
252 				if (self._isChecked(value)) {
253 					checkedArr.push(i);
254 				}
255 			}
256 			return checkedArr;
257 		}
258 	}, {
259 		ATTRS: /** @lends Checkbox.prototype*/{
260 			/**
261 			 * 配置的目标,选择器的字符串
262 			 * @type {String}
263 			 */
264 			target: {
265 				value: '',
266 				setter: function(v) {
267 					return $(v);
268 				},
269 				getter: function(v) {
270 					return $(v);
271 				}
272 			},
273 			/**
274 			 * 美化后的checkbox数组
275 			 * @type {Array}
276 			 * @default []
277 			 */
278 			checkboxs: {
279 				value: []
280 			},
281 			/**
282 			 * 一组样式名
283 			 * @type {Object}
284 			 * @default cls:{init: 'ks-checkbox',checked: 'ks-checkbox-checked',disabled: 'ks-checkbox-disabled',hover: 'ks-checkbox-hover'}
285 			 */
286 
287 			cls: {
288 				value: {
289 					init: 'ks-checkbox',
290 					checked: 'ks-checkbox-checked',
291 					disabled: 'ks-checkbox-disabled',
292 					hover: 'ks-checkbox-hover'
293 				}
294 			},
295 			/**
296 			 * 是否引用css文件
297 			 */
298 			isUseCss: {
299 				value: true
300 			},
301 			/**
302 			 * css模块路径
303 			 */
304 			cssUrl: {
305 				value: 'gallery/form/1.1/checkbox/themes/default/style2.css'
306 			}
307 		}
308 	})
309 	return Checkbox;
310 }, {
311 	requires: ['base', 'node']
312 });
