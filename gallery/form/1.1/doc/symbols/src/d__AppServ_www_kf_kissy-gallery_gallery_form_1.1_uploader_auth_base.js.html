1 /**
  2  * @fileoverview 文件上传验证
  3  * @author: 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/auth/base', function (S, Node,Base) {
  6     var EMPTY = '', $ = Node.all,
  7         console = console || S, LOG_PREFIX = '[uploader-auth]:';
  8 
  9     /**
 10      * @name Auth
 11      * @class 文件上传验证，可以从按钮的data-auth伪属性抓取规则配置
 12      * @constructor
 13      * @extends Base
 14      * @param {Uploader} uploader *，上传组件实例
 15      * @param {Object} config 配置
 16      */
 17     function Auth(uploader, config) {
 18         var self = this;
 19         config = S.merge({uploader:uploader}, config);
 20         Auth.superclass.constructor.call(self, config);
 21         self._init();
 22     }
 23     S.mix(Auth,/** @lends Auth*/{
 24         /**
 25          * 事件
 26          */
 27         event : {
 28             ERROR : 'error'
 29         }
 30     });
 31     /**
 32      * @name Auth#error
 33      * @desc  当验证出错时触发
 34      * @event
 35      * {rule:'require',msg : rule[1],value : isRequire}
 36      * @param {String} ev.rule 规则名
 37      * @param {String} ev.msg 出错消息
 38      * @param {Boolean|String} ev.value 规则值
 39      */
 40     S.extend(Auth, Base, /** @lends Auth.prototype*/{
 41         /**
 42          * 初始化
 43          */
 44         _init:function () {
 45             var self = this, uploader = self.get('uploader'),
 46                 queue = uploader.get('queue');
 47             if (uploader == EMPTY) {
 48                 console.log(LOG_PREFIX + 'uploader不可以为空！');
 49                 return false;
 50             }
 51             self._setSwfButtonExt();
 52             queue.on('add',function(ev){
 53                 var file = ev.file;
 54                 self.testAllowExt(file);
 55                 self.testMaxSize(file);
 56                 self.testRepeat(file);
 57             });
 58             queue.on('remove',function(ev){
 59                 var file = ev.file,status = file.status;
 60                 //删除的是已经成功上传的文件，需要重新检验最大允许上传数
 61                 if(status == 'success'){
 62                     self.testMax();
 63                 }
 64             });
 65             uploader.on('success', function (ev) {
 66                 self.testMax();
 67             });
 68             uploader.on('error', function (ev) {
 69                 //允许继续上传文件
 70                 uploader.set('isAllowUpload', true);
 71             });
 72         },
 73         /**
 74          * 验证上传数、是否必须上传
 75          * @return {Boolean}
 76          */
 77         testAll : function(){
 78             var self = this;
 79             return self.testRequire() && self.testMax();
 80         },
 81         /**
 82          * 获取指定规则
 83          * @param {String} ruleName 规则名
 84          * @return {Array}
 85          */
 86         getRule : function(ruleName){
 87             var self = this,rules = self.get('rules');
 88             return rules[ruleName];
 89         },
 90         /**
 91          * 判断上传方式
 92          * @param type
 93          * @return {Boolean}
 94          */
 95         isUploaderType:function (type) {
 96             var self = this, uploader = self.get('uploader'),
 97                 uploaderType = uploader.get('type');
 98             return type == uploaderType;
 99         },
100         /**
101          * 检验是否必须上传一个文件
102          * @return {Boolean}
103          */
104         testRequire : function(){
105             var self = this,uploader = self.get('uploader'),
106                 urlsInput = uploader.get('urlsInput'),
107                 urls = urlsInput.get('urls'),
108                 rule = self.getRule('require'),
109                 isRequire = rule ? rule[0] : false,
110                 isHasUrls = urls.length > 0;
111             if(!isRequire) return true;
112             if(!isHasUrls){
113                 S.log(LOG_PREFIX + rule[1]);
114                 self.fire(Auth.event.ERROR,{rule:'require',msg : rule[1],value : isRequire});
115             }
116             return isHasUrls;
117         },
118         /**
119          * 测试是否是允许的文件上传类型
120          * @param {Object} file 文件对象
121          * @return {Boolean} 是否通过
122          */
123         testAllowExt:function (file) {
124             if (!S.isObject(file)) return false;
125             var self = this,
126                 fileName = file.name,
127                 allowExts = self.getRule('allowExts'),
128                 exts = [],
129                 fileExt, msg,
130                 isAllow;
131             if (!S.isArray(allowExts)) return false;
132             //扩展名数组
133             exts = self._getExts(allowExts[0].ext);
134 
135             isAllow = _isAllowUpload(fileName);
136             //如果不是支持的文件格式，出现错误
137             if(!isAllow){
138                 fileExt = _getFileExt(fileName);
139                 msg = S.substitute(allowExts[1],{ext : fileExt});
140                 self._stopUpload(file,msg);
141                 self.fire(Auth.event.ERROR,{rule:'allowExts',msg : msg,value : allowExts[0]});
142             }
143             /**
144              * 是否允许上传
145              * @param {String} fileName 文件名
146              * @return {Boolean}
147              */
148             function _isAllowUpload(fileName) {
149                 var isAllow = false, reg;
150                 S.each(exts, function (ext) {
151                     reg = new RegExp('^.+\.' + ext + '$');
152                     //存在该扩展名
153                     if (reg.test(fileName))  return isAllow = true;
154                 });
155                 return isAllow;
156             }
157             /**
158              * 获取文件扩展名
159              * @param {String} file
160              */
161             function _getFileExt(file){
162                 var arr = file.split('.');
163                 return arr[arr.length -1];
164             }
165             return isAllow;
166         },
167         /**
168          * 检验是否达到最大允许上传数
169          * @return {Boolean}
170          */
171         testMax:function () {
172             var self = this, uploader = self.get('uploader'),
173                 queue = uploader.get('queue'),
174                 successFiles = queue.getFiles('success'),
175                 len = successFiles.length,
176                 rule = self.getRule('max');
177             if(rule){
178             	var button = uploader.get('button'),
179 	                isPass = len < rule[0];
180 	            //达到最大允许上传数
181 	            if(!isPass){
182 	                //禁用按钮
183 	                button.set('disabled',true);
184 	                uploader.set('isAllowUpload', false);
185 	                self.fire(Auth.event.ERROR,{rule:'max',msg : rule[1],value : rule[0]});
186 	            }else{
187 	                button.set('disabled',false);
188 	                uploader.set('isAllowUpload', true);
189 	            }
190 	            return isPass;
191             }
192         },
193         /**
194          * 检验是否超过允许最大文件大小，留意iframe上传方式此验证无效
195          * @param {Object} file 文件对象
196          */
197         testMaxSize : function(file){
198             var self = this,
199                 size = file.size,
200                 rule = self.getRule('maxSize');
201             if(rule){
202             	var maxSize = Number(rule[0]) * 1000,
203 	                isAllow = size <= maxSize,
204 	                msg;
205 	            if(!isAllow){
206 	                msg = S.substitute(rule[1],{maxSize:S.convertByteSize(maxSize),size : file.textSize});
207 	                self._stopUpload(file,msg);
208 	                self.fire(Auth.event.ERROR,{rule:'maxSize',msg : msg,value : rule[0]});
209 	            }
210 	            return isAllow;
211             }
212         },
213         /**
214          * 检验文件是否重复（检验文件名，很有可能存在误差，比如不同目录下的相同文件名会被判定为同一文件）
215          * @param {Object} file 文件对象
216          * @return {Boolean}
217          */
218         testRepeat : function(file){
219             if(!S.isObject(file)) return false;
220             var self = this,
221                 fileName = file.name,
222                 rule = self.getRule('allowRepeat');
223             if(rule){
224             	var isAllowRepeat = rule[0],
225 	                msg = rule[1],
226 	                uploader = self.get('uploader'),
227 	                queue = uploader.get('queue'),
228 	                //上传成功的文件
229 	                files = queue.getFiles('success'),
230 	                isRepeat = false ;
231 	            //允许重复文件名，直接返回false
232 	            if(isAllowRepeat) return false;
233 	            S.each(files,function(f){
234 	                if(f.name == fileName){
235 	                    self._stopUpload(file,msg);
236 	                    self.fire(Auth.event.ERROR,{rule:'allowRepeat',msg : msg,value : rule[0]});
237 	                    return isRepeat = true;
238 	                }
239 	            });
240 	            return isRepeat;
241             }
242         },
243         /**
244          * 设置flash按钮的文件格式过滤
245          * @return {Auth}
246          */
247         _setSwfButtonExt:function () {
248             var self = this, uploader = self.get('uploader'),
249                 allowExts = self.getRule('allowExts'),
250                 button = uploader.get('button'),
251                 isFlashType = self.isUploaderType('flash');
252             if (!isFlashType || !S.isArray(allowExts)) return false;
253             //设置文件过滤
254             button.set('fileFilters', allowExts[0]);
255             return self;
256         },
257         /**
258          * 获取扩展名，需额外添加大写扩展名
259          * @param {String} sExt 扩展名字符串，类似*.jpg;*.jpeg;*.png;*.gif;*.bmp
260          * @retunr {Array}
261          */
262         _getExts:function (sExt) {
263             if (!S.isString(sExt)) return false;
264             var exts = sExt.split(';'),
265                 uppercaseExts = [],
266                 reg = /^\*\./;
267             S.each(exts, function (ext) {
268                 ext = ext.replace(reg, '');
269                 uppercaseExts.push(ext.toUpperCase());
270             });
271             S.each(uppercaseExts,function(ext){
272                 exts.push(ext);
273             });
274             return exts;
275         },
276         /**
277          * 阻止文件上传，并改变文件状态为error
278          * @param {Object} file 文件对象
279          * @param {String} msg 错误消息
280          */
281         _stopUpload:function (file,msg) {
282             if(!S.isString(msg)) msg = EMPTY;
283             var self = this, uploader = self.get('uploader'),
284                 queue = uploader.get('queue'),
285                 index = queue.getFileIndex(file.id);
286             //改变文件状态为error
287             queue.fileStatus(index, queue.constructor.status.ERROR, {msg:msg});
288         }
289     }, {ATTRS:/** @lends Auth.prototype*/{
290         /**
291          * 上传组件实例
292          * @type Uploader
293          * @default ""
294          */
295         uploader:{ value:EMPTY },
296         /**
297          * 上传验证规则，每个规则都是一个数组，数组第一个值为规则，第二个值为错误消息
298          * @type Object
299          * @default  { allowExts:[ {desc:"JPG,JPEG,PNG,GIF,BMP", ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"}, '不支持{ext}格式的文件上传！' ], require:[false, '必须至少上传一个文件！'], max:[3, '每次最多上传{max}个文件！'], maxSize:[1000, '文件大小为{size}，文件太大！'], allowRepeat:[false, '该文件已经存在！'] } }
300          *
301          */
302         rules:{
303             value : {
304                 /**
305                  * 允许上传的文件格式，如果是使用flash上传方式，在选择文件时就可以过滤格式
306                  */
307                 allowExts:[
308                     {desc:"JPG,JPEG,PNG,GIF,BMP", ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"},
309                     '不支持{ext}格式的文件上传！'
310                 ],
311                 /**
312                  * 是否必须上传个文件
313                  */
314                 require:[false, '必须至少上传一个文件！'],
315                 /**
316                  * 允许的最大上传数
317                  */
318                 max:[3, '每次最多上传{max}个文件！'],
319                 /**
320                  * 文件最大大小，单位为kb
321                  */
322                 maxSize:[1000, '文件大小为{size}，文件太大！'],
323                 /**
324                  * 允许重复上传相同文件
325                  */
326                 allowRepeat:[false, '该文件已经存在！']
327             }
328         }
329     }});
330     return Auth;
331 }, {requires:['node','base']});
