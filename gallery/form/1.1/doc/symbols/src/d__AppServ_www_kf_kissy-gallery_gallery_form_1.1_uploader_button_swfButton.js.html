1 /**
  2  * @fileoverview flash上传按钮
  3  * @author: 紫英(橘子)<daxingplay@gmail.com>, 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/button/swfButton', function (S, Node, Base, SwfUploader) {
  6     var EMPTY = '', $ = Node.all,
  7         SWF_WRAPPER_ID_PREVFIX = 'swf-uploader-wrapper-';
  8 
  9     /**
 10      * @name SwfButton
 11      * @class flash上传按钮，基于龙藏的AJBrige。只有使用flash上传方式时候才会实例化这个类
 12      * @constructor
 13      * @extends Base
 14      */
 15     function SwfButton(target, config) {
 16         var self = this;
 17         config = S.merge({target:$(target)}, config);
 18         //调用父类构造函数
 19         SwfButton.superclass.constructor.call(self, config);
 20     }
 21 
 22     S.mix(SwfButton, /** @lends SwfButton*/{
 23         /**
 24          * 支持的事件
 25          */
 26         event:{
 27             //组件运行后事件
 28             RENDER : 'render',
 29             //选择文件后事件
 30             CHANGE:'change',
 31             //鼠标在swf中滑过事件
 32             MOUSE_OVER:'mouseOver',
 33             //鼠标在swf中按下事件
 34             MOUSE_DOWN:'mouseDown',
 35             //鼠标在swf中弹起事件
 36             MOUSE_UP:'mouseUp',
 37             //鼠标在swf中移开事件
 38             MOUSE_OUT:'mouseOut',
 39             //鼠标单击事件
 40             CLICK:'click'
 41         }
 42     });
 43     S.extend(SwfButton, Base, /** @lends SwfButton.prototype*/{
 44         /**
 45          *  运行，会实例化AJBrige的Uploader，存储为swfUploader属性
 46          */
 47         render:function () {
 48             var self = this,
 49                 $target = self.get('target'),
 50                 swfUploader,
 51                 multiple = self.get('multiple'),
 52                 fileFilters = self.get('fileFilters') ;
 53             $target.css('position', 'relative');
 54             self.set('swfWrapper',self._createSwfWrapper());
 55             self._setFlashSizeConfig();
 56             swfUploader = self._initSwfUploader();
 57             //SWF 内容准备就绪
 58             swfUploader.on('contentReady', function(ev){
 59                 //多选和文件过滤控制
 60                 swfUploader.browse(multiple, fileFilters);
 61                 //监听鼠标事件
 62                 self._bindBtnEvent();
 63                 //监听选择文件后事件
 64                 swfUploader.on('fileSelect', self._changeHandler, self);
 65                 self._setDisabled(self.get('disabled'));
 66                 self.fire(SwfButton.event.RENDER);
 67             }, self);
 68         },
 69         /**
 70          * 创建flash容器
 71          */
 72         _createSwfWrapper:function () {
 73             var self = this,
 74                 target = self.get('target'),
 75                 tpl = self.get('tpl'),
 76                 //容器id
 77                 id = self.get('swfWrapperId') != EMPTY && self.get('swfWrapperId') || SWF_WRAPPER_ID_PREVFIX + S.guid(),
 78                 //容器html
 79                 html = S.substitute(tpl, {id:id});
 80             self.set('swfWrapperId', id);
 81             return $(html).appendTo(target);
 82         },
 83         /**
 84          * 初始化ajbridge的uploader
 85          * @return {SwfUploader}
 86          */
 87         _initSwfUploader:function () {
 88             var self = this, flash = self.get('flash'),
 89                 id = self.get('swfWrapperId'),
 90                 swfUploader;
 91             try {
 92                 //实例化AJBridge.Uploader
 93                 swfUploader = new SwfUploader(id, flash);
 94                 self.set('swfUploader', swfUploader);
 95             } catch (err) {
 96 
 97             }
 98             return swfUploader;
 99         },
100         /**
101          * 监听swf的各个鼠标事件
102          * @return {SwfButton}
103          */
104         _bindBtnEvent:function () {
105             var self = this, event = SwfButton.event,
106                 swfUploader = self.get('swfUploader');
107             if (!swfUploader) return false;
108             S.each(event, function (ev) {
109                 swfUploader.on(ev, function (e) {
110                     self.fire(ev);
111                 }, self);
112             });
113             return self;
114         },
115         /**
116          * 设置flash配置参数
117          */
118         _setFlashSizeConfig:function () {
119             var self = this, flash = self.get('flash'),
120                 target = self.get('target');
121             S.mix(flash.attrs, {
122                 width:target.width(),
123                 height:target.height()
124             });
125             self.set('flash', flash);
126         },
127         /**
128          * flash中选择完文件后触发的事件
129          */
130         _changeHandler:function (ev) {
131             var self = this, files = ev.fileList;
132             self.fire(SwfButton.event.CHANGE, {files:files});
133         },
134         /**
135          * 设置上传组件的禁用
136          * @param {Boolean} disabled 是否禁用
137          * @return {Boolean}
138          */
139         _setDisabled : function(disabled){
140             var self = this,
141                 swfUploader = self.get('swfUploader'),
142                 cls = self.get('cls'),disabledCls = cls.disabled,
143                 $target = self.get('target'),
144                 $swfWrapper = self.get('swfWrapper');
145             if(!swfUploader || !S.isBoolean(disabled)) return false;
146             if(!disabled){
147                 $target.removeClass(disabledCls);
148                 //显示swf容器
149                 $swfWrapper.show();
150                 //TODO:之所以不使用更简单的unlock()方法，因为这个方法应用无效，有可能是bug
151                 //swfUploader.unlock();
152             }else{
153                 $target.addClass(disabledCls);
154                 //隐藏swf容器
155                 $swfWrapper.hide();
156                 //swfUploader.lock();
157             }
158             return disabled;
159         }
160     }, {ATTRS:/** @lends SwfButton.prototype*/{
161         /**
162          * 按钮目标元素
163          * @type KISSY.Node
164          * @default ""
165          */
166         target:{value:EMPTY},
167         /**
168          * swf容器
169          * @type KISSY.Node
170          * @default ""
171          */
172         swfWrapper : {value : EMPTY},
173         /**
174          * swf容器的id，如果不指定将使用随机id
175          * @type Number
176          * @default ""
177          */
178         swfWrapperId:{value:EMPTY},
179         /**
180          * flash容器模板
181          * @type String
182          */
183         tpl:{
184             value:'<div id="{id}" class="uploader-button-swf" style="position: absolute;top:0;left:0;"></div>'
185         },
186         /**
187          * 是否开启多选支持
188          * @type Boolean
189          * @default true
190          */
191         multiple:{
192             value:true,
193             setter:function (v) {
194                 var self = this, swfUploader = self.get('swfUploader');
195                 if (swfUploader) {
196                     swfUploader.multifile(v);
197                 }
198                 return v;
199             }
200         },
201         /**
202          * 文件过滤，格式类似[{desc:"JPG,JPEG,PNG,GIF,BMP",ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"}]
203          * @type Array
204          * @default []
205          */
206         fileFilters:{
207             value:[],
208             setter:function (v) {
209                 var self = this, swfUploader = self.get('swfUploader');
210                 if (swfUploader && S.isArray(v)) {
211                     swfUploader.filter(v);
212                 }
213                 return v;
214             }
215         },
216         /**
217          * 禁用按钮
218          * @type Boolean
219          * @default false
220          */
221         disabled : {
222             value : false,
223             setter : function(v){
224                 var self = this, swfUploader = self.get('swfUploader');
225                 if (swfUploader) {
226                     self._setDisabled(v);
227                 }
228                 return v;
229             }
230         },
231         /**
232          * 样式
233          * @type Object
234          * @default  { disabled:'uploader-button-disabled' }
235          */
236         cls : {
237             value : { disabled:'uploader-button-disabled' }
238         },
239         /**
240          * flash配置，对于swf文件配路径配置非常关键，使用默认cdn上的路径就好
241          * @type Object
242          * @default { src:'http://a.tbcdn.cn/s/kissy/gallery/form/1.1/uploader/plugins/ajbridge/uploader.swf', id:'swfUploader', params:{ bgcolor:"#fff", wmode:"transparent" }, attrs:{ }, hand:true, btn:true }
243              }
244          */
245         flash:{
246             value:{
247                 src:'http://a.tbcdn.cn/s/kissy/gallery/form/1.1/uploader/plugins/ajbridge/uploader.swf',
248                 id:'swfUploader',
249                 params:{
250                     bgcolor:"#fff",
251                     wmode:"transparent"
252                 },
253                 //属性
254                 attrs:{ },
255                 //手型
256                 hand:true,
257                 //启用按钮模式,激发鼠标事件
258                 btn:true
259             }
260         },
261         /**
262          *  ajbridge的uploader的实例
263          *  @type SwfUploader
264          *  @default ""
265          */
266         swfUploader:{value:EMPTY}
267     }});
268     return SwfButton;
269 }, {requires:['node', 'base', '../plugins/ajbridge/uploader']});
