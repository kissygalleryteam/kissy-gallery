 10 KISSY.add('gallery/form/1.1/uploader/plugins/ajbridge/ajbridge', function(S,Flash) {
 11 
 12     var ID_PRE = '#',
 13         VERSION = '1.0.15',
 14 		PREFIX = 'ks-ajb-',
 15 		LAYOUT = 100,
 16         EVENT_HANDLER = 'KISSY.AJBridge.eventHandler'; // Flash 事件抛出接受通道
 17 
 18     /**
 19      * @constructor
 20      * @param {String} id       注册应用容器 id
 21      * @param {Object} config   基本配置同 S.Flash 的 config
 22      * @param {Boolean} manual  手动进行 init
 23      */
 24     function AJBridge(id, config,manual) {
 25         id = id.replace(ID_PRE, ''); // 健壮性考虑。出于 KISSY 习惯采用 id 选择器
 26         config = Flash._normalize(config||{}); // 标准化参数关键字
 27 
 28         var self = this,
 29             target = ID_PRE + id, // 之所以要求使用 id，是因为当使用 ajbridge 时，程序员自己应该能确切知道自己在做什么
 30             callback = function(data) {
 31                 if (data.status < 1) {
 32                     self.fire('failed', { data: data });
 33                     return;
 34                 }
 35 				
 36                 S.mix(self, data);
 37 
 38                 // 执行激活 静态模式的 flash
 39                 // 如果这 AJBridge 先于 DOMReady 前执行 则失效
 40                 // 建议配合 S.ready();
 41                 if (!data.dynamic || !config.src) {
 42 						self.activate();
 43                 }
 44             };
 45 		
 46 		// 自动产生 id	
 47 		config.id = config.id || S.guid(PREFIX);
 48 
 49         // 注册应用实例
 50         AJBridge.instances[config.id] = self;
 51 
 52         //	动态方式
 53         if (config.src) {
 54             // 强制打开 JS 访问授权，AJBridge 的最基本要求
 55             config.params.allowscriptaccess = 'always';
 56             config.params.flashvars = S.merge(config.params.flashvars, {
 57                 // 配置 JS 入口
 58                 jsEntry: EVENT_HANDLER,
 59                 // 虽然 Flash 通过 ExternalInterface 获得 obejctId
 60                 // 但是依然存在兼容性问题, 因此需要直接告诉
 61                 swfID: config.id
 62             });
 63         }
 64 
 65         // 支持静态方式，但是要求以上三个步骤已静态写入
 66         // 可以参考 test.html
 67 		
 68         // 由于完全基于事件机制，因此需要通过监听之后进行初始化 Flash
 69 		
 70         if(manual)self.__args = [target, config, callback];
 71 		else S.later(Flash.add,LAYOUT,false,Flash,[target, config, callback]);
 72     }
 73 
 74     /**
 75      * 静态方法
 76      */
 77     S.app(AJBridge, {
 78 
 79         version: VERSION,
 80 
 81         instances: { },
 82 
 83         /**
 84          * 处理来自 AJBridge 已定义的事件
 85          * @param {String} id            swf传出的自身ID
 86          * @param {Object} event        swf传出的事件
 87          */
 88         eventHandler: function(id, event) {
 89             var instance = AJBridge.instances[id];
 90             if (instance) {
 91                 instance.__eventHandler(id, event);
 92             }
 93         },
 94 
 95         /**
 96          * 批量注册 SWF 公开的方法
 97          * @param {Class} C
 98          * @param {String|Array} methods
 99          */
100         augment: function (C, methods) {
101             if (S.isString(methods)) {
102                 methods = [methods];
103             }
104             if (!S.isArray(methods)) return;
105 			
106 			
107 
108             S.each(methods, function(methodName) {
109                 C.prototype[methodName] = function() {
110                     try {
111                         return this.callSWF(methodName, S.makeArray(arguments));
112                     } catch(e) { // 当 swf 异常时，进一步捕获信息
113                         this.fire('error', { message: e });
114                     }
115                 }
116             });
117         }
118     });
119 
120     S.augment(AJBridge, S.EventTarget, {
121 
122         init: function() {
123 			if(!this.__args)return;
124             Flash.add.apply(Flash, this.__args);
125 			this.__args = null;
126 			delete this.__args; // 防止重复添加
127         },
128 
129         __eventHandler: function(id, event) {
130             var self = this,
131                 type = event.type;
132 			
133             event.id = id;   //	弥补后期 id 使用
134             switch(type){
135 				case "log":
136 					 S.log(event.message);
137 					break;
138 				default:
139 					self.fire(type, event);
140 			}
141 			
142         },
143 
144         /**
145          * Calls a specific function exposed by the SWF's ExternalInterface.
146          * @param func {String} the name of the function to call
147          * @param args {Array} the set of arguments to pass to the function.
148          */
149         callSWF: function (func, args) {
150             var self = this;
151             args = args || [];
152             try {
153                 if (self.swf[func]) {
154                     return self.swf[func].apply(self.swf, args);
155                 }
156             }
157             // some version flash function is odd in ie: property or method not supported by object
158             catch(e) {
159                 var params = '';
160                 if (args.length !== 0) {
161                     params = "'" + args.join("','") + "'";
162                 }
163                 //avoid eval for compressiong
164                 return (new Function('self', 'return self.swf.' + func + '(' + params + ');'))(self);
165             }
166         }
167     });
168 
169     // 为静态方法动态注册
170     // 注意，只有在 S.ready() 后进行 AJBridge 注册才有效。
171     AJBridge.augment(AJBridge, ['activate', 'getReady','getCoreVersion']);
172 
173     window.AJBridge = S.AJBridge = AJBridge;
174 
175     return AJBridge;
176 }, { requires:["flash"] });
