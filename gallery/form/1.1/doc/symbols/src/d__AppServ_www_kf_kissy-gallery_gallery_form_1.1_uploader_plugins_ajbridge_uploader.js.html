  9 KISSY.add('gallery/form/1.1/uploader/plugins/ajbridge/uploader', function(S,flash,A) {
 10 
 11     /**
 12      * @constructor
 13      * @param {String} id                                    需要注册的SWF应用ID
 14      * @param {Object} config                                配置项
 15      * @param {String} config.ds                             default server 的缩写
 16      * @param {String} config.dsp                            default server parameters 的缩写
 17      * @param {Boolean} config.btn                           启用按钮模式，默认 false
 18      * @param {Boolean} config.hand                          显示手型，默认 false
 19      */
 20     function Uploader(id, config) {
 21         config = config || { };
 22         var flashvars = { };
 23 		
 24 		
 25 		
 26 		S.each(['ds', 'dsp', 'btn', 'hand'], function(key) {
 27 			if(key in config) flashvars[key] = config[key];
 28 		});
 29 		
 30 
 31         config.params = config.params || { };
 32         config.params.flashvars = S.merge(config.params.flashvars, flashvars);
 33 
 34 		Uploader.superclass.constructor.call(this, id, config);
 35     }
 36 
 37     S.extend(Uploader, A);
 38 
 39     A.augment(Uploader,
 40         [
 41             'setFileFilters',
 42             'filter',
 43             'setAllowMultipleFiles',
 44             'multifile',
 45             'browse',
 46             'upload',
 47             'uploadAll',
 48             'cancel',
 49             'getFile',
 50             'removeFile',
 51             'lock',
 52             'unlock',
 53             'setBtnMode',
 54             'useHand',
 55             'clear'
 56         ]
 57         );
 58 
 59     Uploader.version = '1.0.1';
 60     A.Uploader = Uploader;
 61     return A.Uploader;
 62 },{ requires:["flash","./ajbridge"] });
