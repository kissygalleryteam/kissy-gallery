1 /**
  2  * @fileoverview 进度条
  3  * @author 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/plugins/progressBar/progressBar',function(S, Node, Base) {
  6     var EMPTY = '',$ = Node.all,
  7         PROGRESS_BAR = 'progressbar',ROLE = 'role',
  8         ARIA_VALUEMIN = 'aria-valuemin',ARIA_VALUEMAX = 'aria-valuemax',ARIA_VALUENOW = 'aria-valuenow',
  9         DATA_VALUE = 'data-value';
 10     /**
 11      * @name ProgressBar
 12      * @class 进度条
 13      * @constructor
 14      * @extends Base
 15      * @requires Node
 16      */
 17     function ProgressBar(wrapper, config) {
 18         var self = this;
 19         config = S.merge({wrapper:$(wrapper)}, config);
 20         //调用父类构造函数
 21         ProgressBar.superclass.constructor.call(self, config);
 22     }
 23     S.mix(ProgressBar, /** @lends ProgressBar.prototype*/{
 24         /**
 25          * 模板
 26          */
 27         tpl : {
 28             DEFAULT:'<div class="ks-progress-bar-value" data-value="{value}"></div>'
 29         },
 30         /**
 31          * 组件用到的样式
 32          */
 33         cls : {
 34             PROGRESS_BAR : 'ks-progress-bar',
 35             VALUE : 'ks-progress-bar-value'
 36         },
 37         /**
 38          * 组件支持的事件
 39          */
 40         event : {
 41             RENDER : 'render',
 42             CHANGE : 'change',
 43             SHOW : 'show',
 44             HIDE : 'hide'
 45         }
 46     });
 47     //继承于Base，属性getter和setter委托于Base处理
 48     S.extend(ProgressBar, Base, /** @lends ProgressBar.prototype*/{
 49         /**
 50          * 运行
 51          */
 52         render : function() {
 53             var self = this,$wrapper = self.get('wrapper'),
 54                 width = self.get('width');
 55             if(!$wrapper.length) return false;
 56             //给容器添加ks-progress-bar样式名
 57             $wrapper.addClass(ProgressBar.cls.PROGRESS_BAR)
 58                     .width(width);
 59             self._addAttr();
 60             !self.get('visible') && self.hide();
 61             self.set('bar',self._create());
 62             self.fire(ProgressBar.event.RENDER);
 63         },
 64         /**
 65          * 显示进度条
 66          */
 67         show : function(){
 68             var self = this,$wrapper = self.get('wrapper');
 69             $wrapper.fadeIn(self.get('duration'),function(){
 70                 self.set('visible',true);
 71                 self.fire(ProgressBar.event.SHOW,{visible : true});
 72             });
 73         },
 74         /**
 75          * 隐藏进度条
 76          */
 77         hide : function(){
 78             var self = this,$wrapper = self.get('wrapper');
 79             $wrapper.fadeOut(self.get('duration'),function(){
 80                 self.set('visible',false);
 81                 self.fire(ProgressBar.event.HIDE,{visible : false});
 82             });
 83         },
 84         /**
 85          * 创建进度条
 86          * @return {NodeList}
 87          */
 88         _create : function(){
 89             var self = this,
 90                 $wrapper = self.get('wrapper'),
 91                 value = self.get('value'),tpl = self.get('tpl'),
 92                 html = S.substitute(tpl, {value : value}) ;
 93             $wrapper.html('');
 94             return $(html).appendTo($wrapper);
 95 
 96         },
 97         /**
 98          * 给进度条容器添加一些属性
 99          * @return {Object} ProgressBar的实例
100          */
101         _addAttr : function() {
102             var self = this,$wrapper = self.get('wrapper'),value = self.get('value');
103             $wrapper.attr(ROLE, PROGRESS_BAR);
104             $wrapper.attr(ARIA_VALUEMIN, 0);
105             $wrapper.attr(ARIA_VALUEMAX, 100);
106             $wrapper.attr(ARIA_VALUENOW, value);
107             return self;
108         }
109     }, {ATTRS : /** @lends ProgressBar*/{
110         /**
111          * 容器
112          */
113         wrapper : {value : EMPTY},
114         /**
115          * 进度条元素
116          */
117         bar : {value : EMPTY},
118         /**
119          * 进度条宽度
120          */
121         width : { value:100 },
122         /**
123          * 当前进度
124          */
125         value : {
126             value : 0,
127             setter : function(v) {
128                 var self = this,$wrapper = self.get('wrapper'),$bar = self.get('bar'),
129                     speed = self.get('speed'),
130                     width;
131                 if (v > 100) v = 100;
132                 if (v < 0) v = 0;
133                 //将百分比宽度换算成像素值
134                 width = $wrapper.width() * (v / 100);
135                 $bar.animate({'width':width + 'px'},speed,'none',function(){
136                     $wrapper.attr(ARIA_VALUENOW,v);
137                     $bar.attr(DATA_VALUE,v);
138                     self.fire(ProgressBar.event.CHANGE,{value : v,width : width});
139                 });
140                 return v;
141             }
142         },
143         /**
144          * 控制进度条的可见性
145          */
146         visible : { value:true },
147         /**
148          * 显隐动画的速度
149          */
150         duration : {
151           value : 0.3
152         },
153         /**
154          * 模板
155          */
156         tpl : {
157             value : ProgressBar.tpl.DEFAULT
158         },
159         speed : {value : 0.2}
160     }});
161     return ProgressBar;
162 }, {requires : ['node','base']});
