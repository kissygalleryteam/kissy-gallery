1 /**
  2  * @fileoverview 文件上传队列列表显示和处理
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/queue', function (S, Node, Base) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploader-queue]:';
  7 
  8     /**
  9      * @name Queue
 10      * @class 文件上传队列，用于存储文件数据
 11      * @constructor
 12      * @extends Base
 13      * @param {Object} config Queue没有必写的配置
 14      * @param {Uploader} config.uploader Uploader的实例
 15      * @example
 16      * S.use('gallery/form/1.1/uploader/queue/base,gallery/form/1.1/uploader/themes/default/style.css', function (S, Queue) {
 17      *    var queue = new Queue();
 18      *    queue.render();
 19      * })
 20      */
 21     function Queue(config) {
 22         var self = this;
 23         //调用父类构造函数
 24         Queue.superclass.constructor.call(self, config);
 25     }
 26 
 27     S.mix(Queue, /**@lends Queue*/ {
 28         /**
 29          * 模板
 30          */
 31         tpl:{
 32             DEFAULT:'<li id="queue-file-{id}" class="clearfix" data-name="{name}">' +
 33                 '<div class="f-l sprite file-icon"></div>' +
 34                 '<div class="f-l">{name}</div>' +
 35                 '<div class="f-l file-status J_FileStatus"></div>' +
 36                 '</li>'
 37         },
 38         /**
 39          * 支持的事件
 40          */
 41         event:{
 42             //添加完文件后触发
 43             ADD:'add',
 44             //批量添加文件后触发
 45             ADD_FILES:'addFiles',
 46             //删除文件后触发
 47             REMOVE:'remove',
 48             //清理队列所有的文件后触发
 49             CLEAR:'clear',
 50             //当改变文件状态后触发
 51             FILE_STATUS : 'statusChange',
 52             //更新文件数据后触发
 53             UPDATE_FILE : 'updateFile',
 54             // 恢复文件后触发
 55             RESTORE: 'restore'
 56         },
 57         /**
 58          * 文件的状态
 59          */
 60         status:{
 61             WAITING : 'waiting',
 62             START : 'start',
 63             PROGRESS : 'progress',
 64             SUCCESS : 'success',
 65             CANCEL : 'cancel',
 66             ERROR : 'error',
 67             RESTORE: 'restore'
 68         },
 69         //样式
 70         cls:{
 71             QUEUE:'ks-uploader-queue'
 72         },
 73         hook:{
 74             //状态
 75             STATUS:'.J_FileStatus'
 76         },
 77         FILE_ID_PREFIX:'file-'
 78     });
 79     /**
 80      * @name Queue#add
 81      * @desc  添加完文件后触发
 82      * @event
 83      * @param {Number} ev.index 文件在队列中的索引值
 84      * @param {Object} ev.file 文件数据
 85      */
 86     /**
 87      * @name Queue#addFiles
 88      * @desc  批量添加文件后触发
 89      * @event
 90      * @param {Array} ev.files 添加后的文件数据集合
 91      */
 92     /**
 93      * @name Queue#remove
 94      * @desc  删除文件后触发
 95      * @event
 96      * @param {Number} ev.index 文件在队列中的索引值
 97      * @param {Object} ev.file 文件数据
 98      */
 99     /**
100      * @name Queue#clear
101      * @desc  清理队列所有的文件后触发
102      * @event
103      */
104     /**
105      * @name Queue#statusChange
106      * @desc  当改变文件状态后触发
107      * @event
108      * @param {Number} ev.index 文件在队列中的索引值
109      * @param {String} ev.status 文件状态
110      */
111     /**
112      * @name Queue#updateFile
113      * @desc  更新文件数据后触发
114      * @event
115      * @param {Number} ev.index 文件在队列中的索引值
116      * @param {Object} ev.file 文件数据
117      */
118     /**
119      * @name Queue#restore
120      * @desc  恢复文件后触发
121      * @event
122      * @param {Array} ev.files 文件数据集合
123      */
124     //继承于Base，属性getter和setter委托于Base处理
125     S.extend(Queue, Base, /** @lends Queue.prototype*/{
126         /**
127          * 向上传队列添加文件
128          * @param {Object | Array} files 文件数据，传递数组时为批量添加
129          * @example
130          * //测试文件数据
131  var testFile = {'name':'test.jpg',
132      'size':2000,
133      'input':{},
134      'file':{'name':'test.jpg', 'type':'image/jpeg', 'size':2000}
135  };
136  //向队列添加文件
137  queue.add(testFile);
138          */
139         add:function (files, callback) {
140             var self = this,fileData={};
141             //如果存在多个文件，需要批量添加文件
142             if (files.length > 0) {
143                 fileData=[];
144                 S.each(files,function(file){
145                     fileData.push(self._addFile(file));
146                 });
147             } else {
148                 fileData = self._addFile(files);
149             }
150             callback && callback.call(self);
151             return fileData;
152         },
153         /**
154          * 向队列添加单个文件
155          * @param {Object} file 文件数据
156          * @param {Function} callback 添加完成后执行的回调函数
157          * @return {Object} 文件数据对象
158          */
159         _addFile:function (file,callback) {
160             if (!S.isObject(file)) {
161                 S.log(LOG_PREFIX + '_addFile()参数file不合法！');
162                 return false;
163             }
164             var self = this,
165                 //设置文件对象
166                 fileData = self._setAddFileData(file),
167                 //文件索引
168                 index = self.getFileIndex(fileData.id),
169                 fnAdd = self.get('fnAdd');
170             //执行用户自定义的回调函数
171             if(S.isFunction(fnAdd)){
172                 fileData = fnAdd(index,fileData);
173             }
174             self.fire(Queue.event.ADD, {index:index, file:fileData,uploader:self.get('uploader')});
175             callback && callback.call(self, index, fileData);
176             return fileData;
177         },
178         /**
179          * 删除队列中指定id的文件
180          * @param {Number} indexOrFileId 文件数组索引或文件id
181          * @param {Function} callback 删除元素后执行的回调函数
182          * @example
183          * queue.remove(0);
184          */
185         remove:function (indexOrFileId, callback) {
186             var self = this, files = self.get('files'), file;
187             //参数是字符串，说明是文件id，先获取对应文件数组的索引
188             if (S.isString(indexOrFileId)) {
189                 indexOrFileId = self.getFileIndex(indexOrFileId);
190             }
191             //文件数据对象
192             file = files[indexOrFileId];
193             if (!S.isObject(file)) {
194                 S.log(LOG_PREFIX + 'remove()不存在index为' + indexOrFileId + '的文件数据');
195                 return false;
196             }
197             //将该id的文件过滤掉
198             files = S.filter(files, function (file, i) {
199                 return i !== indexOrFileId;
200             });
201             self.set('files', files);
202             self.fire(Queue.event.REMOVE, {index:indexOrFileId, file:file});
203             callback && callback.call(self,indexOrFileId, file);
204             return file;
205         },
206         /**
207          * 清理队列
208          */
209         clear:function () {
210             var self = this, files;
211             _remove();
212             //移除元素
213             function _remove() {
214                 files = self.get('files');
215                 if (!files.length) {
216                     self.fire(Queue.event.CLEAR);
217                     return false;
218                 }
219                 self.remove(0, function () {
220                     _remove();
221                 });
222             }
223         },
224         /**
225          * 获取或设置文件状态，默认的主题共有以下文件状态：'waiting'、'start'、'progress'、'success'、'cancel'、'error' ,每种状态的dom情况都不同，刷新文件状态时候同时刷新状态容器类下的DOM节点内容。
226          * @param {Number} index 文件数组的索引值
227          * @param {String} status 文件状态
228          * @return {Object}
229          * @example
230          * queue.fileStatus(0, 'success');
231          */
232         fileStatus:function (index, status, args) {
233             if (!S.isNumber(index) || !S.isString(status)) return false;
234             var self = this, file = self.getFile(index),
235                 theme = self.get('theme'),
236                 curStatus,statusMethod;
237             if (!file || !theme) return false;
238             //状态
239             curStatus = file['status'];
240             //状态一直直接返回
241             if(curStatus == status) return self;
242 
243             //更新状态
244             self.updateFile(index,{status:status});
245 
246             statusMethod = '_'+status+'Handler';
247             //如果主题存在对应的状态变更监听器，予以执行
248             if(S.isFunction(theme[statusMethod])){
249                 args = S.merge({uploader:self.get('uploader'),index:index,file:file,id:file.id},args);
250                 theme[statusMethod].call(theme,args);
251             }
252             self.fire(Queue.event.FILE_STATUS,{index : index,status : status,args:args,file:file});
253             return  self;
254         },
255         /**
256          * 获取指定索引值的队列中的文件
257          * @param  {Number} index 文件在队列中的索引
258          * @return {Object}
259          */
260         getFile:function (index) {
261             if (!S.isNumber(index)) return false;
262             var self = this, files = self.get('files'),
263                 file = files[index];
264             if (!S.isPlainObject(file)){
265                 S.log('getFile():文件数据为空！');
266                 file = {};
267             }
268             return file;
269         },
270         /**
271          * 根据文件id来查找文件在队列中的索引
272          * @param {String} fileId 文件id
273          * @return {Number} index
274          */
275         getFileIndex:function (fileId) {
276             var self = this, files = self.get('files'), index = -1;
277             S.each(files, function (file, i) {
278                 if (file.id == fileId) {
279                     index = i;
280                     return true;
281                 }
282             });
283             return index;
284         },
285         /**
286          * 更新文件数据对象，你可以追加数据
287          * @param {Number} index 文件数组内的索引值
288          * @param {Object} data 数据
289          * @return {Object}
290          */
291         updateFile:function (index, data) {
292             if (!S.isNumber(index)) return false;
293             if (!S.isObject(data)) {
294                 S.log(LOG_PREFIX + 'updateFile()的data参数有误！');
295                 return false;
296             }
297             var self = this, files = self.get('files'),
298                 file = self.getFile(index);
299             if (!file) return false;
300             S.mix(file, data);
301             files[index] = file;
302             self.set('files', files);
303             self.fire(Queue.event.UPDATE_FILE,{index : index, file : file});
304             return file;
305         },
306         /**
307          * 获取等指定状态的文件对应的文件数组index的数组
308          * @param {String} type 状态类型
309          * @return {Array}
310          * @example
311          * //getFiles()和getFileIds()的作用是不同的，getFiles()类似过滤数组，获取的是指定状态的文件数据，而getFileIds()只是获取指定状态下的文件对应的在文件数组内的索引值。
312          * var indexs = queue.getFileIds('waiting');
313          */
314         getIndexs:function (type) {
315             var self = this, files = self.get('files'),
316                 status, indexs = [];
317             if (!files.length) return indexs;
318             S.each(files, function (file, index) {
319                 if (S.isObject(file)) {
320                     status = file.status;
321                     //文件状态
322                     if (status == type) {
323                         indexs.push(index);
324                     }
325                 }
326             });
327             return indexs;
328         },
329         /**
330          * 获取指定状态下的文件
331          * @param {String} status 状态类型
332          * @return {Array}
333          * @example
334          * //获取等待中的所有文件
335          * var files = queue.getFiles('waiting');
336          */
337         getFiles:function (status) {
338             var self = this, files = self.get('files'), statusFiles = [];
339             if (!files.length) return [];
340             S.each(files, function (file) {
341                 if (file && file.status == status) statusFiles.push(file);
342             });
343             return statusFiles;
344         },
345         /**
346          * 添加文件时先向文件数据对象追加id、size等数据
347          * @param {Object} file 文件数据对象
348          * @return {Object} 新的文件数据对象
349          */
350         _setAddFileData:function (file) {
351             var self = this,
352                 files = self.get('files');
353             if (!S.isObject(file)) {
354                 S.log(LOG_PREFIX + '_updateFileData()参数file不合法！');
355                 return false;
356             }
357             //设置文件唯一id
358             if (!file.id) file.id = S.guid(Queue.FILE_ID_PREFIX);
359             //转换文件大小单位为（kb和mb）
360             if (file.size) file.textSize = S.convertByteSize(file.size);
361             //状态
362             file.status = EMPTY;
363             files.push(file);
364             return file;
365         }
366     }, {ATTRS:/** @lends Queue.prototype*/{
367         /**
368          * 添加完文件数据后执行的回调函数，会在add事件前触发
369          * @type Function
370          * @default  ''
371          */
372         fnAdd:{value:EMPTY},
373         /**
374          * 队列内所有文件数据集合
375          * @type Array
376          * @default []
377          * @example
378          * var ids = [],
379          files = queue.get('files');
380          S.each(files, function (file) {
381          ids.push(file.id);
382          });
383          alert('所有文件id：' + ids);
384          */
385         files:{value:[]},
386         /**
387          * 该队列对应的Uploader实例
388          * @type Uploader
389          * @default ""
390          */
391         uploader:{value:EMPTY},
392         /**
393          * 主题实例
394          * @type Theme
395          * @default ""
396          */
397         theme:{value:EMPTY}
398     }});
399 
400     return Queue;
401 }, {requires:['node', 'base']});
