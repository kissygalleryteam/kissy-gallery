1 /**
  2  * @fileoverview 上传组件主题基类

  5 
  6 KISSY.add('gallery/form/1.1/uploader/theme', function (S, Node, Base) {
  7     var EMPTY = '', $ = Node.all,
  8         //主题样式名前缀
  9         classSuffix = {BUTTON:'-button', QUEUE:'-queue'};
 10 
 11     /**
 12      * @name Theme
 13      * @class 上传组件主题基类
 14      * @constructor
 15      * @extends Base
 16      * @requires Queue
 17      */
 18     function Theme(config) {
 19         var self = this;
 20         //调用父类构造函数
 21         Theme.superclass.constructor.call(self, config);
 22         self._LoaderCss();
 23     }
 24 
 25     S.extend(Theme, Base, /** @lends Theme.prototype*/{
 26         /**
 27          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 28          * @param {Uploader} uploader
 29          */
 30         afterUploaderRender:function (uploader) {
 31 
 32         },
 33         /**
 34          * 获取状态容器
 35          * @param {KISSY.NodeList} target 文件的对应的dom（一般是li元素）
 36          * @return {KISSY.NodeList}
 37          */
 38         _getStatusWrapper:function (target) {
 39             return target.children('.J_FileStatus');
 40         },
 41         /**
 42          * 控制文件对应的li元素的显影
 43          * @param {Boolean} isShow 是否认显示
 44          * @param {NodeList} target li元素
 45          * @param {Function} callback 回调
 46          */
 47         displayFile:function (isShow, target, callback) {
 48             var self = this,
 49                 duration = self.get('duration');
 50             if (!target || !target.length) return false;
 51             target[isShow && 'fadeIn' || 'fadeOut'](duration, function () {
 52                 callback && callback.call(self);
 53             });
 54         },
 55         /**
 56          * 文件处于等待上传状态时触发
 57          */
 58         _waitingHandler:function (ev) {
 59 
 60         },
 61         /**
 62          * 文件处于开始上传状态时触发
 63          */
 64         _startHandler:function (ev) {
 65 
 66         },
 67         /**
 68          * 文件处于正在上传状态时触发
 69          */
 70         _progressHandler:function (ev) {
 71 
 72         },
 73         /**
 74          * 文件处于上传成功状态时触发
 75          */
 76         _successHandler:function (ev) {
 77 
 78         },
 79         /**
 80          * 文件处于上传错误状态时触发
 81          */
 82         _errorHandler:function (ev) {
 83 
 84         },
 85         /**
 86          * 从路径隐藏域抓取文件，往队列添加文件后触发
 87          * @param ev
 88          */
 89         _restoreHandler:function (ev) {
 90 
 91         },
 92         /**
 93          * uploader实例化后执行
 94          */
 95         _UploaderRender:function (callback) {
 96             var self = this;
 97             self._initQueue();
 98             self._addThemeCssName();
 99             //加载插件
100             self._loadPlugins(callback);
101         },
102         /**
103          * 将主题名写入到队列和按钮目标容器，作为主题css样式起始
104          */
105         _addThemeCssName:function () {
106             var self = this, name = self.get('name'),
107                 queue = self.get('queue'),
108                 $queueTarget = $(self.get('queueTarget')),
109                 button = self.get('button'),
110                 $btn;
111             if (name == EMPTY || !queue || !$queueTarget.length) return false;
112             $queueTarget.addClass(name + classSuffix.QUEUE);
113             if (!button) return false;
114             $btn = button.get('target');
115             $btn.addClass(name + classSuffix.BUTTON);
116         },
117         /**
118          * 初始化队列
119          * @return {Queue}
120          */
121         _initQueue:function () {
122             var self = this, queue = self.get('queue');
123             queue.set('fnAdd',function(index, file){
124                 return self._addCallback(index, file);
125             });
126             queue.set('theme', self);
127             queue.on('add', self._addFileHandler, self);
128             queue.on('remove', self._removeFileHandler, self);
129             queue.on('statusChange', function (ev) {
130                 self._setStatusVisibility(ev);
131             });
132             return queue;
133         },
134         /**
135          * 加载css文件
136          */
137         _LoaderCss:function () {
138             var self = this,
139                 isUseCss = self.get('isUseCss'),
140                 cssUrl = self.get('cssUrl');
141             //加载css文件
142             if (!isUseCss) return false;
143             S.use(cssUrl, function () {
144                 S.log(cssUrl + '加载成功！');
145             });
146         },
147         /**
148          * 向队列添加完文件后触发的回调函数（在add事件前触发）
149          * @param {Number} index 文件索引值
150          * @param {Object} file 文件数据
151          * @return {Object} file
152          */
153         _addCallback:function (index, file) {
154             var self = this,
155                 queue = self.get('queue'),
156                 $target = self._appendFileDom(file);
157             //将状态层容器写入到file数据
158             queue.updateFile(index, {
159                 target:$target,
160                 statusWrapper:self._getStatusWrapper($target)
161             });
162             //更换文件状态为等待
163             queue.fileStatus(index,'waiting');
164             self.displayFile(true, $target);
165             //给li下的按钮元素绑定事件
166             self._bindTriggerEvent(index, file);
167             return queue.getFile(index);
168         },
169         /**
170          * 删除队列中的文件后触发的监听器
171          */
172         _removeFileHandler:function (ev) {
173             var self = this,
174                 file = ev.file;
175             self.displayFile(false, file.target);
176         },
177         /**
178          * 给删除、上传、取消等按钮元素绑定事件
179          * @param {Number} index 文件索引值
180          * @param {Object} 文件数据
181          */
182         _bindTriggerEvent:function (index, file) {
183             var self = this,
184                 queue = self.get('queue'),
185                 uploader = self.get('uploader'),
186                 //文件id
187                 fileId = file.id,
188                 //上传链接
189                 $upload = $('.J_Upload_' + fileId),
190                 //取消链接
191                 $cancel = $('.J_Cancel_' + fileId),
192                 //删除链接
193                 $del = $(".J_Del_" + fileId);
194             //点击上传
195             $upload.on('click', function (ev) {
196                 ev.preventDefault();
197                 if (!S.isObject(uploader)) return false;
198                 uploader.upload(index);
199             });
200             //点击取消
201             $cancel.on('click', function (ev) {
202                 ev.preventDefault();
203                 uploader.cancel(index);
204             });
205             //点击删除
206             $del.on('click', function (ev) {
207                 ev.preventDefault();
208                 //删除队列中的文件
209                 queue.remove(fileId);
210             });
211         },
212         /**
213          * 设置状态层的可见性
214          * @param ev
215          */
216         _setStatusVisibility:function (ev) {
217             var $statusWrapper = ev.file.statusWrapper, $status,
218                 file = ev.file, status = file.status;
219             if (!$statusWrapper || !$statusWrapper.length) {
220                 S.log('状态容器层不存在！');
221                 return false;
222             }
223             $status = $statusWrapper.children('.status');
224             $status.hide();
225             $statusWrapper.children('.' + status + '-status').show();
226         },
227         /**
228          * 当队列添加完文件数据后向队列容器插入文件信息DOM结构
229          * @param {Object} fileData 文件数据
230          * @return {KISSY.NodeList}
231          */
232         _appendFileDom:function (fileData) {
233             var self = this, tpl = self.get('fileTpl'),
234                 $target = $(self.get('queueTarget')),
235                 hFile;
236             if (!$target.length) return false;
237             hFile = S.substitute(tpl, fileData);
238             return $(hFile).hide().appendTo($target).data('data-file', fileData);
239         },
240         /**
241          * 根据插件配置加载插件
242          */
243         _loadPlugins:function(callback){
244             var self = this,
245                 plugins = self.get('plugins'),
246                 oPlugin = self.get('oPlugin'),
247                 //模块路径前缀
248                 modPrefix = 'gallery/form/1.1/uploader/plugins/',
249                 mods = [];
250             if(!plugins.length) return false;
251             //拼接模块路径
252             S.each(plugins,function(plugin){
253                 mods.push(modPrefix+plugin+'/' +plugin);
254             });
255             S.use(mods.join(','),function(){
256                  S.each(arguments,function(arg,i){
257                      // 类排除S
258                      if(i>=1) oPlugin[plugins[i-1]] = arg;
259                  });
260                 self.set('oPlugin',oPlugin);
261                 callback && callback.call(self,oPlugin);
262             })
263         }
264     }, {ATTRS:/** @lends Theme.prototype*/{
265         /**
266          *  主题名
267          * @type String
268          * @default ""
269          */
270         name:{value:EMPTY},
271         /**
272          * 是否引用css文件
273          * @type Boolean
274          * @default true
275          */
276         isUseCss:{value:true},
277         /**
278          * css模块路径
279          * @type String
280          * @default ""
281          */
282         cssUrl:{value:EMPTY},
283         /**
284          * 队列使用的模板
285          * @type String
286          * @default ""
287          */
288         fileTpl:{value:EMPTY },
289         /**
290          * 需要加载的插件，需要手动实例化
291          * @type Array
292          * @default []
293          */
294         plugins:{value:[]},
295         /**
296          * 插件类集合
297          * @type Array
298          * @default []
299          */
300         oPlugin:{value:{}},
301         /**
302          * 队列目标元素（一般是ul），队列的实例化过程在Theme中
303          * @type String
304          * @default ""
305          */
306         queueTarget:{value:EMPTY},
307         /**
308          * 动画速度
309          * @type Number
310          * @default 0.3
311          */
312         duration:{value:0.3},
313         /**
314          * Queue（上传队列）实例
315          * @type Queue
316          * @default ""
317          */
318         queue:{value:EMPTY},
319         /**
320          * Uploader 上传组件实例
321          * @type Uploader
322          * @default ""
323          */
324         uploader:{value:EMPTY},
325         /**
326          * Button 按钮实例（_init()下并不存在）
327          * @type Button
328          * @default ""
329          */
330         button:{value:EMPTY},
331         /**
332          * Auth（上传验证）实例
333          * @type Auth
334          * @default ""
335          */
336         auth:{value:EMPTY}
337     }});
338     return Theme;
339 }, {requires:['node', 'base']});
