1 /**
  2  * @fileoverview flash上传方案，基于龙藏写的ajbridge内的uploader
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.1/uploader/type/flash', function (S, Node, UploadType, swfUploader) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploader-FlashType]:';
  7 
  8     /**
  9      * @name FlashType
 10      * @class flash上传方案，基于龙藏写的ajbridge内的uploader
 11      * @constructor
 12      * @extends UploadType
 13      * @requires Node
 14      */
 15     function FlashType(config) {
 16         var self = this;
 17         //调用父类构造函数
 18         FlashType.superclass.constructor.call(self, config);
 19         self._init();
 20     }
 21 
 22     S.mix(FlashType, /** @lends FlashType.prototype*/{
 23         /**
 24          * 事件列表
 25          */
 26         event:S.merge(UploadType.event, {
 27             //swf文件已经准备就绪
 28             SWF_READY: 'swfReady',
 29             //正在上传
 30             PROGRESS:'progress'
 31         })
 32     });
 33 
 34     S.extend(FlashType, UploadType, /** @lends FlashType.prototype*/{
 35         /**
 36          * 初始化
 37          */
 38         _init:function () {
 39             var self = this, swfUploader = self.get('swfUploader');
 40             if(!swfUploader){
 41                 S.log(LOG_PREFIX + 'swfUploader对象为空！');
 42                 return false;
 43             }
 44             //SWF 内容准备就绪
 45             swfUploader.on('contentReady', function(ev){
 46                 self.fire(FlashType.event.SWF_READY);
 47             }, self);
 48             //监听开始上传事件
 49             swfUploader.on('uploadStart', self._uploadStartHandler, self);
 50             //监听文件正在上传事件
 51             swfUploader.on('uploadProgress', self._uploadProgressHandler, self);
 52             //监听文件上传完成事件
 53             swfUploader.on('uploadCompleteData',self._uploadCompleteDataHandler,self);
 54             //监听文件失败事件
 55             swfUploader.on('uploadError',self._uploadErrorHandler,self);
 56         },
 57         /**
 58          * 上传文件
 59          * @param {String} id 文件id
 60          * @return {FlashType}
 61          */
 62         upload:function (id) {
 63             var self = this, swfUploader = self.get('swfUploader'),
 64                 action = self.get('action'), method = 'POST',
 65                 data = self.get('data');
 66             self.set('uploadingId',id);
 67             swfUploader.upload(id, action, method, data);
 68             return self;
 69         },
 70         /**
 71          * 停止上传文件
 72          * @return {FlashType}
 73          */
 74         stop:function () {
 75             var self = this, swfUploader = self.get('swfUploader'),
 76                 uploadingId = self.get('uploadingId');
 77             if(uploadingId != EMPTY){
 78                 swfUploader.cancel(uploadingId);
 79                 self.fire(FlashType.event.STOP, {id : uploadingId});
 80             }
 81             return self;
 82         },
 83         /**
 84          * 开始上传事件监听器
 85          * @param {Object} ev ev.file：文件数据
 86          */
 87         _uploadStartHandler : function(ev){
 88             var self = this;
 89             self.fire(FlashType.event.START, {'file' : ev.file });
 90         },
 91         /**
 92          * 上传中事件监听器
 93          * @param {Object} ev
 94          */
 95         _uploadProgressHandler:function (ev) {
 96             var self = this;
 97             S.mix(ev, {
 98                 //已经读取的文件字节数
 99                 loaded:ev.bytesLoaded,
100                 //文件总共字节数
101                 total : ev.bytesTotal
102             });
103             self.fire(FlashType.event.PROGRESS, { 'loaded':ev.loaded, 'total':ev.total });
104         },
105         /**
106          * 上传完成后事件监听器
107          * @param {Object} ev
108          */
109         _uploadCompleteDataHandler : function(ev){
110             var self = this,result;
111             try {
112                 result = JSON.parse(ev.data);
113             } catch(err) {
114                 S.log(LOG_PREFIX + 'json数据格式不合法！');
115                 self.fire(FlashType.event.ERROR, {msg : '不是合法的json数据'});
116             }
117             self.set('uploadingId',EMPTY);
118             self.fire(FlashType.event.SUCCESS, {result : result});
119         },
120         /**
121          *文件上传失败后事件监听器
122          */
123         _uploadErrorHandler : function(ev){
124             var self = this;
125             self.set('uploadingId',EMPTY);
126             self.fire(FlashType.event.ERROR, {msg : ev.msg});
127         }
128     }, {ATTRS:/** @lends FlashType*/{
129         /**
130          * ajbridge的uploader组件的实例，必须参数
131          */
132         swfUploader:{value:EMPTY},
133         /**
134          * 正在上传的文件id
135          */
136         uploadingId : {value : EMPTY}
137     }});
138     return FlashType;
139 }, {requires:['node', './base']});
