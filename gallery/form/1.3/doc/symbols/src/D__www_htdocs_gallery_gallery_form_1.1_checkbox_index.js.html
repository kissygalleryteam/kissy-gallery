1 /**
  2  * @fileoverview 多选框美化
  3  * @author 伯方<bofang.zxj@taobao.com>
  4  *
  5  **/
  6 KISSY.add('gallery/form/1.3/checkbox/index', function(S, Base, Node) {
  7 	var $ = Node.all;
  8 	/**
  9 	 * @name Checkbox
 10 	 * @class 多选框美化
 11 	 * @constructor
 12 	 * @extends Base
 13 	 * @param {String} target 目标
 14 	 * @param {Object} config * ,组件配置
 15 	 * @param {Object} config.cls *，组件的样式
 16 	 * @example
 17 	 * var ck = new Checkbox('#J_Content input')
 18 	 */
 19 	function Checkbox(target, config) {
 20 		//调用父类构造器
 21 		var self = this;
 22 		Checkbox.superclass.constructor.call(self, config);
 23 		self.set('target', target);
 24 	}
 25 	//方法
 26 	S.extend(Checkbox, Base, /** @lends Checkbox.prototype*/ {
 27 		/**
 28 		 * 开始执行
 29 		 */
 30 		render: function() {
 31 			var self = this;
 32 			//alert(typeof self.get('getLabelFunc') );
 33 			//加载css
 34 			self._loadCss();
 35 			//开始替换
 36 			self._replaceCheckbox();
 37 			//事件绑定
 38 			self._bindEvent();
 39 		},
 40 		/**
 41 		 * 还原checkbox为原生的checkbox
 42 		 * @return {Object} return self
 43 		 */
 44 		recoverCheckbox: function() {
 45 			var self = this,
 46 				targets = self.get('target'),
 47 				checkboxs = self.get('checkboxs');
 48 			$(checkboxs).each(function(value, key) {
 49 				value.hide();
 50 				$(targets[key]).show();
 51 			})
 52 			//self.set('checkboxs',[]);// = null;
 53 			return self;
 54 		},
 55 		/**
 56 		 * 用span替换checkbox，关键步骤
 57 		 */
 58 		_replaceCheckbox: function() {
 59 			var self = this,
 60 				target = self.get('target'),
 61 				html = self._getHtml(0),
 62 				disabledHTML = self._getHtml(2),
 63 				checkedHTML = self._getHtml(1),
 64 				checkbox, checkboxArr = [],
 65 				hasLabel = self.get('hasLabel'),
 66 				accessible = self.get('accessible'),
 67 				getLabelFunc = self.get('getLabelFunc'),
 68 				labelText;
 69 			if (target.length === 0) {
 70 				return false;
 71 			}
 72 			target.each(function(value, key) {
 73 				value.hide();
 74 				if (self._isDisabled(value)) {
 75 					checkbox = $(disabledHTML).insertBefore(value).attr('ks-checkbox-disabled', 'disabled').removeAttr('tabindex');
 76 				} else {
 77 					// 如果本身是选中的状态
 78 					checkbox = self._isChecked(value) ? $(checkedHTML) : $(html);
 79 					checkbox.insertBefore(value);
 80 				}
 81 				// 无障碍访问
 82 				try {
 83 					if (accessible) {
 84 						//优先选择函数提供的查询										
 85 						labelText = getLabelFunc ? getLabelFunc(value).html() : value.next('label').html();
 86 						checkbox.attr('aria-label', labelText);
 87 					}
 88 				} catch (e) {
 89 					S.log('html结构不符合');
 90 					return false;
 91 				}
 92 				checkboxArr.push(checkbox);
 93 			})
 94 			self.set('checkboxs', checkboxArr);
 95 		},
 96 		/**
 97 		 * 加载css
 98 		 */
 99 		_loadCss: function() {
100 			var self = this,
101 				isUseCss = self.get('isUseCss'),
102 				cssUrl = self.get('cssUrl');
103 			//加载css文件
104 			if (cssUrl !== '') {
105 				S.use(cssUrl, function(S) {});
106 			}
107 		},
108 		/**
109 		 * 根据样式返回html字符串
110 		 * @param  {Number} key 0→DEFAULT;1→CHECKED;2→DISABLED
111 		 * @return {String} 返回html
112 		 */
113 		_getHtml: function(key) {
114 			var self = this,
115 				getClass = self.get('cls'),
116 				defaultClass = getClass.init,
117 				checkedClass = getClass.checked,
118 				disabledClass = getClass.disabled,
119 				htmlStr = '<span tabindex="0" class="{defalutName} {secondName}"></span>',
120 				obj = {
121 					defalutName: defaultClass
122 				};
123 			switch (key) {
124 			case 0:
125 				obj.secondName = '';
126 				break;
127 			case 1:
128 				obj.secondName = checkedClass;
129 				break;
130 			case 2:
131 				obj.secondName = disabledClass;
132 				break;
133 			default:
134 				break;
135 			}
136 			return S.substitute(htmlStr, obj);
137 		},
138 		/**
139 		 * 绑定事件，包括mouseenter mouseleave click
140 		 */
141 		_bindEvent: function() {
142 			var self = this,
143 				checkboxs = $(self.get('checkboxs')),
144 				hoverClass = this.get('cls').hover,
145 				hasLabel = self.get('hasLabel'),
146 				targets = self.get('target'),
147 				getLabelFunc = self.get('getLabelFunc'),
148 				nextLabel;
149 			checkboxs.each(function(value, key) {
150 				value.on('mouseenter mouseleave', function(ev) {
151 					//如果本身是选中状态或者是禁用状态，则不做处理
152 					if (self._isChecked(value) || self._isDisabled(value)) {
153 						return;
154 					}
155 					//value.toggleClass('ks-checkbox-hover') 在初始化的时候就已经选中的无效
156 					switch (ev.type) {
157 					case 'mouseenter':
158 						value.addClass(hoverClass);
159 						break;
160 					case 'mouseleave':
161 						value.removeClass(hoverClass);
162 						break;
163 					default:
164 						break;
165 					}
166 					//单击				
167 				}).on('click', function() {
168 					if (self._isDisabled(value)) return;
169 					self._clickHandler.call(self, key);
170 					//按键 enter
171 				}).on('keyup', function(ev) {
172 					if (ev.keyCode === 13) {
173 						value.fire('click');
174 					}
175 				});
176 				//如果需要 label-for
177 				if (hasLabel) {
178 					try {
179 						nextLabel = getLabelFunc ? getLabelFunc($(targets[key])) : value.next('label');
180 						//将label绑定和checkbox一样的事件
181 						nextLabel.on('click', function() {
182 							value.fire('click');
183 						}).on('mouseenter', function() {
184 							value.fire('mouseenter');
185 						}).on('mouseleave', function() {
186 							value.fire('mouseleave');
187 						})
188 					} catch (e) {
189 						S.log('html结构不符合');
190 						return false;
191 					}
192 				}
193 			})
194 		},
195 		/**
196 		 * 单击事件
197 		 * @param  {Number} targetIndex 数组checkboxs的索引
198 		 */
199 		_clickHandler: function(targetIndex) {
200 			var that = this,
201 				targets = that.get('target'),
202 				checkbox = $(that.get('checkboxs')[targetIndex]),
203 				checkedClass = this.get('cls').checked;
204 			//触发原生dom节点的点击事件
205 			$(targets[targetIndex]).fire('click');
206 			checkbox.toggleClass(checkedClass);
207 		},
208 		/**
209 		 * 判断是否处于禁用状态
210 		 * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
211 		 * @return {Boolean}
212 		 */
213 		_isDisabled: function(target) {
214 			var protoDisabled = $(target).attr('disabled'),
215 				modifyDisabled = $(target).attr('ks-checkbox-disabled');
216 			return protoDisabled === 'disabled' || modifyDisabled === 'disabled';
217 		},
218 		/**
219 		 * 判断是否处于禁用状态
220 		 * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
221 		 * @return {Boolean}
222 		 */
223 		_isChecked: function(target) {
224 			var protoChecked = $(target).prop('checked'),
225 				hasCheckedClass = $(target).hasClass(this.get('cls').checked);
226 			return protoChecked || hasCheckedClass;
227 		},
228 		/**
229 		 * 根据索引禁用单个checkbox
230 		 * @param {Number} targetElement 数组checkboxs的索引
231 		 */
232 		setDisabled: function(targetElement) {
233 			var self = this,
234 				checkboxs = self.get('checkboxs'),
235 				targets = self.get('target'),
236 				checkbox, target, getClass = this.get('cls'),
237 				checkedClass = getClass.checked,
238 				disabledClass = getClass.disabled,
239 				hoverClass = getClass.hover;
240 			//如果传递的是数字索引
241 			if (typeof targetElement === 'number') {
242 				checkbox = $(checkboxs[targetElement]);
243 				target = $(targets[targetElement]);
244 				checkbox.attr('ks-checkbox-disabled', 'disabled').removeClass(checkedClass + ' ' + hoverClass).addClass(disabledClass);
245 				target.attr('disabled', 'disabled');
246 			}
247 			checkbox.removeAttr('tabindex');
248 			return self;
249 		},
250 		/**
251 		 * 根据索引恢复单个checkbox
252 		 * @param {Number} targetElement 数组checkboxs的索引
253 		 */
254 		setAvailabe: function(targetElement) {
255 			var self = this,
256 				checkboxs = self.get('checkboxs'),
257 				targets = self.get('target'),
258 				checkbox, target, disabledClass = this.get('cls').disabled;
259 			//如果传递的是数字索引
260 			if (typeof targetElement === 'number') {
261 				checkbox = $(checkboxs[targetElement]);
262 				target = $(targets[targetElement]);
263 				checkbox.removeAttr('ks-checkbox-disabled', 'disabled').removeClass(disabledClass);
264 				target.removeAttr('disabled', 'disabled');
265 			}
266 			checkbox.attr('tabindex', '0');
267 			return self;
268 		},
269 		/**
270 		 * 模拟的checkbox全选
271 		 * @return {Object} return self
272 		 */
273 		selectAll: function() {
274 			var self = this,
275 				checkboxs = self.get('checkboxs');
276 			$(checkboxs).each(function(value, key) {
277 				if (self._isChecked(value)) return;
278 				value.fire('click');
279 			})
280 			return self;
281 		},
282 		/**
283 		 * 清空模拟的checkbox
284 		 * @return {Object} return self
285 		 */
286 		resetAll: function() {
287 			var self = this,
288 				checkboxs = self.get('checkboxs'),
289 				hoverClass = self.get('cls').hover;
290 			$(checkboxs).each(function(value, key) {
291 				if (!self._isChecked(value)) return;
292 				value.fire('click').removeClass(hoverClass);
293 			})
294 			return self;
295 		},
296 		/**
297 		 * 获取所有选中的checkboxs索引
298 		 * @return {Array} 选中的checkboxs索引数组集合
299 		 */
300 		getAllChecked: function() {
301 			var self = this,
302 				target = self.get('target'),
303 				checkedArr = [],
304 				value;
305 			for (i = 0, len = target.length; i < len; i++) {
306 				value = $(target[i]);
307 				if (self._isDisabled(value)) {
308 					continue;
309 				}
310 				if (self._isChecked(value)) {
311 					checkedArr.push(i);
312 				}
313 			}
314 			return checkedArr;
315 		}
316 	}, {
317 		ATTRS: /** @lends Checkbox.prototype*/
318 		{
319 			/**
320 			 * 配置的目标,选择器的字符串
321 			 * @type {String}
322 			 */
323 			target: {
324 				value: '',
325 				setter: function(v) {
326 					return $(v);
327 				},
328 				getter: function(v) {
329 					return $(v);
330 				}
331 			},
332 			/**
333 			 * 美化后的checkbox数组
334 			 * @type {Array}
335 			 * @default []
336 			 */
337 			checkboxs: {
338 				value: []
339 			},
340 			/**
341 			 * 一组样式名
342 			 * @type {Object}
343 			 * @default cls:{init: 'ks-checkbox',checked: 'ks-checkbox-checked',disabled: 'ks-checkbox-disabled',hover: 'ks-checkbox-hover'}
344 			 */
345 			cls: {
346 				value: {
347 					init: 'ks-checkbox',
348 					checked: 'ks-checkbox-checked',
349 					disabled: 'ks-checkbox-disabled',
350 					hover: 'ks-checkbox-hover'
351 				}
352 			},
353 			/**
354 			 * css模块路径
355 			 * @default gallery/form/1.3/checkbox/themes/default/style2.css
356 			 */
357 			cssUrl: {
358 				value: 'gallery/form/1.3/checkbox/themes/default/style2.css'
359 			},
360 			/**
361 			 * 是否需要label for的对应
362 			 * @default false
363 			 */
364 			hasLabel: {
365 				value: false
366 			},
367 			/**
368 			 * 通过checkbox查找对应label的函数
369 			 * @default undefined			 
370 			 * @type {Function}
371 			 */
372 			getLabelFunc: {
373 				value: undefined,
374 				setter: function(v) {
375 					return v;
376 				},
377 				getter: function(v) {
378 					return v;
379 				}
380 			},
381 			/**
382 			 * 无障碍，建立在label的基础上,查找label里面的innerHTML
383 			 * @default false
384 			 */
385 			accessible: {
386 				value: false
387 			}
388 		}
389 	})
390 	return Checkbox;
391 }, {
392 	requires: ['base', 'node']
393 });
