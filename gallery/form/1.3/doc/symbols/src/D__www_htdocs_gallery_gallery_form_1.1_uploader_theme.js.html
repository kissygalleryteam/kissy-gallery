1 /**
  2  * @fileoverview 上传组件主题基类
  3  * @author 剑平（明河）<minghe36@126.com>

  5 
  6 KISSY.add('gallery/form/1.3/uploader/theme', function (S, Node, Base) {
  7     var EMPTY = '', $ = Node.all,
  8         //主题样式名前缀
  9         classSuffix = {BUTTON:'-button', QUEUE:'-queue'};
 10 
 11     /**
 12      * @name Theme
 13      * @class 上传组件主题基类
 14      * @constructor
 15      * @extends Base
 16      * @requires Queue
 17      */
 18     function Theme(config) {
 19         var self = this;
 20         //调用父类构造函数
 21         Theme.superclass.constructor.call(self, config);
 22         self._LoaderCss();
 23     }
 24 
 25     S.extend(Theme, Base, /** @lends Theme.prototype*/{
 26         /**
 27          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 28          * @param {Uploader} uploader
 29          */
 30         afterUploaderRender:function (uploader) {
 31 
 32         },
 33         /**
 34          * 获取状态容器
 35          * @param {KISSY.NodeList} target 文件的对应的dom（一般是li元素）
 36          * @return {KISSY.NodeList}
 37          */
 38         _getStatusWrapper:function (target) {
 39             return target.children('.J_FileStatus');
 40         },
 41         /**
 42          * 控制文件对应的li元素的显影
 43          * @param {Boolean} isShow 是否认显示
 44          * @param {NodeList} target li元素
 45          * @param {Function} callback 回调
 46          */
 47         displayFile:function (isShow, target, callback) {
 48             var self = this,
 49                 duration = self.get('duration');
 50             if (!target || !target.length) return false;
 51             target[isShow && 'fadeIn' || 'fadeOut'](duration, function () {
 52                 callback && callback.call(self);
 53             });
 54         },
 55         /**
 56          * 文件处于等待上传状态时触发
 57          */
 58         _waitingHandler:function (ev) {
 59 
 60         },
 61         /**
 62          * 文件处于开始上传状态时触发
 63          */
 64         _startHandler:function (ev) {
 65 
 66         },
 67         /**
 68          * 文件处于正在上传状态时触发
 69          */
 70         _progressHandler:function (ev) {
 71 
 72         },
 73         /**
 74          * 文件处于上传成功状态时触发
 75          */
 76         _successHandler:function (ev) {
 77 
 78         },
 79         /**
 80          * 文件处于上传错误状态时触发
 81          */
 82         _errorHandler:function (ev) {
 83 
 84         },
 85         /**
 86          * 从路径隐藏域抓取文件，往队列添加文件后触发
 87          * @param ev
 88          */
 89         _restoreHandler:function (ev) {
 90 
 91         },
 92         /**
 93          * uploader实例化后执行
 94          */
 95         _UploaderRender:function (callback) {
 96             var self = this;
 97             self._initQueue();
 98             self._addThemeCssName();
 99             //加载插件
100             self._loadPlugins(callback);
101         },
102         /**
103          * 将主题名写入到队列和按钮目标容器，作为主题css样式起始
104          */
105         _addThemeCssName:function () {
106             var self = this, name = self.get('name'),
107                 queue = self.get('queue'),
108                 $queueTarget = $(self.get('queueTarget')),
109                 button = self.get('button'),
110                 $btn;
111             if (name == EMPTY || !queue || !$queueTarget.length) return false;
112             $queueTarget.addClass(name + classSuffix.QUEUE);
113             if (!button) return false;
114             $btn = button.get('target');
115             $btn.addClass(name + classSuffix.BUTTON);
116         },
117         /**
118          * 初始化队列
119          * @return {Queue}
120          */
121         _initQueue:function () {
122             var self = this, queue = self.get('queue');
123             queue.set('fnAdd',function(index, file){
124                 return self._addCallback(index, file);
125             });
126             queue.set('theme', self);
127             queue.on('add', self._addFileHandler, self);
128             queue.on('remove', self._removeFileHandler, self);
129             queue.on('statusChange', function (ev) {
130                 self._setStatusVisibility(ev);
131             });
132             return queue;
133         },
134         /**
135          * 加载css文件
136          */
137         _LoaderCss:function () {
138             var self = this,
139                 cssUrl = self.get('cssUrl');
140             //加载css文件
141             if (cssUrl == EMPTY) return false;
142             S.use(cssUrl, function () {
143                 S.log(cssUrl + '加载成功！');
144             });
145         },
146         /**
147          * 向队列添加完文件后触发的回调函数（在add事件前触发）
148          * @param {Number} index 文件索引值
149          * @param {Object} file 文件数据
150          * @return {Object} file
151          */
152         _addCallback:function (index, file) {
153             var self = this,
154                 queue = self.get('queue'),
155                 $target = self._appendFileDom(file);
156             //将状态层容器写入到file数据
157             queue.updateFile(index, {
158                 target:$target,
159                 statusWrapper:self._getStatusWrapper($target)
160             });
161             //更换文件状态为等待
162             queue.fileStatus(index,'waiting');
163             self.displayFile(true, $target);
164             //给li下的按钮元素绑定事件
165             // TODO 这里的绑定事件应该只是imageUploader这个主题的吧，不应该放在公共的Theme下
166             self._bindTriggerEvent(index, file);
167             return queue.getFile(index);
168         },
169         /**
170          * 删除队列中的文件后触发的监听器
171          */
172         _removeFileHandler:function (ev) {
173             var self = this,
174                 file = ev.file;
175             self.displayFile(false, file.target);
176         },
177         /**
178          * 给删除、上传、取消等按钮元素绑定事件
179          * TODO 这个是不是也应该放在imageUploader里面呢？
180          * @param {Number} index 文件索引值
181          * @param {Object} 文件数据
182          */
183         _bindTriggerEvent:function (index, file) {
184             var self = this,
185                 queue = self.get('queue'),
186                 uploader = self.get('uploader'),
187                 //文件id
188                 fileId = file.id,
189                 //上传链接
190                 $upload = $('.J_Upload_' + fileId),
191                 //取消链接
192                 $cancel = $('.J_Cancel_' + fileId),
193                 //删除链接
194                 $del = $(".J_Del_" + fileId);
195             //点击上传
196             $upload.on('click', function (ev) {
197                 ev.preventDefault();
198                 if (!S.isObject(uploader)) return false;
199                 uploader.upload(index);
200             });
201             //点击取消
202             $cancel.on('click', function (ev) {
203                 ev.preventDefault();
204                 uploader.cancel(index);
205             });
206             //点击删除
207             $del.on('click', function (ev) {
208                 ev.preventDefault();
209                 //删除队列中的文件
210                 queue.remove(fileId);
211             });
212         },
213         /**
214          * 设置状态层的可见性
215          * @param ev
216          */
217         _setStatusVisibility:function (ev) {
218             var $statusWrapper = ev.file.statusWrapper, $status,
219                 file = ev.file, status = file.status;
220             if (!$statusWrapper || !$statusWrapper.length) {
221                 S.log('状态容器层不存在！');
222                 return false;
223             }
224             $status = $statusWrapper.children('.status');
225             $status.hide();
226             $statusWrapper.children('.' + status + '-status').show();
227         },
228         /**
229          * 当队列添加完文件数据后向队列容器插入文件信息DOM结构
230          * @param {Object} fileData 文件数据
231          * @return {KISSY.NodeList}
232          */
233         _appendFileDom:function (fileData) {
234             var self = this, tpl = self.get('fileTpl'),
235                 $target = $(self.get('queueTarget')),
236                 hFile;
237             if (!$target.length) return false;
238             hFile = S.substitute(tpl, fileData);
239             return $(hFile).hide().appendTo($target).data('data-file', fileData);
240         },
241         /**
242          * 根据插件配置加载插件
243          */
244         _loadPlugins:function(callback){
245             var self = this,
246                 plugins = self.get('plugins'),
247                 oPlugin = self.get('oPlugin'),
248                 //模块路径前缀
249                 modPrefix = 'gallery/form/1.3/uploader/plugins/',
250                 mods = [];
251             if(!plugins.length){
252                 callback && callback.call(self,oPlugin);
253                 return false;
254             }
255             //拼接模块路径
256             S.each(plugins,function(plugin){
257                 mods.push(modPrefix+plugin+'/' +plugin);
258             });
259             S.use(mods.join(','),function(){
260                  S.each(arguments,function(arg,i){
261                      // 类排除S
262                      if(i>=1) oPlugin[plugins[i-1]] = arg;
263                  });
264                 self.set('oPlugin',oPlugin);
265                 callback && callback.call(self,oPlugin);
266             })
267         }
268     }, {ATTRS:/** @lends Theme.prototype*/{
269         /**
270          *  主题名
271          * @type String
272          * @default ""
273          */
274         name:{value:EMPTY},
275         /**
276          * css模块路径
277          * @type String
278          * @default ""
279          */
280         cssUrl:{value:EMPTY},
281         /**
282          * 队列使用的模板
283          * @type String
284          * @default ""
285          */
286         fileTpl:{value:EMPTY },
287         /**
288          * 需要加载的插件，需要手动实例化
289          * @type Array
290          * @default []
291          */
292         plugins:{value:[]},
293         /**
294          * 插件类集合
295          * @type Array
296          * @default []
297          */
298         oPlugin:{value:{}},
299         /**
300          * 队列目标元素（一般是ul），队列的实例化过程在Theme中
301          * @type String
302          * @default ""
303          */
304         queueTarget:{value:EMPTY},
305         /**
306          * 动画速度
307          * @type Number
308          * @default 0.3
309          */
310         duration:{value:0.3},
311         /**
312          * Queue（上传队列）实例
313          * @type Queue
314          * @default ""
315          */
316         queue:{value:EMPTY},
317         /**
318          * Uploader 上传组件实例
319          * @type Uploader
320          * @default ""
321          */
322         uploader:{value:EMPTY},
323         /**
324          * Button 按钮实例（_init()下并不存在）
325          * @type Button
326          * @default ""
327          */
328         button:{value:EMPTY},
329         /**
330          * Auth（上传验证）实例
331          * @type Auth
332          * @default ""
333          */
334         auth:{value:EMPTY}
335     }});
336     return Theme;
337 }, {requires:['node', 'base']});
