1 /**
  2  * @fileoverview 默认主题
  3  * @author 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/themes/default/index', function (S, Node, Theme) {
  6     var EMPTY = '', $ = Node.all;
  7 
  8     /**
  9      * @name DefaultTheme
 10      * @class 默认主题
 11      * @constructor
 12      * @extends Theme
 13      * @requires Theme
 14      * @requires  ProgressBar
 15      * @author 剑平（明河）<minghe36@126.com>
 16      */
 17     function DefaultTheme(config) {
 18         var self = this;
 19         //调用父类构造函数
 20         DefaultTheme.superclass.constructor.call(self, config);
 21     }
 22 
 23     S.extend(DefaultTheme, Theme, /** @lends DefaultTheme.prototype*/{
 24         /**
 25          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 26          * @param {Uploader} uploader
 27          */
 28         afterUploaderRender:function (uploader) {
 29 
 30         },
 31         /**
 32          * 获取状态容器
 33          * @param {KISSY.NodeList} target 文件的对应的dom（一般是li元素）
 34          * @return {KISSY.NodeList}
 35          */
 36         _getStatusWrapper:function(target){
 37             return target.children('.J_FileStatus');
 38         },
 39         /**
 40          * 文件处于等待上传状态时触发
 41          */
 42        _waitingHandler:function(ev){
 43 
 44         },
 45         /**
 46          * 文件处于开始上传状态时触发
 47          */
 48         _startHandler : function(ev){
 49              var self = this,
 50                  uploader = ev.uploader,
 51                  index = ev.index,
 52                  queue = self.get('queue'),
 53                  //上传方式
 54                  uploadType = uploader.get('type'),
 55                  $progressBar = $('.J_ProgressBar_' + ev.id);
 56             //如果是ajax或flash异步上传，加入进度条
 57             if(uploadType == 'ajax' || uploadType == 'flash'){
 58                 var ProgressBar = self.get('oPlugin').progressBar,
 59                     progressBar = new ProgressBar($progressBar);
 60                 progressBar.render();
 61                 self.set('progressBar',progressBar);
 62                 //将进度条实例写入到队列的文件数据上备用
 63                 queue.updateFile(index,{progressBar:progressBar});
 64             }
 65         },
 66         /**
 67          * 文件处于正在上传状态时触发
 68          */
 69         _progressHandler:function(ev){
 70             var file = ev.file,
 71                 //已加载字节数
 72                 loaded = ev.loaded,
 73                 //总字节数
 74                 total = ev.total,
 75                 val = Math.ceil((loaded/total) * 100),
 76                 progressBar = file.progressBar;
 77             if(!progressBar) return false;
 78             //处理进度
 79             progressBar.set('value',val);
 80         },
 81         /**
 82          * 文件处于上传成功状态时触发
 83          */
 84         _successHandler:function(ev){
 85 
 86         },
 87         /**
 88          * 文件处于上传错误状态时触发
 89          */
 90         _errorHandler:function(ev){
 91             var msg = ev.msg,
 92                 id = ev.id;
 93             //打印错误消息
 94             $('.J_ErrorMsg_' + id).html(msg);
 95         }
 96     }, {ATTRS:/** @lends DefaultTheme.prototype*/{
 97         /**
 98          *  主题名（文件名）
 99          * @type String
100          * @default "defaultTheme"
101          */
102         name:{value:'defaultTheme'},
103         /**
104          * css模块路径
105          * @type String
106          * @default "gallery/form/1.3/uploader/themes/default/style.css"
107          */
108         cssUrl:{value:'gallery/form/1.3/uploader/themes/default/style.css'},
109         /**
110          * 队列使用的模板
111          * @type String
112          * @default ""
113          */
114         fileTpl:{value:
115             '<li id="queue-file-{id}" class="clearfix" data-name="{name}">' +
116                 '<div class="f-l sprite file-icon"></div>' +
117                 '<div class="f-l">{name}</div>' +
118                 '<div class="f-l status-wrapper J_FileStatus">' +
119                     '<div class="status waiting-status">等待上传，<a class="J_Upload_{id}" href="#Upload">点此上传</a> </div>' +
120                     '<div class="status start-status progress-status clearfix">' +
121                         '<div class="J_ProgressBar_{id} f-l uploader-progress"><img class="loading" src="http://img01.taobaocdn.com/tps/i1/T1F5tVXjRfXXXXXXXX-16-16.gif" alt="loading" /></div>' +
122                         ' <a  class="J_Cancel_{id} f-l upload-cancel" href="#uploadCancel">取消</a>' +
123                     '</div> ' +
124                     ' <div class="status success-status"><a href="#fileDel" class="J_Del_{id}">删除</a></div>' +
125                     '<div class="status cancel-status">已经取消上传，<a href="#reUpload" id="J_ReUpload_{id}" class="J_Upload_{id}">点此重新上传</a> </div>' +
126                     '<div class="status error-status upload-error"><span class="J_ErrorMsg_{id}"></span><a href="#fileDel" class="J_Del_{id}">删除</a></div>' +
127                 '</div>' +
128                 '</li>'
129         },
130         /**
131          * 需要加载的插件，需要手动实例化
132          * @type Array
133          * @default [''progressBar'] 进度条
134          */
135         plugins:{
136             value:['progressBar']
137         }
138     }});
139     return DefaultTheme;
140 }, {requires:['node', '../../theme']});
