1 /**
  2  * @fileoverview 存储文件路径信息的隐藏域
  3  * @author: 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/urlsInput',function(S, Node, Base) {
  6     var EMPTY = '',$ = Node.all,LOG_PREFIX = '[uploader-urlsInput]:';
  7     /**
  8      * @name UrlsInput
  9      * @class 存储文件路径信息的隐藏域
 10      * @constructor
 11      * @extends Base
 12      * @param {String} wrapper 容器钩子
 13      * @param {Object} config 组件配置（下面的参数为配置项，配置会写入属性，详细的配置说明请看属性部分）
 14      * @param {String} config.name *，隐藏域名称，当此name的隐藏域不存在时组件会创建一个
 15      * @param {String} config.split  多个路径间的分隔符
 16      * @param {String} config.tpl   隐藏域模板
 17      *
 18      */
 19     function UrlsInput(wrapper, config) {
 20         var self = this;
 21         //调用父类构造函数
 22         UrlsInput.superclass.constructor.call(self, config);
 23         self.set('wrapper', $(wrapper));
 24     }
 25 
 26     S.mix(UrlsInput, /**@lends UrlsInput*/ {
 27         /**
 28          * 隐藏域模板， '<input type="hidden" id="{name}" name="{name}" value="{value}" />'
 29          *
 30          */
 31         TPL : '<input type="hidden" id="{name}" name="{name}" value="{value}" />'
 32     });
 33     //继承于Base，属性getter和setter委托于Base处理
 34     S.extend(UrlsInput, Base, /** @lends UrlsInput.prototype*/{
 35         /**
 36          * 运行组件，实例化类后必须调用render()才真正运行组件逻辑
 37          * @return {UrlsInput}
 38          */
 39         render : function() {
 40             var self = this,$wrapper = self.get('wrapper'),
 41                 name = self.get('name'),
 42                 elInput = document.getElementsByName(name)[0];
 43             if (!S.isObject($wrapper)) {
 44                 S.log(LOG_PREFIX + 'container参数不合法！');
 45                 return false;
 46             }
 47             //如果已经存在隐藏域，那么不自动创建
 48             if(elInput){
 49             	S.log(LOG_PREFIX + 'urls input found');
 50                 self.set('input',$(elInput));
 51             }else{
 52                 self._create();
 53             }
 54             return self;
 55         },
 56         /**
 57          * 向路径隐藏域添加路径
 58          * @param {String} url 路径
 59          * @return {UrlsInput}
 60          */
 61         add : function(url){
 62             if(!S.isString(url)){
 63                 S.log(LOG_PREFIX + 'add()的url参数不合法！');
 64                 return false;
 65             }
 66             var self = this,urls = self.get('urls'),
 67                 //判断路径是否已经存在
 68                 isExist = self.isExist(url);
 69             //TODO:第一个路径会出现为空的情况，日后完善
 70             if(urls[0] == EMPTY) urls = [];
 71             if(isExist){
 72                 S.log(LOG_PREFIX + 'add()，文件路径已经存在！');
 73                 return self;
 74             }
 75             urls.push(url);
 76             self.set('urls',urls);
 77             self._val();
 78             return self;
 79         },
 80         /**
 81          * 删除隐藏域内的指定路径
 82          * @param {String} url 路径
 83          * @return {Array} urls 删除后的路径
 84          */
 85         remove : function(url){
 86             if(!url) return false;
 87             var self = this,urls = self.get('urls'),
 88                 isExist = self.isExist(url) ,
 89                 reg = new RegExp(url);
 90             if(!isExist){
 91                 S.log(LOG_PREFIX + 'remove()，不存在该文件路径！');
 92                 return false;
 93             }
 94             urls = S.filter(urls,function(sUrl){
 95                 return !reg.test(sUrl);
 96             });
 97             self.set('urls',urls);
 98             self._val();
 99             return urls;
100         },
101         /**
102          * 解析当前input的值，取得文件路径
103          * @return {Array}
104          */
105         parse: function(){
106         	var self = this,
107         		input = self.get('input');
108     		if(input){
109     			var urls = $(input).val(),
110     				split = self.get('split'),
111     				files;
112     			files = urls.split(split);
113                 self.set('urls',files);
114     			return files;
115     		}else{
116     			S.log(LOG_PREFIX + 'cannot find urls input.');
117     			return [];
118     		}
119         },
120         /**
121          * 设置隐藏域的值
122          * @return {String} 
123          */
124         _val : function(){
125             var self = this,urls = self.get('urls'),
126                 $input = self.get('input'),
127                 //多个路径间的分隔符
128                 split = self.get('split'),
129                 sUrl = urls.join(split);
130             $input.val(sUrl);
131             return sUrl;
132         },
133         /**
134          * 是否已经存在指定路径
135          * @param {String} url 路径
136          * @return {Boolean}
137          */
138         isExist : function(url){
139             var self = this,b = false,urls = self.get('urls'),
140                 reg = new RegExp(url);
141             if(!urls.length) return false;
142             S.each(urls,function(val){
143                 if(reg.test(val)){
144                     return b = true;
145                 }
146             });
147             return b;
148         },
149         /**
150          * 创建隐藏域
151          */
152         _create : function() {
153             var self = this,
154             	container = self.get('wrapper'),
155                 tpl = self.get('tpl'),
156                 name = self.get('name'), 
157                 urls = self.get('urls'),
158                 input;
159             if(!container || container.length <= 0){
160             	S.log(LOG_PREFIX + 'UrlsInput container not specified!', 'warn');
161             	return false;
162             }
163             if (!S.isString(tpl) || !S.isString('name')){
164                 S.log(LOG_PREFIX + '_create()，tpl和name属性不合法！');
165                 return false;
166             }
167             input = $(S.substitute(tpl, {name : name,value : urls}));
168             container.append(input);
169             self.set('input', input);
170             S.log(LOG_PREFIX + 'input created.');
171             return input;
172         }
173 
174     }, {ATTRS : /** @lends UrlsInput.prototype*/{
175         /**
176          * 隐藏域名称
177          * @type String
178          * @default ""
179          */
180         name : {value : EMPTY},
181         /**
182          * 文件路径
183          * @type Array
184          * @default []
185          */
186         urls : { value : [] },
187         /**
188          * input模板
189          * @type String
190          * @default  '<input type="hidden" id="{name}" name="{name}" value="{value}" />'
191          */
192         tpl : {value : UrlsInput.TPL},
193         /**
194          * 多个路径间的分隔符
195          * @type String
196          * @default ","
197          */
198         split : {value : ',',
199             setter : function(v){
200                 var self = this;
201                 self._val();
202                 return v;
203             }
204         },
205         /**
206          * 文件路径隐藏input
207          * @type KISSY.Node
208          * @default ""
209          */
210         input : {value : EMPTY},
211         /**
212          * 隐藏域容器
213          *@type KISSY.Node
214          * @default ""
215          */
216         wrapper : {value : EMPTY}
217     }});
218 
219     return UrlsInput;
220 }, {requires:['node','base']});
