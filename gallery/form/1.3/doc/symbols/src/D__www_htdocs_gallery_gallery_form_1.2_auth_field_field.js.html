  6 KISSY.add('gallery/form/1.3/auth/field/field', function (S, Event, Base, JSON, DOM, Factory,
  7                                                          Rule, PropertyRule, Msg, Utils, undefined) {
  8 
  9     var EMPTY = '',
 10         CONFIG_NAME = 'data-valid';
 11 
 12     /**
 13      * field默认配置
 14      * @type {Object}
 15      */
 16     var defaultConfig = {
 17         event:'blur',
 18         style:{
 19             'success':'ok',
 20             'error':'error'
 21         }
 22     };
 23 
 24     var Field = function (el, config) {
 25         var self = this;
 26 
 27         self._validateDone = {};
 28         //储存上一次的校验结果
 29         self._cache = {};
 30 
 31         /**
 32          * 配置有3个地方，属性，new的参数，默认参数
 33          */
 34         //初始化json配置
 35         if (el && DOM.attr(el, CONFIG_NAME)) {
 36             var cfg = DOM.attr(el, CONFIG_NAME);
 37 
 38             cfg = Utils.toJSON(cfg);
 39             //把所有伪属性都当作rule处理
 40             var propertyConfig = {
 41                 rules:cfg
 42             };
 43 
 44             config = S.merge(propertyConfig, config);
 45         }
 46 
 47         config = S.merge(defaultConfig, config);
 48 
 49         self._cfg = config || {};
 50         //保存rule的集合
 51         self._storage = {};
 52 
 53         self._init(el);
 54 
 55         Field.superclass.constructor.call(self);
 56 
 57         return self;
 58     };
 59 
 60     S.extend(Field, Base, {
 61         _init:function (el) {
 62             var self = this,
 63                 _cfg = self._cfg,
 64                 _el = S.one(el),
 65                 _ruleCfg = S.merge({}, _cfg.rules);
 66 
 67 
 68             //如果为checkbox/radio则保存为数组
 69             if (['checkbox','radio'].indexOf(_el.attr('type')) > -1) {
 70                 var form = _el.getDOMNode().form, elName = _el.attr('name');
 71                 var els = [];
 72                 S.each(document.getElementsByName(elName), function(item) {
 73                     if (item.form == form) {
 74                         els.push(item);
 75                     }
 76                 });
 77                 self.set('el', els);
 78             } else {
 79                 self.set('el', el);
 80             }
 81 
 82             var resetAfterValidate = function () {
 83                 self.fire('afterFieldValidation');
 84             };
 85 
 86             //msg init
 87             if (self._cfg.msg) {
 88                 self._msg = new Msg(_el, self._cfg.msg);
 89                 var style = self._cfg.style;
 90 
 91                 self.on('afterRulesValidate', function (ev) {
 92                     var result = ev.result,
 93                         curRule = ev.curRule,
 94                         msg = self._cache[curRule].msg || EMPTY;
 95 
 96                     //这里的value还没被当前覆盖
 97                     if (self.get('result') !== result || self.get('msg') !== msg) {
 98                         if (msg) {
 99                             self._msg.show({
100                                 style:result ? style['success'] : style['error'],
101                                 msg:msg
102                             });
103                         } else {
104                             self._msg.hide();
105                         }
106                     }
107                 });
108             }
109 
110             //监听校验结果
111             self.on('afterRulesValidate', function (ev) {
112                 var result = ev.result,
113                     curRule = ev.curRule,
114                     msg = self._cache[curRule].msg || EMPTY;
115 
116                 self.set('result', result);
117                 self.set('message', msg);
118 
119                 self.fire('validate', {
120                     result:result,
121                     msg:msg,
122                     errRule:result ? '' : curRule
123                 });
124 
125                 //校验结束
126                 self.fire('afterValidate');
127                 resetAfterValidate();
128             });
129 
130             //add html property
131             S.each(Factory.HTML_PROPERTY, function (item) {
132 
133                 if (_el.hasAttr(item)) {
134                     //从工厂中创建属性规则
135                     var rule = Factory.create(item, {
136                         //属性的value必须在这里初始化
137                         propertyValue:_el.attr(item),
138                         el:self.get('el'), //bugfix for change value
139                         msg:_ruleCfg[item]
140                     });
141 
142                     self.add(item, rule);
143                 }
144             });
145 
146             //add custom rule
147             S.each(_ruleCfg, function(ruleCfg, name){
148                 if(!self._storage[name] && Factory.rules[name]) {
149                     //如果集合里没有，但是有配置，可以认定是自定义属性，入口为form.add
150                     var rule = Factory.create(name, {
151                         el:self.get('el'), //bugfix for change value
152                         msg:ruleCfg
153                     });
154 
155                     self.add(name, rule);
156                 }
157             });
158 
159             //element event bind
160             if (_cfg.event != 'none') {
161                 Event.on(self.get('el'), _cfg.event || Utils.getEvent(_el), function (ev) {
162                     self.validate();
163                 });
164             }
165 
166         },
167 
168         add:function (name, rule, cfg) {
169             var self = this,
170                 _storage = self._storage;
171 
172             if (rule instanceof PropertyRule || rule instanceof Rule) {
173                 _storage[name] = rule;
174             } else if(S.isFunction(rule)) {
175                 _storage[name] = new Rule(name, rule, {
176                     el:self._el
177                     //TODO args
178                 });
179             }
180 
181             if(_storage[name]) {
182                 _storage[name].on('validate', function (ev) {
183                     S.log('[after rule validate]: name:' + ev.name + ',result:' + ev.result + ',msg:' + ev.msg);
184                     //set cache
185                     self._cache[ev.name]['result'] = ev.result;
186                     self._cache[ev.name]['msg'] = ev.msg;
187                 });
188             }
189 
190             this._cache[name] = {};
191 
192             return self;
193         },
194 
195         remove:function (name) {
196             var _storage = this._storage;
197             delete _storage[name];
198             delete this._cache[name];
199 
200             return this;
201         },
202 
203         /**
204          *
205          * @param name
206          * @param cfg {Object}
207          * @param cfg.args
208          * @param cfg.msg
209          *
210          * @return {Boolean}
211          */
212         validate:function (name, cfg) {
213             var result = true,
214                 self = this,
215                 _storage = self._storage,
216                 cfg = cfg||{},
217                 curRule = EMPTY;
218 
219             if (name) {
220                 if (_storage[name]) {
221                     //校验开始
222                     self.fire('beforeValidate');
223 
224                     result = _storage[name].validate(cfg.args);
225                     curRule = name;
226                 }
227             } else {
228                 //校验开始
229                 self.fire('beforeValidate');
230 
231                 for (var key in _storage) {
232                     curRule = key;
233                     if (!_storage[key].validate(cfg.args)) {
234                         result = false;
235                         break;
236                     }
237                 }
238             }
239 
240             // 保证有规则才触发
241             if (curRule) {
242                 self.fire('afterRulesValidate', {
243                     result:result,
244                     curRule:curRule
245                 });
246             }
247 
248             //TODO GROUPS
249 
250             return result;
251         }
252     }, {
253         ATTRS:{
254             message:{
255                 value:EMPTY
256             },
257             result:{},
258             el:{}
259         }
260     });
261 
262     return Field;
263 }, {
264     requires:[
265         'event',
266         'base',
267         'json',
268         'dom',
269         '../rule/ruleFactory',
270         '../rule/rule',
271         '../rule/html/propertyRule',
272         '../msg/base',
273         '../utils'
274     ]
275 });
