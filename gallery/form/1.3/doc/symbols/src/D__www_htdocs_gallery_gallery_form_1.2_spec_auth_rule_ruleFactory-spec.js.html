  6 describe('rule factory test suite', function() {
  7     KISSY.use('form/validation/rule/html/propertyFactory', function (S, PropertyFactory) {
  8         it('create factory', function() {
  9             var factory = new PropertyFactory();
 10             expect(factory).toBeDefined();
 11         });
 12 
 13         it('create rule', function() {
 14             var factory = new PropertyFactory();
 15             var rule = factory.create('required');
 16             var result = rule.validate();
 17             expect(result).toBeFalsy();
 18         });
 19 
 20         it('use init args', function() {
 21             var factory = new PropertyFactory();
 22             var rule = factory.create('required', {
 23                 args:['', 1]
 24             });
 25             var result = rule.validate();
 26             expect(result).toBeTruthy();
 27         });
 28 
 29         it('init msg', function() {
 30             var factory = new PropertyFactory();
 31             var rule = factory.create('required', {
 32                 msg:'fail'
 33             });
 34             var msg = '';
 35             var result = rule.on('validate', function(e) {
 36                 msg = e.msg;
 37             }).validate();
 38             expect(result).toBeFalsy();
 39             expect(msg).toEqual('fail');
 40         });
 41     });
 42 });
