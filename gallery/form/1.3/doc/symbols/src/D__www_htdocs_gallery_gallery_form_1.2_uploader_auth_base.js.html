1 /**
  2  * @fileoverview 文件上传验证
  3  * @author: 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/auth/base', function (S, Node,Base) {
  6     var EMPTY = '', $ = Node.all,
  7         console = console || S, LOG_PREFIX = '[uploader-auth]:';
  8 
  9     /**
 10      * @name Auth
 11      * @class 文件上传验证，可以从按钮的data-auth伪属性抓取规则配置
 12      * @constructor
 13      * @extends Base
 14      * @param {Uploader} uploader *，上传组件实例
 15      * @param {Object} config 配置
 16      */
 17     function Auth(uploader, config) {
 18         var self = this;
 19         config = S.merge({uploader:uploader}, config);
 20         Auth.superclass.constructor.call(self, config);
 21         self._init();
 22     }
 23     S.mix(Auth,/** @lends Auth*/{
 24         /**
 25          * 事件
 26          */
 27         event : {
 28             ERROR : 'error'
 29         }
 30     });
 31     /**
 32      * @name Auth#error
 33      * @desc  当验证出错时触发
 34      * @event
 35      * {rule:'require',msg : rule[1],value : isRequire}
 36      * @param {String} ev.rule 规则名
 37      * @param {String} ev.msg 出错消息
 38      * @param {Boolean|String} ev.value 规则值
 39      */
 40     S.extend(Auth, Base, /** @lends Auth.prototype*/{
 41         /**
 42          * 初始化
 43          */
 44         _init:function () {
 45             var self = this, uploader = self.get('uploader'),
 46                 queue = uploader.get('queue');
 47             if (uploader == EMPTY) {
 48                 console.log(LOG_PREFIX + 'uploader不可以为空！');
 49                 return false;
 50             }
 51             self._setSwfButtonExt();
 52             queue.on('add',function(ev){
 53                 var file = ev.file;
 54                 self.testAllowExt(file);
 55                 self.testMaxSize(file);
 56                 self.testRepeat(file);
 57             });
 58             queue.on('remove',function(ev){
 59                 var file = ev.file,status = file.status;
 60                 //删除的是已经成功上传的文件，需要重新检验最大允许上传数
 61                 if(status == 'success'){
 62                     self.testMax();
 63                 }
 64             });
 65             queue.on('statusChange',function(ev){
 66                 var status = ev.status;
 67                 if(status == 'success') self.testMax();
 68             });
 69             uploader.on('error', function (ev) {
 70                 //允许继续上传文件
 71                 uploader.set('isAllowUpload', true);
 72             });
 73         },
 74         /**
 75          * 验证上传数、是否必须上传
 76          * @return {Boolean}
 77          */
 78         testAll : function(){
 79             var self = this;
 80             return self.testRequire() && self.testMax();
 81         },
 82         /**
 83          * 获取指定规则
 84          * @param {String} ruleName 规则名
 85          * @return {Array}
 86          */
 87         getRule : function(ruleName){
 88             var self = this,rules = self.get('rules');
 89             return rules[ruleName];
 90         },
 91         /**
 92          * 判断上传方式
 93          * @param type
 94          * @return {Boolean}
 95          */
 96         isUploaderType:function (type) {
 97             var self = this, uploader = self.get('uploader'),
 98                 uploaderType = uploader.get('type');
 99             return type == uploaderType;
100         },
101         /**
102          * 检验是否必须上传一个文件
103          * @return {Boolean}
104          */
105         testRequire : function(){
106             var self = this,uploader = self.get('uploader'),
107                 urlsInput = uploader.get('urlsInput'),
108                 urls = urlsInput.get('urls'),
109                 rule = self.getRule('require'),
110                 isRequire = rule ? rule[0] : false,
111                 isHasUrls = urls.length > 0;
112             if(!isRequire) return true;
113             if(!isHasUrls){
114                 S.log(LOG_PREFIX + rule[1]);
115                 self._fireUploaderError('require',rule);
116             }
117             return isHasUrls;
118         },
119         /**
120          * 测试是否是允许的文件上传类型
121          * @param {Object} file 文件对象
122          * @return {Boolean} 是否通过
123          */
124         testAllowExt:function (file) {
125             if (!S.isObject(file)) return false;
126             var self = this,
127                 fileName = file.name,
128                 allowExts = self.getRule('allowExts'),
129                 exts = [],
130                 fileExt, msg,
131                 isAllow;
132             if (!S.isArray(allowExts)) return false;
133             //扩展名数组
134             exts = self._getExts(allowExts[0].ext);
135 
136             isAllow = _isAllowUpload(fileName);
137             //如果不是支持的文件格式，出现错误
138             if(!isAllow){
139                 fileExt = _getFileExt(fileName);
140                 msg = S.substitute(allowExts[1],{ext : fileExt});
141                 self._fireUploaderError('allowExts',[allowExts[0],msg],file);
142             }
143             /**
144              * 是否允许上传
145              * @param {String} fileName 文件名
146              * @return {Boolean}
147              */
148             function _isAllowUpload(fileName) {
149                 var isAllow = false, reg;
150                 S.each(exts, function (ext) {
151                     reg = new RegExp('^.+\.' + ext + '$');
152                     //存在该扩展名
153                     if (reg.test(fileName))  return isAllow = true;
154                 });
155                 return isAllow;
156             }
157             /**
158              * 获取文件扩展名
159              * @param {String} file
160              */
161             function _getFileExt(file){
162                 var arr = file.split('.');
163                 return arr[arr.length -1];
164             }
165             return isAllow;
166         },
167         /**
168          * 检验是否达到最大允许上传数
169          * @return {Boolean}
170          */
171         testMax:function () {
172             var self = this, uploader = self.get('uploader'),
173                 queue = uploader.get('queue'),
174                 successFiles = queue.getFiles('success'),
175                 len = successFiles.length,
176                 rule = self.getRule('max'),
177                 msg;
178             if(rule){
179             	var button = uploader.get('button'),
180 	                isPass = len < rule[0];
181 	            //达到最大允许上传数
182 	            if(!isPass){
183 	                //禁用按钮
184 	                uploader.set('disabled',true);
185 	                uploader.set('isAllowUpload', false);
186                     msg = S.substitute(rule[1],{max : rule[0]});
187                     self._fireUploaderError('max',[rule[0],msg]);
188 	            }else{
189 	                button.set('disabled',false);
190 	                uploader.set('isAllowUpload', true);
191 	            }
192 	            return isPass;
193             }
194         },
195         /**
196          * 检验是否超过允许最大文件大小，留意iframe上传方式此验证无效
197          * @param {Object} file 文件对象
198          */
199         testMaxSize : function(file){
200             var self = this,
201                 size = file.size,
202                 rule = self.getRule('maxSize');
203             if(rule){
204             	var maxSize = Number(rule[0]) * 1000,
205 	                isAllow = size <= maxSize,
206 	                msg;
207 	            if(!isAllow){
208 	                msg = S.substitute(rule[1],{maxSize:S.convertByteSize(maxSize),size : file.textSize});
209                     self._fireUploaderError('maxSize',[rule[0],msg],file);
210 	            }
211 	            return isAllow;
212             }
213         },
214         /**
215          * 检验文件是否重复（检验文件名，很有可能存在误差，比如不同目录下的相同文件名会被判定为同一文件）
216          * @param {Object} file 文件对象
217          * @return {Boolean}
218          */
219         testRepeat : function(file){
220             if(!S.isObject(file)) return false;
221             var self = this,
222                 fileName = file.name,
223                 rule = self.getRule('allowRepeat');
224             if(rule){
225             	var isAllowRepeat = rule[0],
226 	                uploader = self.get('uploader'),
227 	                queue = uploader.get('queue'),
228 	                //上传成功的文件
229 	                files = queue.getFiles('success'),
230 	                isRepeat = false ;
231 	            //允许重复文件名，直接返回false
232 	            if(isAllowRepeat) return false;
233 	            S.each(files,function(f){
234 	                if(f.name == fileName){
235                         self._fireUploaderError('allowRepeat',rule,file);
236 	                    return isRepeat = true;
237 	                }
238 	            });
239 	            return isRepeat;
240             }
241         },
242         /**
243          * 设置flash按钮的文件格式过滤
244          * @return {Auth}
245          */
246         _setSwfButtonExt:function () {
247             var self = this, uploader = self.get('uploader'),
248                 allowExts = self.getRule('allowExts'),
249                 button = uploader.get('button'),
250                 isFlashType = self.isUploaderType('flash');
251             if (!isFlashType || !S.isArray(allowExts)) return false;
252             //设置文件过滤
253             button.set('fileFilters', allowExts[0]);
254             return self;
255         },
256         /**
257          * 获取扩展名，需额外添加大写扩展名
258          * @param {String} sExt 扩展名字符串，类似*.jpg;*.jpeg;*.png;*.gif;*.bmp
259          * @retunr {Array}
260          */
261         _getExts:function (sExt) {
262             if (!S.isString(sExt)) return false;
263             var exts = sExt.split(';'),
264                 uppercaseExts = [],
265                 reg = /^\*\./;
266             S.each(exts, function (ext) {
267                 ext = ext.replace(reg, '');
268                 uppercaseExts.push(ext.toUpperCase());
269             });
270             S.each(uppercaseExts,function(ext){
271                 exts.push(ext);
272             });
273             return exts;
274         },
275         /**
276          * 触发uploader的error事件
277          * @param ruleName
278          * @param rule
279          * @param file
280          */
281         _fireUploaderError:function(ruleName,rule,file){
282             var self = this,
283                 uploader = self.get('uploader'),
284                 queue = uploader.get('queue'),
285                 params = {status:-1,rule:ruleName},
286                 index = -1;
287             if(file){
288                 S.mix(params,{file:file});
289                 index = queue.getFileIndex(params.file.id);
290             }
291             if(rule) S.mix(params,{msg : rule[1],value : rule[0]});
292             queue.fileStatus(index, 'error', params);
293             self.fire(Auth.event.ERROR,params);
294             uploader.fire('error',params);
295         }
296     }, {ATTRS:/** @lends Auth.prototype*/{
297         /**
298          * 上传组件实例
299          * @type Uploader
300          * @default ""
301          */
302         uploader:{ value:EMPTY },
303         /**
304          * 上传验证规则，每个规则都是一个数组，数组第一个值为规则，第二个值为错误消息
305          * @type Object
306          * @default  { allowExts:[ {desc:"JPG,JPEG,PNG,GIF,BMP", ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"}, '不支持{ext}格式的文件上传！' ], require:[false, '必须至少上传一个文件！'], max:[3, '每次最多上传{max}个文件！'], maxSize:[1000, '文件大小为{size}，文件太大！'], allowRepeat:[false, '该文件已经存在！'] } }
307          *
308          */
309         rules:{
310             value : {
311                 /**
312                  * 允许上传的文件格式，如果是使用flash上传方式，在选择文件时就可以过滤格式
313                  */
314                 allowExts:[
315                     {desc:"JPG,JPEG,PNG,GIF,BMP", ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"},
316                     '不支持{ext}格式的文件上传！'
317                 ],
318                 /**
319                  * 是否必须上传个文件
320                  */
321                 require:[false, '必须至少上传一个文件！'],
322                 /**
323                  * 允许的最大上传数
324                  */
325                 max:[3, '每次最多上传{max}个文件！'],
326                 /**
327                  * 文件最大大小，单位为kb
328                  */
329                 maxSize:[1000, '文件大小为{size}，文件太大！'],
330                 /**
331                  * 允许重复上传相同文件
332                  */
333                 allowRepeat:[false, '该文件已经存在！']
334             }
335         }
336     }});
337     return Auth;
338 }, {requires:['node','base']});
