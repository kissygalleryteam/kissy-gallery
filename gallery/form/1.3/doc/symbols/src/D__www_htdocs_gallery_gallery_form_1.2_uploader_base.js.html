140     S.extend(Uploader, Base, /** @lends Uploader.prototype*/{
141         /**
142          * 运行组件，实例化类后必须调用render()才真正运行组件逻辑
143          * @return {Uploader}
144          */
145         render:function () {
146             var self = this, serverConfig = self.get('serverConfig'),
147                 type = self.get('type'),
148                 UploadType = self.getUploadType(type), uploadType,
149                 uploaderTypeEvent = UploadType.event,
150                 button;
151             if (!UploadType) return false;
152             button = self._renderButton();
153             //路径input实例
154             self.set('urlsInput', self._renderUrlsInput());
155             self._renderQueue();
156             //如果是flash异步上传方案，增加swfUploader的实例作为参数
157             if (self.get('type') == Uploader.type.FLASH) {
158                 S.mix(serverConfig, {swfUploader:button.get('swfUploader')});
159             }
160             //实例化上传方式类
161             uploadType = new UploadType(serverConfig);
162             //监听上传器上传完成事件
163             uploadType.on(uploaderTypeEvent.SUCCESS, self._uploadCompleteHanlder, self);
164             //监听上传器上传进度事件
165             if (uploaderTypeEvent.PROGRESS) uploadType.on(uploaderTypeEvent.PROGRESS, self._uploadProgressHandler, self);
166             //监听上传器上传停止事件
167             uploadType.on(uploaderTypeEvent.STOP, self._uploadStopHanlder, self);
168             self.set('uploadType', uploadType);
169             self.fire(Uploader.event.RENDER);
170             return self;
171         },
172         /**
173          * 上传指定队列索引的文件
174          * @param {Number} index 文件对应的在上传队列数组内的索引值
175          * @example
176          * //上传队列中的第一个文件，uploader为Uploader的实例
177          * uploader.upload(0)
178          */
179         upload:function (index) {
180             if (!S.isNumber(index)) return false;
181             var self = this, uploadType = self.get('uploadType'),
182                 type=self.get('type'),
183                 queue = self.get('queue'),
184                 file = queue.get('files')[index],
185                 uploadParam;
186             if (!S.isPlainObject(file)) {
187                 S.log(LOG_PREFIX + '队列中不存在id为' + index + '的文件');
188                 return false;
189             }
190             //如果有文件正在上传，予以阻止上传
191             if (self.get('curUploadIndex') != EMPTY) {
192                 alert('第' + self.get('curUploadIndex') + '文件正在上传，请上传完后再操作！');
193                 return false;
194             }
195             //文件上传域，如果是flash上传,input为文件数据对象
196             uploadParam = file.input.id || file.input;
197             //如果是ajax上传直接传文件数据
198             if(type == 'ajax') uploadParam = file.data;
199             if(file['status'] === 'error'){
200                 return false;
201             }
202             //触发文件上传前事件
203             self.fire(Uploader.event.START, {index:index, file:file});
204             //阻止文件上传
205             if (!self.get('isAllowUpload')) return false;
206             //设置当前上传的文件id
207             self.set('curUploadIndex', index);
208             //改变文件上传状态为start
209             queue.fileStatus(index, Uploader.status.START);
210             //开始上传
211             uploadType.upload(uploadParam);
212         },
213         /**
214          * 取消文件上传，当index参数不存在时取消当前正在上传的文件的上传。cancel并不会停止其他文件的上传（对应方法是stop）
215          * @param {Number} index 队列数组索引
216          * @return {Uploader}
217          */
218         cancel:function (index) {
219             var self = this, uploadType = self.get('uploadType'),
220                 queue = self.get('queue'),
221                 statuses = Uploader.status,
222                 status = queue.fileStatus(index);
223             if(S.isNumber(index) && status != statuses.SUCCESS){
224                 uploadType.stop();
225                 queue.fileStatus(index,statuses.CANCEL);
226             }else{
227                 //取消上传后刷新状态，更改路径等操作请看_uploadStopHanlder()
228                 uploadType.stop();
229                 //存在批量上传操作，继续上传其他文件
230                 self._continueUpload();
231             }
232             return self;
233         },
234         /**
235          * 停止上传动作
236          * @return {Uploader}
237          */
238         stop : function(){
239             var self = this;
240             self.set('uploadFilesStatus',EMPTY);
241             self.cancel();
242             return self;
243         },
244         /**
245          * 批量上传队列中的指定状态下的文件
246          * @param {String} status 文件上传状态名
247          * @return {Uploader}
248          * @example
249          * //上传队列中所有等待的文件
250          * uploader.uploadFiles("waiting")
251          */
252         uploadFiles:function (status) {
253             var self = this;
254             if (!S.isString(status)) status = Uploader.status.WAITING;
255             self.set('uploadFilesStatus', status);
256             self._uploaderStatusFile(status);
257             return self;
258         },
259         /**
260          * 上传队列中的指定状态下的文件
261          * @param {String} status 文件上传状态名
262          * @return {Uploader}
263          */
264         _uploaderStatusFile:function (status) {
265             var self = this, queue = self.get('queue'),
266                 fileIndexs = queue.getIndexs(status);
267             //没有存在需要上传的文件，退出上传
268             if (!fileIndexs.length) {
269                 self.set('uploadFilesStatus', EMPTY);
270                 self.fire(Uploader.event.UPLOAD_FILES);
271                 return false;
272             }
273             //开始上传等待中的文件
274             self.upload(fileIndexs[0]);
275             return self;
276         },
277         /**
278          * 是否支持ajax方案上传
279          * @return {Boolean}
280          */
281         isSupportAjax:function () {
282             var isSupport = false;
283             try{
284                 if(FormData) isSupport = true;
285             }catch(e){
286                 isSupport = false;
287             }
288             return isSupport;

290         /**
291          * 是否支持flash方案上传
292          * @return {Boolean}
293          */
294         isSupportFlash:function () {
295             var fpv = S.UA.fpv();
296             return S.isArray(fpv) && fpv.length > 0;
297         },
298         /**
299          * 获取上传方式类（共有iframe、ajax、flash三种方式）
300          * @type {String} type 上传方式
301          * @return {IframeType|AjaxType|FlashType}
302          */
303         getUploadType:function (type) {
304             var self = this, types = Uploader.type,
305                 UploadType;
306             //如果type参数为auto，那么type=['ajax','flash','iframe']
307             if (type == types.AUTO) type = [types.AJAX, types.FLASH,types.IFRAME];
308             //如果是数组，遍历获取浏览器支持的上传方式
309             if (S.isArray(type) && type.length > 0) {
310                 S.each(type, function (t) {
311                     UploadType = self._getType(t);
312                     if (UploadType) return false;
313                 });
314             } else {
315                 UploadType = self._getType(type);
316             }
317             return UploadType;
318         },
319         /**
320          * 获取上传方式
321          * @param {String} type 上传方式（根据type返回对应的上传类，比如iframe返回IframeType）
322          */
323         _getType:function (type) {
324             var self = this, types = Uploader.type, UploadType,
325                 isSupportAjax = self.isSupportAjax(),
326                 isSupportFlash = self.isSupportFlash();
327             switch (type) {
328                 case types.IFRAME :
329                     UploadType = IframeType;
330                     break;
331                 case types.AJAX :
332                     UploadType = isSupportAjax && AjaxType || false;
333                     break;
334                 case types.FLASH :
335                     UploadType = isSupportFlash && FlashType || false;
336                     break;
337                 default :
338                     S.log(LOG_PREFIX + 'type参数不合法');
339                     return false;
340             }
341             if(UploadType) S.log(LOG_PREFIX + '使用' + type+'上传方式');
342             self.set('type', type);
343             return UploadType;
344         },
345         /**
346          * 运行Button上传按钮组件
347          * @return {Button}
348          */
349         _renderButton:function () {
350             var self = this, button,Button,
351                 type = self.get('type'),
352                 buttonTarget = self.get('target'),
353                 multiple = self.get('multiple'),
354                 disabled = self.get('disabled'),
355                 name = self.get('name'),
356                 config = {name:name,multiple:multiple,disabled:disabled};
357             if(type == Uploader.type.FLASH){
358                 Button = SwfButton;
359                 S.mix(config,{size:self.get('swfSize')});
360             }else{
361                 Button = HtmlButton;
362             }
363             button = new Button(buttonTarget,config);
364             //监听按钮改变事件
365             button.on('change', self._select, self);
366             //运行按钮实例
367             button.render();
368             self.set('button',button);
369             return button;
370         },
371         /**
372          * 运行Queue队列组件
373          * @return {Queue} 队列实例
374          */
375         _renderQueue:function () {
376             var self = this, queue = new Queue(),
377                 urlsInput = self.get('urlsInput');
378             if (!S.isObject(queue)) {
379                 S.log(LOG_PREFIX + 'queue参数不合法');
380                 return false;
381             }
382             //将上传组件实例传给队列，方便队列内部执行取消、重新上传的操作
383             queue.set('uploader', self);
384             //监听队列的删除事件
385             queue.on('remove', function (ev) {
386                 //删除该文件路径，sUrl为服务器端返回的文件路径，而url是客服端文件路径
387                 if(ev.file.sUrl && urlsInput) urlsInput.remove(ev.file.sUrl);
388             });
389             self.set('queue',queue);
390             return queue;
391         },
392         /**
393          * 选择完文件后
394          * @param {Object} ev 事件对象
395          */
396         _select:function (ev) {
397             var self = this, autoUpload = self.get('autoUpload'),
398                 queue = self.get('queue'),
399                 curId = self.get('curUploadIndex'),
400                 files = ev.files;
401             S.each(files, function (file) {
402                 //文件大小，IE浏览器下不存在
403                 if (!file.size) file.size = 0;
404                 //chrome文件名属性名为fileName，而firefox为name
405                 if (!file.name) file.name = file.fileName || EMPTY;
406                 //如果是flash上传，并不存在文件上传域input
407                 file.input = ev.input || file;
408             });
409             files = self._processExceedMultiple(files);
410             self.fire(Uploader.event.SELECT, {files : files});
411             //阻止文件上传
412             if (!self.get('isAllowUpload')) return false;
413             queue.add(files, function () {
414                 //如果不存在正在上传的文件，且允许自动上传，上传该文件
415                 if (curId == EMPTY && autoUpload) {
416                     self.uploadFiles();
417                 }
418             });
419         },
420         /**
421          * 超过最大多选数予以截断
422          */
423         _processExceedMultiple:function(files){
424             var self = this,multipleLen = self.get('multipleLen');
425             if(multipleLen < 0 || !S.isArray(files) || !files.length) return files;
426             return S.filter(files,function(file,index){
427                  return index < multipleLen;
428             });
429         },
430         /**
431          * 向上传按钮容器内增加用于存储文件路径的input
432          */
433         _renderUrlsInput:function () {
434             var self = this, button = self.get('button'), inputWrapper = button.get('target'),
435                 name = self.get('urlsInputName'),
436                 urlsInput = new UrlsInput(inputWrapper, {name:name});
437             urlsInput.render();
438             return urlsInput;
439         },
440         /**
441          * 当上传完毕后返回结果集的处理
442          */
443         _uploadCompleteHanlder:function (ev) {
444             var self = this, result = ev.result, status, event = Uploader.event,
445                 queue = self.get('queue'), index = self.get('curUploadIndex');
446             if (!S.isObject(result)) return false;
447             //将服务器端的数据保存到队列中的数据集合
448             queue.updateFile(index,{result:result});
449             //文件上传状态
450             status = Number(result.status);
451             // 只有上传状态为1时才是成功的
452             if (status === 1) {
453                 //修改队列中文件的状态为success（上传完成）
454                 queue.fileStatus(index, Uploader.status.SUCCESS);
455                 self._success(result.data);
456                 self.fire(event.SUCCESS,{index : index,file : queue.getFile(index),result:result});
457             } else {
458                 var msg = result.msg || result.message  || EMPTY;
459                 //修改队列中文件的状态为error（上传失败）
460                 queue.fileStatus(index, Uploader.status.ERROR, {msg:msg,result:result});
461                 self.fire(event.ERROR, {status:status,result:result,index:index,file:queue.getFile(index)});
462             }
463             //置空当前上传的文件在队列中的索引值
464             self.set('curUploadIndex', EMPTY);
465             self.fire(event.COMPLETE,{index : index,file : queue.getFile(index),result:result});
466             //存在批量上传操作，继续上传
467             self._continueUpload();
468         },
469         /**
470          * 取消上传后调用的方法
471          */
472         _uploadStopHanlder:function () {
473             var self = this, queue = self.get('queue'),
474                 index = self.get('curUploadIndex');
475             //更改取消上传后的状态
476             queue.fileStatus(index, Uploader.status.CANCEL);
477             //重置当前上传文件id
478             self.set('curUploadIndex', EMPTY);
479             self.fire(Uploader.event.CANCEL,{index : index});
480         },
481         /**
482          * 如果存在批量上传，则继续上传
483          */
484         _continueUpload : function(){
485             var self = this,
486                 uploadFilesStatus = self.get('uploadFilesStatus');
487             if (uploadFilesStatus != EMPTY) {
488                 self._uploaderStatusFile(uploadFilesStatus);
489             }
490         },
491         /**
492          * 上传进度监听器
493          */
494         _uploadProgressHandler:function (ev) {
495             var self = this, queue = self.get('queue'),
496                 index = self.get('curUploadIndex'),
497                 file = queue.getFile(index);
498             S.mix(ev,{file : file});
499             queue.fileStatus(index, Uploader.status.PROGRESS, ev);
500             self.fire(Uploader.event.PROGRESS,ev);
501         },
502         /**
503          * 上传成功后执行的回调函数
504          * @param {Object} data 服务器端返回的数据
505          */
506         _success:function (data) {
507             if (!S.isObject(data)) return false;
508             var self = this, url = data.url,
509                 urlsInput = self.get('urlsInput'),
510                 fileIndex = self.get('curUploadIndex'),
511                 queue = self.get('queue');
512             if (!S.isString(url) || !S.isObject(urlsInput)) return false;
513             //追加服务器端返回的文件url
514             queue.updateFile(fileIndex, {'sUrl':url});
515             //向路径隐藏域添加路径
516             urlsInput.add(url);
517         },
518         /**
519          * 添加默认数据到队列，不带参数的情况下，抓取restoreHook容器内的数据，添加到队列内
520          * @param {Array} data 文件数据
521          */
522         restore: function(data){
523         	var self = this,
524                 queue = self.get('queue'),
525                 urlsInput = self.get('urlsInput');
526             if(!data) data = self._getRestoreData();
527             if(!data.length) return false;
528             S.each(data,function(file){
529                 //服务器端路径赋值
530                 if(!file.sUrl && file.result) file.sUrl = file.result.data.url;
531                 //向队列添加文件
532                 var fileData = queue.add(file),
533                     id = fileData.id,index = queue.getFileIndex(id);
534                 urlsInput.add(file.sUrl);
535                 //改变文件状态为成功
536                 queue.fileStatus(index,'success',{index:index,id:id,file:fileData});
537             });
538         },
539         /**
540          * 抓取restoreHook容器内的数据
541          * @return {Array}
542          */
543         _getRestoreData:function(){
544             var self = this,
545                 restoreHook = self.get('restoreHook'),
546                 $restore = $(restoreHook);
547             if(!$restore.length) return [];
548             return S.JSON.parse(S.trim($restore.html()));
549         }
550     }, {ATTRS:/** @lends Uploader.prototype*/{
551         /**
552          * Button按钮的实例
553          * @type Button
554          * @default {}
555          */
556         button:{value:{}},
557         /**
558          * Queue队列的实例
559          * @type Queue
560          * @default {}
561          */
562         queue:{value:{}},
563         /**
564          * 采用的上传方案，当值是数组时，比如“type” : ["flash","ajax","iframe"]，按顺序获取浏览器支持的方式，该配置会优先使用flash上传方式，如果浏览器不支持flash，会降级为ajax，如果还不支持ajax，会降级为iframe；当值是字符串时，比如“type” : “ajax”，表示只使用ajax上传方式。这种方式比较极端，在不支持ajax上传方式的浏览器会不可用；当“type” : “auto”，auto是一种特例，等价于["ajax","flash","iframe"]。
