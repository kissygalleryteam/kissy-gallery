1 /**
  2  * @fileoverview flash上传按钮
  3  * @author: 紫英(橘子)<daxingplay@gmail.com>, 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/button/swfButton', function (S, Node, Base, SwfUploader) {
  6     var EMPTY = '', $ = Node.all,
  7         SWF_WRAPPER_ID_PREVFIX = 'swf-uploader-wrapper-';
  8 
  9     /**
 10      * @name SwfButton
 11      * @class flash上传按钮，基于龙藏的AJBrige。只有使用flash上传方式时候才会实例化这个类
 12      * @constructor
 13      * @extends Base
 14      */
 15     function SwfButton(target, config) {
 16         var self = this;
 17         config = S.merge({target:$(target)}, config);
 18         //调用父类构造函数
 19         SwfButton.superclass.constructor.call(self, config);
 20     }
 21 
 22     S.mix(SwfButton, /** @lends SwfButton*/{
 23         /**
 24          * 支持的事件
 25          */
 26         event:{
 27             //组件运行后事件
 28             RENDER : 'render',
 29             //选择文件后事件
 30             CHANGE:'change',
 31             //鼠标在swf中滑过事件
 32             MOUSE_OVER:'mouseOver',
 33             //鼠标在swf中按下事件
 34             MOUSE_DOWN:'mouseDown',
 35             //鼠标在swf中弹起事件
 36             MOUSE_UP:'mouseUp',
 37             //鼠标在swf中移开事件
 38             MOUSE_OUT:'mouseOut',
 39             //鼠标单击事件
 40             CLICK:'click'
 41         }
 42     });
 43     S.extend(SwfButton, Base, /** @lends SwfButton.prototype*/{
 44         /**
 45          *  运行，会实例化AJBrige的Uploader，存储为swfUploader属性
 46          */
 47         render:function () {
 48             var self = this,
 49                 $target = self.get('target'),
 50                 swfUploader,
 51                 multiple = self.get('multiple'),
 52                 fileFilters = self.get('fileFilters') ;
 53             $target.css('position', 'relative');
 54             self.set('swfWrapper',self._createSwfWrapper());
 55             self._setFlashSizeConfig();
 56             swfUploader = self._initSwfUploader();
 57             //SWF 内容准备就绪
 58             swfUploader.on('contentReady', function(ev){
 59                 //多选和文件过滤控制
 60                 swfUploader.browse(multiple, fileFilters);
 61                 //监听鼠标事件
 62                 self._bindBtnEvent();
 63                 //监听选择文件后事件
 64                 swfUploader.on('fileSelect', self._changeHandler, self);
 65                 self._setDisabled(self.get('disabled'));
 66                 self.fire(SwfButton.event.RENDER);
 67             }, self);
 68         },
 69         /**
 70          * 创建flash容器
 71          */
 72         _createSwfWrapper:function () {
 73             var self = this,
 74                 target = self.get('target'),
 75                 tpl = self.get('tpl'),
 76                 //容器id
 77                 id = self.get('swfWrapperId') != EMPTY && self.get('swfWrapperId') || SWF_WRAPPER_ID_PREVFIX + S.guid(),
 78                 //容器html
 79                 html = S.substitute(tpl, {id:id});
 80             self.set('swfWrapperId', id);
 81             return $(html).appendTo(target);
 82         },
 83         /**
 84          * 初始化ajbridge的uploader
 85          * @return {SwfUploader}
 86          */
 87         _initSwfUploader:function () {
 88             var self = this, flash = self.get('flash'),
 89                 id = self.get('swfWrapperId'),
 90                 swfUploader;
 91             S.mix(flash,{id:'swfUploader'+S.guid()});
 92             try {
 93                 //实例化AJBridge.Uploader
 94                 swfUploader = new SwfUploader(id, flash);
 95                 self.set('swfUploader', swfUploader);
 96             } catch (err) {
 97 
 98             }
 99             return swfUploader;
100         },
101         /**
102          * 监听swf的各个鼠标事件
103          * @return {SwfButton}
104          */
105         _bindBtnEvent:function () {
106             var self = this, event = SwfButton.event,
107                 swfUploader = self.get('swfUploader');
108             if (!swfUploader) return false;
109             S.each(event, function (ev) {
110                 swfUploader.on(ev, function (e) {
111                     self.fire(ev);
112                 }, self);
113             });
114             return self;
115         },
116         /**
117          * 设置flash配置参数
118          */
119         _setFlashSizeConfig:function () {
120             var self = this, flash = self.get('flash'),
121                 target = self.get('target'),
122                 size = self.get('size');
123             if(!S.isEmptyObject(size)){
124                 S.mix(flash.attrs, size);
125             }else{
126                 S.mix(flash.attrs, {
127                     width:target.innerWidth(),
128                     height:target.innerHeight()
129                 });
130             }
131             self.set('flash', flash);
132         },
133         /**
134          * flash中选择完文件后触发的事件
135          */
136         _changeHandler:function (ev) {
137             var self = this, files = ev.fileList;
138             self.fire(SwfButton.event.CHANGE, {files:files});
139         },
140         /**
141          * 设置上传组件的禁用
142          * @param {Boolean} disabled 是否禁用
143          * @return {Boolean}
144          */
145         _setDisabled : function(disabled){
146             var self = this,
147                 swfUploader = self.get('swfUploader'),
148                 cls = self.get('cls'),disabledCls = cls.disabled,
149                 $target = self.get('target'),
150                 $swfWrapper = self.get('swfWrapper');
151             if(!swfUploader || !S.isBoolean(disabled)) return false;
152             if(!disabled){
153                 $target.removeClass(disabledCls);
154                 //显示swf容器
155                 $swfWrapper.css('top',0);
156                 //TODO:之所以不使用更简单的unlock()方法，因为这个方法应用无效，有可能是bug
157                 //swfUploader.unlock();
158             }else{
159                 $target.addClass(disabledCls);
160                 //隐藏swf容器
161                 $swfWrapper.css('top','-3000px');
162                 //swfUploader.lock();
163             }
164             return disabled;
165         },
166         /**
167          * 显示按钮
168          */
169         show:function(){
170              var self = this,
171                  $target = self.get('target');
172              $target.show();
173         },
174         /**
175          * 隐藏按钮
176          */
177         hide:function(){
178             var self = this,
179                 $target = self.get('target');
180             $target.hide();
181         }
182     }, {ATTRS:/** @lends SwfButton.prototype*/{
183         /**
184          * 按钮目标元素
185          * @type KISSY.Node
186          * @default ""
187          */
188         target:{value:EMPTY},
189         /**
190          * swf容器
191          * @type KISSY.Node
192          * @default ""
193          */
194         swfWrapper : {value : EMPTY},
195         /**
196          * swf容器的id，如果不指定将使用随机id
197          * @type Number
198          * @default ""
199          */
200         swfWrapperId:{value:EMPTY},
201         /**
202          * flash容器模板
203          * @type String
204          */
205         tpl:{
206             value:'<div id="{id}" class="uploader-button-swf" style="position: absolute;top:0;left:0;z-index:2000;"></div>'
207         },
208         /**
209          * 是否开启多选支持
210          * @type Boolean
211          * @default true
212          */
213         multiple:{
214             value:true,
215             setter:function (v) {
216                 var self = this, swfUploader = self.get('swfUploader');
217                 if (swfUploader) {
218                     swfUploader.multifile(v);
219                 }
220                 return v;
221             }
222         },
223         /**
224          * 文件过滤，格式类似[{desc:"JPG,JPEG,PNG,GIF,BMP",ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"}]
225          * @type Array
226          * @default []
227          */
228         fileFilters:{
229             value:[],
230             setter:function (v) {
231                 var self = this, swfUploader = self.get('swfUploader');
232                 if(S.isObject(v)) v = [v];
233                 if (swfUploader && S.isArray(v)) {
234                     S.later(function(){
235                         swfUploader.filter(v);
236                     },500);
237                 }
238                 return v;
239             }
240         },
241         /**
242          * 禁用按钮
243          * @type Boolean
244          * @default false
245          */
246         disabled : {
247             value : false,
248             setter : function(v){
249                 var self = this, swfUploader = self.get('swfUploader');
250                 if (swfUploader) {
251                     self._setDisabled(v);
252                 }
253                 return v;
254             }
255         },
256         /**
257          * 样式
258          * @type Object
259          * @default  { disabled:'uploader-button-disabled' }
260          */
261         cls : {
262             value : { disabled:'uploader-button-disabled' }
263         },
264         /**
265          * 强制设置flash的尺寸，比如{width:100,height:100}，默认为自适应按钮容器尺寸
266          * @type Object
267          * @default {}
268          */
269         size : {value:{} },
270         /**
271          * flash配置，对于swf文件配路径配置非常关键，使用默认cdn上的路径就好
272          * @type Object
273          * @default { src:'http://a.tbcdn.cn/s/kissy/gallery/form/1.3/uploader/plugins/ajbridge/uploader.swf', id:'swfUploader', params:{ bgcolor:"#fff", wmode:"transparent" }, attrs:{ }, hand:true, btn:true }
274              }
275          */
276         flash:{
277             value:{
278                 src:'http://a.tbcdn.cn/s/kissy/gallery/form/1.3/uploader/plugins/ajbridge/uploader.swf',
279                 id:'swfUploader',
280                 params:{
281                     bgcolor:"#fff",
282                     wmode:"transparent"
283                 },
284                 //属性
285                 attrs:{ },
286                 //手型
287                 hand:true,
288                 //启用按钮模式,激发鼠标事件
289                 btn:true
290             }
291         },
292         /**
293          *  ajbridge的uploader的实例
294          *  @type SwfUploader
295          *  @default ""
296          */
297         swfUploader:{value:EMPTY}
298     }});
299     return SwfButton;
300 }, {requires:['node', 'base', '../plugins/ajbridge/uploader']});
