1 /**
  2  * @fileoverview 运行文件上传组件
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/index',function (S, Base, Node, Uploader,Auth) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploaderRender]:',
  7         dataName = {
  8             CONFIG:'data-config',
  9             BUTTON_CONFIG : 'data-button-config',
 10             THEME_CONFIG : 'data-theme-config',
 11             AUTH : 'data-auth'
 12         },
 13         //所支持的内置主题
 14         THEMES = ['default','imageUploader', 'ershouUploader','uploadify'],
 15         //内置主题路径前缀
 16         THEME_PREFIX='gallery/form/1.3/uploader/themes/';
 17     S.namespace('form');
 18     /**
 19      * 解析组件在页面中data-config成为组件的配置
 20      * @param {String} hook 组件钩子
 21      * @param {String} dataConfigName 配置名
 22      * @return {Object}
 23      */
 24     S.form.parseConfig = function(hook, dataConfigName) {
 25         var config = {}, sConfig, DATA_CONFIG = dataConfigName || dataName.CONFIG;
 26         sConfig = $(hook).attr(DATA_CONFIG);
 27         if (!S.isString(sConfig)) return {};
 28         try {
 29             config = S.JSON.parse(sConfig);
 30         } catch (err) {
 31             S.log(LOG_PREFIX + '请检查' + hook + '上' + DATA_CONFIG + '属性内的json格式是否符合规范！');
 32         }
 33         return config;
 34     };
 35 
 36     /**
 37      * @name RenderUploader
 38      * @class 异步文件上传入口文件，会从按钮的data-config='{}' 伪属性中抓取组件配置
 39      * @version 1.2
 40      * @constructor
 41      * @param {String | HTMLElement} buttonTarget *，上传按钮目标元素
 42      * @param {String | HTMLElement} queueTarget 文件队列目标元素，再不需要显示文件信息的情况下这个参数可以设置为null
 43      * @param {Object} config 配置，该配置会覆盖data-config伪属性中的数据
 44      * @requires Uploader
 45      * @requires Auth
 46      * @example
 47      * <a id="J_UploaderBtn" class="uploader-button" data-config=
 48      '{"type" : "auto",
 49      "serverConfig":{"action":"upload.php"},
 50      "name":"Filedata",
 51      "urlsInputName":"fileUrls"}'
 52      href="#">
 53      选择要上传的文件
 54      </a>
 55      <ul id="J_UploaderQueue">
 56 
 57      </ul>
 58      * @example
 59      *
 60 KISSY.use('gallery/form/1.3/uploader/index', function (S, RenderUploader) {
 61      var ru = new RenderUploader('#J_UploaderBtn', '#J_UploaderQueue');
 62      ru.on("init", function (ev) {
 63         var uploader = ev.uploader;
 64      })
 65 })
 66      */
 67     function RenderUploader(buttonTarget, queueTarget, config) {
 68         var self = this;
 69         //合并配置
 70         config = S.mix(S.form.parseConfig(buttonTarget), config);
 71         //超类初始化
 72         RenderUploader.superclass.constructor.call(self, config);
 73         self.set('buttonTarget', buttonTarget);
 74         self.set('queueTarget', queueTarget);
 75         self.set('uploaderConfig', config);
 76         self._init();
 77     }
 78     /**
 79      * @name RenderUploader#init
 80      * @desc 上传组件完全初始化成功后触发，对uploader的操作务必先监听init事件
 81      * @event
 82      * @param {Uploader} ev.uploader   Uploader的实例
 83      * @param {Button} ev.button   Button的实例
 84      * @param {Queue} ev.queue   Queue的实例
 85      * @param {Auth} ev.auth   Auth的实例
 86      */
 87 
 88     S.extend(RenderUploader, Base, /** @lends RenderUploader.prototype*/{
 89         /**
 90          * 初始化组件
 91          */
 92         _init:function () {
 93             var self = this,
 94                 //主题路径
 95                 theme = self.get('theme'),
 96                 uploader;
 97             //不使用主题
 98             if(theme == EMPTY){
 99                 uploader = self._initUploader();
100                 self.set('button', uploader.get('button'));
101                 S.later(function(){
102                     self.fire('init', {uploader:uploader,button:uploader.get('button'),queue:uploader.get('queue'),auth:uploader.get('auth')});
103                 },500);
104             }else{
105                 self._initThemes(function (theme) {
106                     uploader = self._initUploader();
107                     self.set('button', uploader.get('button'));
108                     theme.set('uploader',uploader);
109                     theme.set('button',uploader.get('button'));
110                     theme.set('queue',uploader.get('queue'));
111                     theme.set('auth',uploader.get('auth'));
112                     theme._UploaderRender(function(){
113                         theme.afterUploaderRender(uploader);
114                         // 抓取restoreHook容器内的数据，生成文件DOM
115                         uploader.restore();
116                         self.fire('init', {uploader:uploader,button:uploader.get('button'),queue:uploader.get('queue'),auth:uploader.get('auth')});
117                     });
118                 });
119             }
120         },
121         /**
122          * 初始化Uploader
123          * @return {Uploader}
124          */
125         _initUploader:function(){
126             var self = this, uploaderConfig = self.get('uploaderConfig'),
127                 name = self.get('name');
128             S.mix(uploaderConfig.serverConfig,{'fileDataName':name});
129             //配置增加按钮实例和队列实例
130             S.mix(uploaderConfig, {target:self.get('buttonTarget')});
131             var uploader = new Uploader(uploaderConfig);
132             uploader.render();
133             self.set('uploader', uploader);
134             self._auth();
135             return uploader;
136         },
137         /**
138          * 初始化主题
139          * @param {Function} callback 主题加载完成后的执行的回调函数
140          */
141         _initThemes:function (callback) {
142             var self = this, theme = self.get('theme'),
143                 target = self.get('buttonTarget'),
144                 cf = self.get('themeConfig'),
145                 //从html标签的伪属性中抓取配置
146                 config = S.form.parseConfig(target,dataName.THEME_CONFIG);
147             S.mix(config,cf);
148             self.set('themeConfig',config);
149             //如果只是传递主题名，组件自行拼接
150             theme = self._getThemeName(theme);
151             S.use(theme, function (S, Theme) {
152                 var queueTarget = self.get('queueTarget'), theme;
153                 S.mix(config,{queueTarget:queueTarget,buttonTarget:self.get('buttonTarget')});
154                 theme = new Theme(config);
155                 theme.on('render',function(){
156                     callback && callback.call(self, theme);
157                 });
158                 theme.render();
159             })
160         },
161         /**
162          * 获取正确的主题名
163          * @param {String} theme 主题名
164          * @return {String}
165          */
166         _getThemeName:function(theme){
167             var themeName = theme;
168             S.each(THEMES,function(t){
169                if(t == theme){
170                    themeName = THEME_PREFIX + theme;
171                }
172             });
173             themeName = themeName + '/index';
174             return themeName;
175         },
176         /**
177          * 文件上传验证
178          */
179         _auth:function () {
180             var self = this,buttonTarget = self.get('buttonTarget'),
181                 uploader = self.get('uploader'),
182                 cf = self.get('authConfig'),
183                 config = S.form.parseConfig(buttonTarget,dataName.AUTH);
184             S.mix(config,cf);
185             self.set('authConfig',config);
186             if(S.isEmptyObject(config)) return false;
187             auth = new Auth(uploader,{rules : config});
188             uploader.set('auth',auth);
189             return auth;
190         }
191     }, {
192         ATTRS:/** @lends RenderUploader.prototype*/{
193             /**
194              * 主题引用路径，当值为""时，不使用uploader主题。非内置主题，值为模块路径，比如"refund/rfUploader"
195              * @type String
196              * @default  “default”
197              */
198             theme:{value:'default' },
199             /**
200              * 主题配置，会覆盖data-theme-config中的配置，不再推荐使用伪属性的方式配置主题参数
201              * @type Object
202              * @default {}
203              * @since 1.2
204              * @example
205  //配置主题样式路径
206 themeConfig:{
207     cssUrl:'gallery/form/1.3/uploader/themes/default/style.css'
208 }
209              */
210             themeConfig:{value:{}},
211             /**
212              * 按钮目标元素
213              * @type String|HTMLElement|KISSY.NodeList
214              * @default ""
215              */
216             buttonTarget:{value:EMPTY},
217             /**
218              * 队列目标元素
219              * @default ""
220              * @type String|HTMLElement|KISSY.NodeList
221              */
222             queueTarget:{value:EMPTY},
223             /**
224              * 上传组件配置
225              * @type Object
226              * @default {}
227              */
228             uploaderConfig:{},
229             /**
230              * 验证配置
231              * @type Object
232              * @default {}
233              * @since 1.2
234              * @example
235              //验证配置
236              authConfig: {
237                  allowExts:[
238                      {desc:"JPG,JPEG,PNG,GIF,BMP", ext:"*.jpg;*.jpeg;*.png;*.gif;*.bmp"},
239                      '不支持{ext}格式的文件上传！'
240                  ],
241                  max:[3, '每次最多上传{max}个文件！']
242              }
243              */
244             authConfig:{value:{}},
245             /**
246              * Button（上传按钮）的实例
247              * @type Button
248              * @default ""
249              */
250             button:{value:EMPTY},
251             /**
252              * Queue（上传队列）的实例
253              * @type Queue
254              * @default ""
255              */
256             queue:{value:EMPTY},
257             /**
258              * 上传组件实例
259              * @type Uploader
260              * @default ""
261              */
262             uploader:{value:EMPTY}
263         }
264     });
265     return RenderUploader;
266 }, {requires:['base', 'node', './base','./auth/base']});
