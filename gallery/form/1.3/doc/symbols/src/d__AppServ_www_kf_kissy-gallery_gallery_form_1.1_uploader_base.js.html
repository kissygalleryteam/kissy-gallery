139     S.extend(Uploader, Base, /** @lends Uploader.prototype*/{
140         /**
141          * 运行组件，实例化类后必须调用render()才真正运行组件逻辑
142          * @return {Uploader}
143          */
144         render:function () {
145             var self = this, serverConfig = self.get('serverConfig'),
146                 type = self.get('type'),
147                 UploadType = self.getUploadType(type), uploadType,
148                 uploaderTypeEvent = UploadType.event,
149                 button;
150             if (!UploadType) return false;
151             //路径input实例
152             self.set('urlsInput', self._renderUrlsInput());
153             self._renderQueue();
154             button = self._renderButton();
155             //如果是flash异步上传方案，增加swfUploader的实例作为参数
156             if (self.get('type') == Uploader.type.FLASH) {
157                 S.mix(serverConfig, {swfUploader:button.get('swfUploader')});
158             }
159             //实例化上传方式类
160             uploadType = new UploadType(serverConfig);
161             //监听上传器上传完成事件
162             uploadType.on(uploaderTypeEvent.SUCCESS, self._uploadCompleteHanlder, self);
163             //监听上传器上传进度事件
164             if (uploaderTypeEvent.PROGRESS) uploadType.on(uploaderTypeEvent.PROGRESS, self._uploadProgressHandler, self);
165             //监听上传器上传停止事件
166             uploadType.on(uploaderTypeEvent.STOP, self._uploadStopHanlder, self);
167             self.set('uploadType', uploadType);
168             self.fire(Uploader.event.RENDER);
169             return self;
170         },
171         /**
172          * 上传指定队列索引的文件
173          * @param {Number} index 文件对应的在上传队列数组内的索引值
174          * @example
175          * //上传队列中的第一个文件，uploader为Uploader的实例
176          * uploader.upload(0)
177          */
178         upload:function (index) {
179             if (!S.isNumber(index)) return false;
180             var self = this, uploadType = self.get('uploadType'),
181                 type=self.get('type'),
182                 queue = self.get('queue'),
183                 file = queue.get('files')[index],
184                 uploadParam;
185             if (!S.isPlainObject(file)) {
186                 S.log(LOG_PREFIX + '队列中不存在id为' + index + '的文件');
187                 return false;
188             }
189             //如果有文件正在上传，予以阻止上传
190             if (self.get('curUploadIndex') != EMPTY) {
191                 alert('第' + self.get('curUploadIndex') + '文件正在上传，请上传完后再操作！');
192                 return false;
193             }
194             //文件上传域，如果是flash上传,input为文件数据对象
195             uploadParam = file.input.id || file.input;
196             //如果是ajax上传直接传文件数据
197             if(type == 'ajax') uploadParam = file.data;
198             if(file['status'] === 'error'){
199                 return false;
200             }
201             //触发文件上传前事件
202             self.fire(Uploader.event.START, {index:index, file:file});
203             //阻止文件上传
204             if (!self.get('isAllowUpload')) return false;
205             //设置当前上传的文件id
206             self.set('curUploadIndex', index);
207             //改变文件上传状态为start
208             queue.fileStatus(index, Uploader.status.START);
209             //开始上传
210             uploadType.upload(uploadParam);
211         },
212         /**
213          * 取消文件上传，当index参数不存在时取消当前正在上传的文件的上传。cancel并不会停止其他文件的上传（对应方法是stop）
214          * @param {Number} index 队列数组索引
215          * @return {Uploader}
216          */
217         cancel:function (index) {
218             var self = this, uploadType = self.get('uploadType'),
219                 queue = self.get('queue'),
220                 statuses = Uploader.status,
221                 status = queue.fileStatus(index);
222             if(S.isNumber(index) && status != statuses.SUCCESS){
223                 queue.fileStatus(index,statuses.CANCEL);
224             }else{
225                 //取消上传后刷新状态，更改路径等操作请看_uploadStopHanlder()
226                 uploadType.stop();
227                 //存在批量上传操作，继续上传其他文件
228                 self._continueUpload();
229             }
230             return self;
231         },
232         /**
233          * 停止上传动作
234          * @return {Uploader}
235          */
236         stop : function(){
237             var self = this;
238             self.set('uploadFilesStatus',EMPTY);
239             self.cancel();
240             return self;
241         },
242         /**
243          * 批量上传队列中的指定状态下的文件
244          * @param {String} status 文件上传状态名
245          * @return {Uploader}
246          * @example
247          * //上传队列中所有等待的文件
248          * uploader.uploadFiles("waiting")
249          */
250         uploadFiles:function (status) {
251             var self = this;
252             if (!S.isString(status)) status = Uploader.status.WAITING;
253             self.set('uploadFilesStatus', status);
254             self._uploaderStatusFile(status);
255             return self;
256         },
257         /**
258          * 上传队列中的指定状态下的文件
259          * @param {String} status 文件上传状态名
260          * @return {Uploader}
261          */
262         _uploaderStatusFile:function (status) {
263             var self = this, queue = self.get('queue'),
264                 fileIndexs = queue.getIndexs(status);
265             //没有存在需要上传的文件，退出上传
266             if (!fileIndexs.length) {
267                 self.set('uploadFilesStatus', EMPTY);
268                 self.fire(Uploader.event.UPLOAD_FILES);
269                 return false;
270             }
271             //开始上传等待中的文件
272             self.upload(fileIndexs[0]);
273             return self;
274         },
275         /**
276          * 是否支持ajax方案上传
277          * @return {Boolean}
278          */
279         isSupportAjax:function () {
280             var isSupport = false;
281             try{
282                 if(FormData) isSupport = true;
283             }catch(e){
284                 isSupport = false;
285             }
286             return isSupport;
287         },
288         /**
289          * 是否支持flash方案上传
290          * @return {Boolean}
291          */
292         isSupportFlash:function () {
293             var fpv = S.UA.fpv();
294             return S.isArray(fpv) && fpv.length > 0;
295         },
296         /**
297          * 获取上传方式类（共有iframe、ajax、flash三种方式）
298          * @type {String} type 上传方式
299          * @return {IframeType|AjaxType|FlashType}
300          */
301         getUploadType:function (type) {
302             var self = this, types = Uploader.type,
303                 UploadType;
304             //如果type参数为auto，那么type=['ajax','flash','iframe']
305             if (type == types.AUTO) type = [types.AJAX, types.IFRAME];
306             //如果是数组，遍历获取浏览器支持的上传方式
307             if (S.isArray(type) && type.length > 0) {
308                 S.each(type, function (t) {
309                     UploadType = self._getType(t);
310                     if (UploadType) return false;
311                 });
312             } else {
313                 UploadType = self._getType(type);
314             }
315             return UploadType;
316         },
317         /**
318          * 获取上传方式
319          * @param {String} type 上传方式（根据type返回对应的上传类，比如iframe返回IframeType）
320          */
321         _getType:function (type) {
322             var self = this, types = Uploader.type, UploadType,
323                 isSupportAjax = self.isSupportAjax(),
324                 isSupportFlash = self.isSupportFlash();
325             switch (type) {
326                 case types.IFRAME :
327                     UploadType = IframeType;
328                     break;
329                 case types.AJAX :
330                     UploadType = isSupportAjax && AjaxType || false;
331                     break;
332                 case types.FLASH :
333                     UploadType = isSupportFlash && FlashType || false;
334                     break;
335                 default :
336                     S.log(LOG_PREFIX + 'type参数不合法');
337                     return false;
338             }
339             if(UploadType) S.log(LOG_PREFIX + '使用' + type+'上传方式');
340             self.set('type', type);
341             return UploadType;
342         },
343         /**
344          * 运行Button上传按钮组件
345          * @return {Button}
346          */
347         _renderButton:function () {
348             var self = this, button = self.get('button');
349             if (!S.isObject(button)) {
350                 S.log(LOG_PREFIX + 'button参数不合法！');
351                 return false;
352             }
353             //监听按钮改变事件
354             button.on('change', self._select, self);
355             //运行按钮实例
356             button.render();
357             return button;
358         },
359         /**
360          * 运行Queue队列组件
361          * @return {Queue} 队列实例
362          */
363         _renderQueue:function () {
364             var self = this, queue = self.get('queue'),
365                 urlsInput = self.get('urlsInput');
366             if (!S.isObject(queue)) {
367                 S.log(LOG_PREFIX + 'queue参数不合法');
368                 return false;
369             }
370             //将上传组件实例传给队列，方便队列内部执行取消、重新上传的操作
371             queue.set('uploader', self);
372             //监听队列的删除事件
373             queue.on('remove', function (ev) {
374                 //删除该文件路径，sUrl为服务器端返回的文件路径，而url是客服端文件路径
375                 urlsInput.remove(ev.file.sUrl);
376             });
377             return queue;
378         },
379         /**
380          * 选择完文件后
381          * @param {Object} ev 事件对象
382          */
383         _select:function (ev) {
384             var self = this, autoUpload = self.get('autoUpload'),
385                 queue = self.get('queue'),
386                 curId = self.get('curUploadIndex'),
387                 files = ev.files;
388             S.each(files, function (file) {
389                 //文件大小，IE浏览器下不存在
390                 if (!file.size) file.size = 0;
391                 //chrome文件名属性名为fileName，而firefox为name
392                 if (!file.name) file.name = file.fileName || EMPTY;
393                 //如果是flash上传，并不存在文件上传域input
394                 file.input = ev.input || file;
395             });
396             self.fire(Uploader.event.SELECT, {files : files});
397             //阻止文件上传
398             if (!self.get('isAllowUpload')) return false;
399             queue.add(files, function () {
400                 //如果不存在正在上传的文件，且允许自动上传，上传该文件
401                 if (curId == EMPTY && autoUpload) {
402                     self.uploadFiles();
403                 }
404             });
405         },
406         /**
407          * 向上传按钮容器内增加用于存储文件路径的input
408          */
409         _renderUrlsInput:function () {
410             var self = this, button = self.get('button'), inputWrapper = button.get('target'),
411                 name = self.get('urlsInputName'),
412                 urlsInput = new UrlsInput(inputWrapper, {name:name});
413             urlsInput.render();
414             return urlsInput;
415         },
416         /**
417          * 当上传完毕后返回结果集的处理
418          */
419         _uploadCompleteHanlder:function (ev) {
420             var self = this, result = ev.result, status, event = Uploader.event,
421                 queue = self.get('queue'), index = self.get('curUploadIndex');
422             if (!S.isObject(result)) return false;
423             //将服务器端的数据保存到队列中的数据集合
424             queue.updateFile(index,{result:result});
425             //文件上传状态
426             status = Number(result.status);
427             // 只有上传状态为1时才是成功的
428             if (status === 1) {
429                 //修改队列中文件的状态为success（上传完成）
430                 queue.fileStatus(index, Uploader.status.SUCCESS);
431                 self._success(result.data);
432                 self.fire(event.SUCCESS,{index : index,file : queue.getFile(index),result:result});
433             } else {
434                 var msg = result.msg || result.message  || EMPTY;
435                 //修改队列中文件的状态为error（上传失败）
436                 queue.fileStatus(index, Uploader.status.ERROR, {msg:msg,result:result});
437                 self.fire(event.ERROR, {status:status,result:result});
438             }
439             //置空当前上传的文件在队列中的索引值
440             self.set('curUploadIndex', EMPTY);
441             self.fire(event.COMPLETE,{index : index,file : queue.getFile(index),result:result});
442             //存在批量上传操作，继续上传
443             self._continueUpload();
444         },
445         /**
446          * 取消上传后调用的方法
447          */
448         _uploadStopHanlder:function () {
449             var self = this, queue = self.get('queue'),
450                 index = self.get('curUploadIndex');
451             //更改取消上传后的状态
452             queue.fileStatus(index, Uploader.status.CANCEL);
453             //重置当前上传文件id
454             self.set('curUploadIndex', EMPTY);
455             self.fire(Uploader.event.CANCEL,{index : index});
456         },
457         /**
458          * 如果存在批量上传，则继续上传
459          */
460         _continueUpload : function(){
461             var self = this,
462                 uploadFilesStatus = self.get('uploadFilesStatus');
463             if (uploadFilesStatus != EMPTY) {
464                 self._uploaderStatusFile(uploadFilesStatus);
465             }
466         },
467         /**
468          * 上传进度监听器
469          */
470         _uploadProgressHandler:function (ev) {
471             var self = this, queue = self.get('queue'),
472                 index = self.get('curUploadIndex'),
473                 file = queue.getFile(index);
474             S.mix(ev,{file : file});
475             queue.fileStatus(index, Uploader.status.PROGRESS, ev);
476             self.fire(Uploader.event.PROGRESS,ev);
477         },
478         /**
479          * 上传成功后执行的回调函数
480          * @param {Object} data 服务器端返回的数据
481          */
482         _success:function (data) {
483             if (!S.isObject(data)) return false;
484             var self = this, url = data.url,
485                 urlsInput = self.get('urlsInput'),
486                 fileIndex = self.get('curUploadIndex'),
487                 queue = self.get('queue');
488             if (!S.isString(url) || !S.isObject(urlsInput)) return false;
489             //追加服务器端返回的文件url
490             queue.updateFile(fileIndex, {'sUrl':url});
491             //向路径隐藏域添加路径
492             urlsInput.add(url);
493         },
494         /**
495          * 添加默认数据到队列，不带参数的情况下，抓取restoreHook容器内的数据，添加到队列内
496          * @param {Array} data 文件数据
497          */
498         restore: function(data){
499         	var self = this,
500                 queue = self.get('queue'),
501                 urlsInput = self.get('urlsInput');
502             if(!data) data = self._getRestoreData();
503             if(!data.length) return false;
504             S.each(data,function(file){
505                 //服务器端路径赋值
506                 if(!file.sUrl && file.result) file.sUrl = file.result.data.url;
507                 //向队列添加文件
508                 var fileData = queue.add(file),
509                     id = fileData.id,index = queue.getFileIndex(id);
510                 urlsInput.add(file.sUrl);
511                 //改变文件状态为成功
512                 queue.fileStatus(index,'success',{index:index,id:id,file:fileData});
513             });
514         },
515         /**
516          * 抓取restoreHook容器内的数据
517          * @return {Array}
518          */
519         _getRestoreData:function(){
520             var self = this,
521                 restoreHook = self.get('restoreHook'),
522                 $restore = $(restoreHook);
523             if(!$restore.length) return [];
524             return S.JSON.parse($restore.html());
525         }

531          */
