1 /**
  2  * @fileoverview 运行文件上传组件
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/index',function (S, Base, Node, Uploader, Button,SwfButton,Auth,Queue) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploaderRender]:',
  7         dataName = {
  8             CONFIG:'data-config',
  9             BUTTON_CONFIG : 'data-button-config',
 10             THEME_CONFIG : 'data-theme-config',
 11             AUTH : 'data-auth'
 12         },
 13         //所支持的内置主题
 14         THEMES = ['default','imageUploader'],
 15         //内置主题路径前缀
 16         THEME_PREFIX='gallery/form/1.3/uploader/themes/';
 17     S.namespace('form');
 18     /**
 19      * 解析组件在页面中data-config成为组件的配置
 20      * @param {String} hook 组件钩子
 21      * @param {String} dataConfigName 配置名
 22      * @return {Object}
 23      */
 24     S.form.parseConfig = function(hook, dataConfigName) {
 25         var config = {}, sConfig, DATA_CONFIG = dataConfigName || dataName.CONFIG;
 26         sConfig = $(hook).attr(DATA_CONFIG);
 27         if (!S.isString(sConfig)) return {};
 28         try {
 29             config = S.JSON.parse(sConfig);
 30         } catch (err) {
 31             S.log(LOG_PREFIX + '请检查' + hook + '上' + DATA_CONFIG + '属性内的json格式是否符合规范！');
 32         }
 33         return config;
 34     };
 35 
 36     /**
 37      * @name RenderUploader
 38      * @class 异步文件上传入口文件，会从按钮的data-config='{}' 伪属性中抓取组件配置
 39      * @version 1.1.3
 40      * @constructor
 41      * @param {String | HTMLElement} buttonTarget *，上传按钮目标元素
 42      * @param {String | HTMLElement} queueTarget *，文件队列目标元素
 43      * @param {Object} config 配置，该配置好覆盖data-config伪属性中的数据
 44      * @requires Uploader
 45      * @requires Button
 46      * @requires SwfButton
 47      * @requires Auth
 48      * @requires Queue
 49      * @example
 50      * <a id="J_UploaderBtn" class="uploader-button" data-config=
 51      '{"type" : "auto",
 52      "serverConfig":{"action":"upload.php"},
 53      "name":"Filedata",
 54      "urlsInputName":"fileUrls"}'
 55      href="#">
 56      选择要上传的文件
 57      </a>
 58      <ul id="J_UploaderQueue">
 59 
 60      </ul>
 61      * @example
 62      *
 63 KISSY.use('gallery/form/1.3/uploader/index', function (S, RenderUploader) {
 64      var ru = new RenderUploader('#J_UploaderBtn', '#J_UploaderQueue');
 65      ru.on("init", function (ev) {
 66         var uploader = ev.uploader;
 67      })
 68 })
 69      */
 70     function RenderUploader(buttonTarget, queueTarget, config) {
 71         var self = this;
 72         //合并配置
 73         config = S.mix(S.form.parseConfig(buttonTarget), config);
 74         //超类初始化
 75         RenderUploader.superclass.constructor.call(self, config);
 76         self.set('buttonTarget', buttonTarget);
 77         self.set('queueTarget', queueTarget);
 78         self.set('uploaderConfig', config);
 79         self._init();
 80     }
 81     /**
 82      * @name RenderUploader#init
 83      * @desc 上传组件完全初始化成功后触发，对uploader的操作务必先监听init事件
 84      * @event
 85      * @param {Uploader} ev.uploader   Uploader的实例
 86      * @param {Uploader} ev.button   Button的实例
 87      * @param {Uploader} ev.queue   Queue的实例
 88      * @param {Uploader} ev.auth   Auth的实例
 89      */
 90 
 91     S.extend(RenderUploader, Base, /** @lends RenderUploader.prototype*/{
 92         /**
 93          * 初始化组件
 94          */
 95         _init:function () {
 96             var self = this,
 97                 //按钮
 98                 button = self._initButton(),
 99                 //队列
100                 queue = self._initQueue(),
101                 //上传组件
102                 uploader = self._initUploader(button,queue),
103                 //上传验证
104                 auth = self._auth(),
105                 classes = {uploader:uploader,button:button,queue:queue,auth:auth},
106                 //主题路径
107                 theme = self.get('theme');
108             self.set('button', button);
109             //不使用主题
110             if(theme == EMPTY){
111                 self.fire('init', classes);
112             }else{
113                 self._initThemes(function (theme) {
114                     theme.set('uploader',uploader);
115                     theme.set('button',button);
116                     theme.set('queue',queue);
117                     theme.set('auth',auth);
118                     theme._UploaderRender(function(){
119                         // 抓取restoreHook容器内的数据，生成文件DOM
120                         uploader.restore();
121                         theme.afterUploaderRender(uploader);
122                         self.fire('init', classes);
123                     });
124                 });
125             }
126         },
127         /**
128          * 初始化Uploader
129          * @param { Button} button Button的实例
130          * @param { Queue} queue Queue的实例
131          * @return {Uploader}
132          */
133         _initUploader:function(button,queue){
134             var self = this, uploaderConfig = self.get('uploaderConfig'),
135                 name = self.get('name');
136             S.mix(uploaderConfig.serverConfig,{'fileDataName':name});
137             //配置增加按钮实例和队列实例
138             S.mix(uploaderConfig, {button:button, queue:queue});
139             var uploader = new Uploader(uploaderConfig);
140             uploader.render();
141             self.set('uploader', uploader);
142             return uploader;
143         },
144         /**
145          * 初始化模拟的上传按钮
146          * @return {Button}
147          */
148         _initButton:function () {
149             var self = this,
150                 target = self.get('buttonTarget'),
151                 //从html标签的伪属性中抓取配置
152                 config = S.form.parseConfig(target,dataName.BUTTON_CONFIG),
153                 name = self.get('name'),
154                 type = self.get('type');
155             //合并配置
156             config = S.merge({name:name},config);
157             //实例化上传按钮
158             return type != 'flash' && new Button(target, config) || new SwfButton(target);
159         },
160         /**
161          * 初始化队列
162          * @return {Queue}
163          */
164         _initQueue:function(){
165             return new Queue();
166         },
167         /**
168          * 初始化主题
169          * @param {Function} callback 主题加载完成后的执行的回调函数
170          */
171         _initThemes:function (callback) {
172             var self = this, theme = self.get('theme'),
173                 target = self.get('buttonTarget'),
174                 //从html标签的伪属性中抓取配置
175                 config = S.form.parseConfig(target,dataName.THEME_CONFIG);
176             //如果只是传递主题名，组件自行拼接
177             theme = self._getThemeName(theme);
178             S.use(theme, function (S, Theme) {
179                 var queueTarget = self.get('queueTarget'), theme;
180                 S.mix(config,{queueTarget:queueTarget});
181                 theme = new Theme(config);
182                 callback && callback.call(self, theme);
183             })
184         },
185         /**
186          * 获取正确的主题名
187          * @param {String} theme 主题名
188          * @return {String}
189          */
190         _getThemeName:function(theme){
191             var themeName = theme;
192             S.each(THEMES,function(t){
193                if(t == theme){
194                    themeName = THEME_PREFIX + theme;
195                }
196             });
197             themeName = themeName + '/index';
198             return themeName;
199         },
200         /**
201          * 文件上传验证
202          */
203         _auth:function () {
204             var self = this,buttonTarget = self.get('buttonTarget'),
205                 uploader = self.get('uploader'),
206                 rules, auth = EMPTY;
207             //存在验证配置
208             if($(buttonTarget).attr(dataName.AUTH)){
209                 rules = S.form.parseConfig(buttonTarget,dataName.AUTH);
210                 auth = new Auth(uploader,{rules : rules});
211                 uploader.set('auth',auth);
212             }
213             return auth;
214         }
215     }, {
216         ATTRS:/** @lends RenderUploader.prototype*/{
217             /**
218              * 主题引用路径
219              * @type String
220              * @default  “gallery/form/1.3/uploader/themes/default”
221              */
222             theme:{value:'gallery/form/1.3/uploader/themes/default' },
223             /**
224              * 按钮目标元素
225              * @type String|HTMLElement|KISSY.NodeList
226              * @default ""
227              */
228             buttonTarget:{value:EMPTY},
229             /**
230              * 队列目标元素
231              * @default ""
232              * @type String|HTMLElement|KISSY.Node
233              */
234             queueTarget:{value:EMPTY},
235             /**
236              * 上传组件配置
237              * @type Object
238              * @default {}
239              */
240             uploaderConfig:{},
241             /**
242              * Button（上传按钮）的实例
243              * @type Button
244              * @default ""
245              */
246             button:{value:EMPTY},
247             /**
248              * Queue（上传队列）的实例
249              * @type Queue
250              * @default ""
251              */
252             queue:{value:EMPTY},
253             /**
254              * 上传组件实例
255              * @type Uploader
256              * @default ""
257              */
258             uploader:{value:EMPTY}
259         }
260     });
261     return RenderUploader;
262 }, {requires:['base', 'node', './base', './button/base','./button/swfButton','./auth/base','./queue']});
