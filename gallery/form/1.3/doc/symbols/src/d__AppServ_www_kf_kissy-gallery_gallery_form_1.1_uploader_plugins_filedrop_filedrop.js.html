1 /**
  2  * @fileoverview  文件拖拽上传插件
  3  *  @author 飞绿
  4  */
  5 KISSY.add('gallery/form/1.3/uploader/plugins/filedrop/filedrop', function (S, Node, Base) {
  6     var EMPTY = '',
  7         $ = Node.all,
  8         UA = S.UA;
  9     /**
 10      * @name FileDrop
 11      * @class 文件拖拽上传插件
 12      * @constructor
 13      *  @author 飞绿
 14      * @extends Base
 15      * @param {Object} config 组件配置（下面的参数为配置项，配置会写入属性，详细的配置说明请看属性部分）
 16      * @param {Button} config.button *，Button按钮的实例
 17      */
 18     var FileDrop = function (config) {
 19         var self = this;
 20         // console.log(config);
 21         FileDrop.superclass.constructor.call(self, config);
 22 //        console.log($(config.target), self.get('target'));
 23         self.set('mode', getMode());
 24     };
 25 
 26     var getMode = function () {
 27         if (UA.webkit >= 7 || UA.firefox >= 3.6) {
 28             return 'supportDrop';
 29         }
 30         if (UA.ie) {
 31             return 'notSupportDropIe';
 32         }
 33         if (UA.webkit < 7 || UA.firefox < 3.6) {
 34             return 'notSupportDrop';
 35         }
 36     };
 37 
 38     S.mix(FileDrop, {
 39         event:{
 40             'AFTER_DROP':'afterdrop'
 41         }
 42     });
 43 
 44     S.extend(FileDrop, Base, /** @lends FileDrop.prototype*/ {
 45         /**
 46          * 运行
 47          */
 48         render:function () {
 49 //            console.log('render', this.get('target'));
 50             var self = this,mode = self.get('mode'),
 51                 uploader = self.get('uploader'),
 52                 $dropArea;
 53             if(mode != 'supportDrop'){
 54                 S.log('该浏览器不支持拖拽上传！');
 55                 return false;
 56             }
 57             if(!uploader){
 58                 S.log('缺少Uploader的实例！');
 59                 return false;
 60             }
 61             $dropArea = self._createDropArea();
 62             if($dropArea.length){
 63                 $dropArea.on('click',self._clickHandler,self);
 64             }
 65             console.log('after randeer');
 66             // self.fire('afterRender', {'buttonWrap': self.get('buttonWrap'), 'config': {'tpl' : self.get('btnTpl')}});
 67             self.fire('afterRender', {'buttonTarget':self.get('buttonWrap')});
 68         },
 69         /**
 70          * 显示拖拽区域
 71          */
 72         show:function () {
 73             var self = this,
 74                 dropContainer = self.get('dropContainer');
 75             dropContainer.show();
 76         },
 77         /**
 78          * 隐藏拖拽区域
 79          */
 80         hide:function () {
 81             var self = this,
 82                 dropContainer = self.get('dropContainer');
 83             dropContainer.hide();
 84         },
 85         /**
 86          * ?
 87          */
 88         reset:function () {
 89         },
 90         /**
 91          * 创建拖拽区域
 92          */
 93         _createDropArea:function () {
 94             var self = this,
 95                 target = $(self.get('target')),
 96                 mode = self.get('mode'),
 97                 html = S.substitute(self.get('tpl')[mode], {name:self.get('name')}),
 98                 dropContainer = $(html),
 99                 buttonWrap = dropContainer.all('.J_ButtonWrap');
100             // console.log(buttonWrap);
101             dropContainer.appendTo(target);
102             dropContainer.on('dragover', function (ev) {
103                 ev.stopPropagation();
104                 ev.preventDefault();
105             });
106             dropContainer.on('drop', function (ev) {
107                 ev.stopPropagation();
108                 ev.preventDefault();
109                 self._dropHandler(ev);
110             });
111             self.set('dropContainer', dropContainer);
112             self.set('buttonWrap', buttonWrap);
113             self._setStyle();
114             return dropContainer;
115         },
116         /**
117          * 设置拖拽层样式
118          * @author 明河新增
119          */
120         _setStyle:function(){
121              var self = this,$dropContainer = self.get('dropContainer');
122             if(!$dropContainer.length) return false;
123             $dropContainer.parent().css('position','relative');
124             $dropContainer.css({'position':'absolute','top':'0','left':'0',width:'100%',height:'100%','zIndex':'1000'});
125         },
126         /**
127          * 点击拖拽区域后触发
128          * @author 明河新增
129          * @param ev
130          */
131         _clickHandler:function(ev){
132             var self = this,$target = $(ev.target),uploader = self.get('uploader'),
133                 button = uploader.get('button'),
134                 $input = button.get('fileInput');
135             //触发input的选择文件
136             $input.fire('click');
137         },
138         /**
139          * 处理拖拽时间
140          */
141         _dropHandler:function (ev) {
142             var self = this,
143                 event = FileDrop.event,
144                 fileList = ev.originalEvent.dataTransfer.files,
145                 files = [],
146                 uploader = self.get('uploader');
147 
148             if (!fileList.length || uploader == EMPTY)  return false;
149             S.each(fileList, function (f) {
150                 if (S.isObject(f)) {
151                     files.push({'name':f.name, 'type':f.type, 'size':f.size,'data':f});
152                 }
153             });
154             self.fire(event.AFTER_DROP, {files:files});
155             uploader._select({files:files});
156         },
157         _setDisabled:function () {
158         }
159     }, {
160         ATTRS:/** @lends FileDrop.prototype*/{
161             target:{
162                 value:EMPTY
163             },
164             uploader:{value:EMPTY},
165             dropContainer:{
166                 value:EMPTY
167             },
168             /**
169              * 模板
170              * @type Object
171              * @default {}
172              */
173             tpl:{
174                 value:{
175                     supportDrop:'<div class="drop-wrapper">' +
176                         '<p>直接拖拽图片到这里，</p>' +
177                         '<p class="J_ButtonWrap">或者' +
178                         '</p>' +
179                         '</div>',
180                     notSupportDropIe:'<div class="drop-wrapper">' +
181                         '<p>您的浏览器只支持传统的图片上传，</p>' +
182                         '<p class="suggest J_ButtonWrap">推荐使用chrome浏览器或firefox浏览器' +
183                         '</p>' +
184                         '</div>',
185                     notSupportDrop:'<div class="drop-wrapper">' +
186                         '<p>您的浏览器只支持传统的图片上传，</p>' +
187                         '<p class="suggest J_ButtonWrap">推荐升级您的浏览器' +
188                         '</p>' +
189                         '</div>'
190                 }
191             },
192             name:{
193                 value:'',
194                 setter:function (v) {
195                 }
196             },
197             disabled:{
198                 value:false,
199                 setter:function (v) {
200                     this._setDisabled(v);
201                     return v;
202                 }
203             },
204             cls:{
205                 disabled:'drop-area-disabled'
206             }
207         }
208     });
209 
210     return FileDrop;
211 }, {requires:['node', 'base']});
