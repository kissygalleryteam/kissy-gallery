1 /**
  2  * @fileoverview 图片上传主题（带图片预览），第一版由紫英同学完成，苏河同学做了大量优化，明河整理优化
  3  * @author 苏河、紫英、明河
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/themes/imageUploader/index', function (S, Node, Theme) {
  6     var EMPTY = '', $ = Node.all;
  7 
  8     /**
  9      * @name ImageUploader
 10      * @class 图片上传主题（带图片预览），第一版由紫英同学完成，苏河同学做了大量优化，明河整理优化
 11      * @constructor
 12      * @extends Theme
 13      * @requires Theme
 14      * @requires  ProgressBar
 15      * @author 苏河、紫英、明河
 16      */
 17     function ImageUploader(config) {
 18         var self = this;
 19         //调用父类构造函数
 20         ImageUploader.superclass.constructor.call(self, config);
 21     }
 22 
 23     S.extend(ImageUploader, Theme, /** @lends ImageUploader.prototype*/{
 24         /**
 25          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 26          * @param {Uploader} uploader
 27          */
 28         afterUploaderRender:function (uploader) {
 29             var self = this,
 30                 Preview = self.get('oPlugin').preview,
 31                 preview = new Preview(),
 32                 queue = self.get('queue');
 33             //图片预览
 34             self.set('preview',preview);
 35            //达到最大允许上传数隐藏上传按钮
 36             self._maxHideBtn(uploader);
 37             self._renderFiledrop();
 38             queue.on('add',self._addFileHandler,self);
 39         },
 40         /**
 41          * 在完成文件dom插入后执行的方法
 42          * @param {Object} ev 类似{index:0,file:{},target:$target}
 43          */
 44         _addFileHandler:function(ev){
 45             var self = this,file = ev.file,$target = file.target,$delBtn = $('.J_Del_'+file.id),
 46                 $mask = $('.J_Mask_' + file.id) ;
 47             //显示/隐藏删除按钮
 48             $target.on('mouseover mouseout',function(ev){
 49                 if(ev.type == 'mouseover'){
 50                     $delBtn.show();
 51                     $mask.show();
 52                 }else{
 53                     $delBtn.hide();
 54                     $mask.hide();
 55                 }
 56             });
 57             $delBtn.data('data-file',file);
 58             //点击删除按钮
 59             $delBtn.on('click',self._delHandler,self);
 60         },
 61         /**
 62          * 获取状态容器
 63          * @param {KISSY.NodeList} target 文件的对应的dom（一般是li元素）
 64          * @return {KISSY.NodeList}
 65          */
 66         _getStatusWrapper:function (target) {
 67             return target.children('.J_FileStatus');
 68         },
 69         /**
 70          * 运行文件拖拽插件
 71          * @return {Filedrop}
 72          */
 73         _renderFiledrop:function(){
 74             //文件拖拽支持
 75             var self = this,button = self.get('button'),
 76                 target = button.get('target'),
 77                 Filedrop = self.get('oPlugin').filedrop,
 78             filedrop = new Filedrop({
 79                 target:target,
 80                 uploader:this.get('uploader'),
 81                 tpl:{supportDrop:'<div class="drop-wrapper"></div>' }
 82             });
 83             filedrop.render();
 84             return filedrop;
 85         },
 86         /**
 87          * 文件处于等待上传状态时触发
 88          */
 89         _waitingHandler:function (ev) {
 90             var self = this,preview = self.get('preview'),
 91                 file = ev.file,input = file.input,
 92                 $imageWrapper = $('.J_Pic_'+file.id);
 93             if(preview && input && $imageWrapper.length){
 94                 preview.preview(ev.file.input, $imageWrapper);
 95             }
 96         },
 97         /**
 98          * 文件处于开始上传状态时触发
 99          */
100         _startHandler:function (ev) {
101             var self = this,
102                 uploader = ev.uploader,
103                 index = ev.index,
104                 queue = self.get('queue'),
105                 //上传方式
106                 uploadType = uploader.get('type'),
107                 $progressBar = $('.J_ProgressBar_' + ev.id);
108             //如果是ajax或flash异步上传，加入进度条
109             if(uploadType == 'ajax' || uploadType == 'flash'){
110                 var ProgressBar = self.get('oPlugin').progressBar,progressBar = new ProgressBar($progressBar);
111                 progressBar.render();
112                 self.set('progressBar',progressBar);
113                 //将进度条实例写入到队列的文件数据上备用
114                 queue.updateFile(index,{progressBar:progressBar});
115             }
116         },
117         /**
118          * 文件处于正在上传状态时触发
119          */
120         _progressHandler:function (ev) {
121             var file = ev.file,
122                 //已加载字节数
123                 loaded = ev.loaded,
124                 //总字节数
125                 total = ev.total,
126                 val = Math.ceil(loaded/total) * 100,
127                 progressBar = file.progressBar;
128             if(!progressBar) return false;
129             //处理进度
130             progressBar.set('value',val);
131         },
132         /**
133          * 文件处于上传成功状态时触发
134          */
135         _successHandler:function (ev) {
136             var self = this,
137                 file = ev.file,
138                 //服务器端返回的数据
139                 result = file.result;
140             self._setCount();
141             //获取服务器返回的图片路径写入到src上
142             if(result) self._changeImageSrc(ev.id,result);
143             self._setDisplayMsg(false,ev.file);
144         },
145          /**
146          * 文件处于上传错误状态时触发
147          */
148         _errorHandler:function (ev) {
149             var self = this,msg = ev.msg,
150                 id = ev.id;
151             //打印错误消息
152             $('.J_ErrorMsg_' + id).html(msg);
153              self._setDisplayMsg(true,ev.file);
154              //向控制台打印错误消息
155              S.log(msg);
156         },
157         /**
158          * 显示“你还可以上传几张图片”
159          */
160         _setCount:function(){
161             var self = this,
162                 //用于显示上传数的容器
163                 elCount = $(self.get('elCount')),
164                 len = self.getFilesLen(),
165                 auth = self.get('auth');
166             if(!elCount.length || !auth) return false;
167             var rules = auth.get('rules'),
168                 //max的值类似[5, '最多上传{max}个文件！']
169                 max = rules.max;
170             if(!max) return false;
171             elCount.text(max[0]-len);
172         },
173         /**
174          * 显示/隐藏遮罩层（遮罩层在出现状态消息的时候出现）
175          */
176         _setDisplayMsg:function(isShow,data){
177             if(!data) return false;
178             var $mask = $('.J_Mask_' + data.id),
179                 $statusWrapper = data.statusWrapper;
180             $mask[isShow && 'show' || 'hide']();
181             if(isShow){
182                 $mask.show();
183                 $statusWrapper.show();
184             }else{
185                 $mask.hide();
186                 $statusWrapper.hide();
187             }
188         },
189         /**
190          * 达到最大允许上传数隐藏按钮
191          * @param {Uploader} uploader
192          */
193         _maxHideBtn:function(uploader){
194             //监听上传验证的error事件
195             var self = this,auth = self.get('auth');
196             auth.on('error',function(ev){
197                 var rule = ev.rule,button = uploader.get('button'),$btn = button.get('target');
198                 //图片达到最大允许上传数，隐藏按钮
199                 if(rule == 'max'){
200                     button.hide();
201                     //隐藏按钮之上的li容器
202                     $btn.parent('li').hide();
203                 }
204             })
205         },
206         /**
207          * 删除图片后触发
208          */
209         _delHandler:function(ev){
210              var self = this,uploader = self.get('uploader'),queue = self.get('queue'),
211                  file = $(ev.target).data('data-file'),index = queue.getFileIndex(file.id),
212                  status = file.status;
213             //如果文件还在上传，取消上传
214              if(status == 'start' || status == 'progress'){
215                  uploader.cancel(index);
216              }
217             //统计允许上传文件个数
218             self._setCount();
219         },
220         /**
221          * 获取成功上传的图片张数，不传参的情况获取成功上传的张数
222          * @param {String} status 状态
223          * @return {Number} 图片数量
224          */
225         getFilesLen:function(status){
226             if(!status) status = 'success';
227             var self = this,
228             queue = self.get('queue'),
229             //成功上传的文件数
230             successFiles = queue.getFiles(status);
231             return successFiles.length;
232         },
233         /**
234          * 将服务器返回的图片路径写到预览图片区域，部分浏览器不支持图片预览
235          * @param {String} id  文件id
236          * @param {Object} result  服务器端返回的结果集
237           */
238         _changeImageSrc:function(id,result){
239             var data = result.data,url,
240                 $img = $('.J_Pic_' + id);
241             if(!S.isObject(data)) return false;
242             url = data.url;
243             //不存在预览图片，IE8使用滤镜处理图片预览有问题
244             if($img.attr('src') == EMPTY || S.UA.ie == 8){
245                 $img.attr('src',url);
246             }
247         }
248     }, {ATTRS:/** @lends ImageUploader.prototype*/{
249         /**
250          *  主题名（文件名），此名称跟样式息息相关
251          * @type String
252          * @default "imageUploader"
253          */
254         name:{value:'imageUploader'},
255         /**
256          * 是否引用css文件
257          * @type Boolean
258          * @default true
259          */
260         isUseCss:{value:true},
261         /**
262          * css模块路径
263          * @type String
264          * @default "gallery/form/1.3/uploader/themes/imageUploader/style.css"
265          */
266         cssUrl:{value:'gallery/form/1.3/uploader/themes/imageUploader/style.css'},
267         /**
268          * 队列使用的模板
269          * @type String
270          * @default ""
271          */
272         fileTpl:{value:
273             '<li id="queue-file-{id}" class="clearfix" data-name="{name}">' +
274                 '<div class="tb-pic120">' +
275                     '<a href="javascript:void(0);"><img class="J_Pic_{id}" src="" /></a>' +
276                 '</div>' +
277                 '<div class=" J_Mask_{id} pic-mask"></div>' +
278                 '<div class="status-wrapper J_FileStatus">' +
279                     '<div class="status waiting-status tips-upload-waiting"><p class="tips-text">等待上传，请稍候</p></div>' +
280                     '<div class="status start-status progress-status tips-uploading">' +
281                         '<div class="J_ProgressBar_{id}"><s class="loading-icon"></s>上传中...</div>' +
282                     '</div>' +
283                     '<div class="status success-status tips-upload-success">' +
284                       '上传成功！' +
285                     '</div>' +
286                     '<div class="status error-status tips-upload-error">' +
287                         '<p class="J_ErrorMsg_{id} tips-text">上传失败，请重试！</p></div>' +
288                 '</div>' +
289                 '<a class="J_Del_{id} del-pic" href="#">删除</a>' +
290             '</li>'
291         },
292         /**
293          * 需要加载的插件，需要手动实例化
294          * @type Array
295          * @default ['preview','progressBar','filedrop'] 图片预览、进度条、文件拖拽
296          */
297         plugins:{
298           value:['preview','progressBar','filedrop']
299         },
300         /**
301          * 统计上传张数的容器
302          * @type KISSY.NodeList
303          * @default '#J_UploadCount'
304          */
305         elCount:{value:'#J_UploadCount'}
306     }});
307     return ImageUploader;
308 }, {requires:['node', '../../theme']});
