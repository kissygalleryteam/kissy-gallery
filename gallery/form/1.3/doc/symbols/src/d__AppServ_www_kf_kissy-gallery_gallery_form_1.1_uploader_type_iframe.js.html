1 /**
  2  * @fileoverview iframe方案上传
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/type/iframe',function(S, Node, UploadType) {
  6     var EMPTY = '',$ = Node.all,LOG_PREFIX = '[uploader-iframeType]:',ID_PREFIX = 'ks-uploader-iframe-';
  7 
  8     /**
  9      * @name IframeType
 10      * @class iframe方案上传，全浏览器支持
 11      * @constructor
 12      * @extends UploadType
 13      * @param {Object} config 组件配置（下面的参数为配置项，配置会写入属性，详细的配置说明请看属性部分）
 14       *
 15      */
 16     function IframeType(config) {
 17         var self = this;
 18         //调用父类构造函数
 19         IframeType.superclass.constructor.call(self, config);
 20     }
 21 
 22     S.mix(IframeType, /**@lends IframeType*/ {
 23         /**
 24          * 会用到的html模板
 25          */
 26         tpl : {
 27             IFRAME : '<iframe src="javascript:false;" name="{id}" id="{id}" border="no" width="1" height="1" style="display: none;" />',
 28             FORM : '<form method="post" enctype="multipart/form-data" action="{action}" target="{target}">{hiddenInputs}</form>',
 29             HIDDEN_INPUT : '<input type="hidden" name="{name}" value="{value}" />'
 30         },
 31         /**
 32          * 事件列表
 33          */
 34         event : S.mix(UploadType.event,{
 35               //创建iframe和form后触发
 36             CREATE : 'create',
 37             //删除form后触发
 38             REMOVE : 'remove'
 39         })
 40     });
 41     //继承于Base，属性getter和setter委托于Base处理
 42     S.extend(IframeType, UploadType, /** @lends IframeType.prototype*/{
 43         /**
 44          * 上传文件
 45          * @param {HTMLElement} fileInput 文件input
 46          */
 47         upload : function(fileInput) {
 48             var self = this,$input = $(fileInput),form;
 49             if (!$input.length) return false;
 50             self.fire(IframeType.event.START, {input : $input});
 51             self.set('fileInput', $input);
 52             //创建iframe和form
 53             self._create();
 54             form = self.get('form');
 55             //提交表单到iframe内
 56             form.getDOMNode().submit();
 57         },
 58         /**
 59          * 停止上传
 60          * @return {IframeType}
 61          */
 62         stop : function() {
 63             var self = this,iframe = self.get('iframe');
 64             iframe.attr('src', 'javascript:"<html></html>";');
 65             self.fire(IframeType.event.STOP);
 66             self.fire(IframeType.event.ERROR, {status : 'abort',msg : '上传失败，原因：abort'});
 67             return self;
 68         },
 69         /**
 70          * 将参数数据转换成hidden元素
 71          * @param {Object} data 对象数据
 72          * @return {String} hiddenInputHtml hidden元素html片段
 73          */
 74         dataToHidden : function(data) {
 75             if (!S.isObject(data) || S.isEmptyObject(data)) {
 76                 S.log(LOG_PREFIX + 'data参数不是对象或者为空！');
 77                 return false;
 78             }
 79             var self = this,hiddenInputHtml = EMPTY,
 80                 //hidden元素模板
 81                 tpl = self.get('tpl'),hiddenTpl = tpl.HIDDEN_INPUT;
 82             if (!S.isString(hiddenTpl)) return false;
 83             for (var k in data) {
 84                 hiddenInputHtml += S.substitute(hiddenTpl, {'name' : k,'value' : data[k]});
 85             }
 86             return hiddenInputHtml;
 87         },
 88         /**
 89          * 创建一个空的iframe，用于文件上传表单提交后返回服务器端数据
 90          * @return {NodeList}
 91          */
 92         _createIframe : function() {
 93             var self = this,
 94                 //iframe的id
 95                 id = self.get('id'),
 96                 //iframe模板
 97                 tpl = self.get('tpl'),iframeTpl = tpl.IFRAME,
 98                 existIframe = self.get('iframe'),
 99                 iframe,$iframe;
100             //先判断是否已经存在iframe，存在直接返回iframe
101             if (!S.isEmptyObject(existIframe)) return existIframe;
102             if (!S.isString(iframeTpl)) {
103                 S.log(LOG_PREFIX + 'iframe的模板不合法！');
104                 return false;
105             }
106             if (!S.isString(id)) {
107                 S.log(LOG_PREFIX + 'id必须存在且为字符串类型！');
108                 return false;
109             }
110             //创建处理上传的iframe
111             iframe = S.substitute(tpl.IFRAME, { 'id' : id });
112             $iframe = $(iframe);
113             //监听iframe的load事件
114             $iframe.on('load', self._iframeLoadHandler, self);
115             $('body').append($iframe);
116             self.set('iframe', $iframe);
117             return $iframe;
118         },
119         /**
120          * iframe加载完成后触发（文件上传结束后）
121          */
122         _iframeLoadHandler : function(ev) {
123             var self = this,iframe = ev.target,
124                 errorEvent = IframeType.event.ERROR,
125                 doc = iframe.contentDocument || window.frames[iframe.id].document,
126                 result;
127             if (!doc || !doc.body) {
128                 self.fire(errorEvent, {msg : '服务器端返回数据有问题！'});
129                 return false;
130             }
131             result = doc.body.innerHTML;
132             S.log(LOG_PREFIX + '服务器端输出:'+result);
133             //如果不存在json结果集，直接退出
134             if (result == EMPTY) return false;
135             try {
136                 result = JSON.parse(result);
137             } catch(err) {
138                 S.log(LOG_PREFIX + 'json数据格式不合法！');
139                 self.fire(errorEvent, {msg : '数据：' + result + '不是合法的json数据'});
140             }
141             self.fire(IframeType.event.SUCCESS, {result : result});
142             self._remove();
143         },
144         /**
145          * 创建文件上传表单
146          * @return {NodeList}
147          */
148         _createForm : function() {
149             var self = this,
150                 //iframe的id
151                 id = self.get('id'),
152                 //form模板
153                 tpl = self.get('tpl'),formTpl = tpl.FORM,
154                 //想要传送给服务器端的数据
155                 data = self.get('data'),
156                 //服务器端处理文件上传的路径
157                 action = self.get('action'),
158                 fileInput = self.get('fileInput'),
159                 hiddens,form = EMPTY,$form;
160             if (!S.isString(formTpl)) {
161                 S.log(LOG_PREFIX + 'form模板不合法！');
162                 return false;
163             }
164             if (!S.isObject(data)) {
165                 S.log(LOG_PREFIX + 'data参数不合法！');
166                 return false;
167             }
168             if (!S.isString(action)) {
169                 S.log(LOG_PREFIX + 'action参数不合法！');
170                 return false;
171             }
172             hiddens = self.dataToHidden(data);
173             if (hiddens == EMPTY) return false;
174             form = S.substitute(formTpl, {'action' : action,'target' : id,'hiddenInputs' : hiddens});
175             //克隆文件域，并添加到form中
176             $form = $(form).append(fileInput);
177             $('body').append($form);
178             self.set('form', $form);
179             return $form;
180         },
181         /**
182          * 创建iframe和form
183          */
184         _create : function() {
185             var self = this,
186                 iframe = self._createIframe(),
187                 form = self._createForm();
188             self.fire(IframeType.event.CREATE, {iframe : iframe,form : form});
189         },
190         /**
191          * 移除表单
192          */
193         _remove : function() {
194             var self = this,form = self.get('form'),iframe = self.get('iframe');
195             //移除表单
196             form.remove();
197             //重置form属性
198             self.reset('form');
199             self.fire(IframeType.event.REMOVE, {form : form});
200         }
201     }, {ATTRS : /** @lends IframeType.prototype*/{
202         /**
203          * iframe方案会用到的html模板，一般不需要修改
204          * @type {}
205          * @default
206          * {
207          IFRAME : '<iframe src="javascript:false;" name="{id}" id="{id}" border="no" width="1" height="1" style="display: none;" />',
208          FORM : '<form method="post" enctype="multipart/form-data" action="{action}" target="{target}">{hiddenInputs}</form>',
209          HIDDEN_INPUT : '<input type="hidden" name="{name}" value="{value}" />'
210          }
211          */
212         tpl : {value : IframeType.tpl},
213         /**
214          * 只读，创建的iframeid,id为组件自动创建
215          * @type String
216          * @default  'ks-uploader-iframe-' +随机id
217          */
218         id : {value : ID_PREFIX + S.guid()},
219         /**
220          * iframe
221          */
222         iframe : {value : {}},
223         form : {value : {}},
224         fileInput : {value : EMPTY}
225     }});
226 
227     return IframeType;
228 }, {requires:['node','./base']});
